88 Expr::set(MarSystem* m, Ex& e, Rp& r)
89 {

91  timer_=NULL;
92  sched_=NULL;
93  symbol_table_=new ExRecord();
94  symbol_table_->inc_ref();
95  e.parse(this,init_expr_,expr_);
96  r.parse(this,rept_,rate_);
97  if (rept_&&rept_->getEvalType()!="mrs_bool") {
98  MRSWARN("Expr:: Repetition expression must evaluate to bool: "+rept_->getEvalType());
99  rept_->deref(); rept_=NULL;
100  if (rate_) rate_->deref(); rate_=NULL;
101  }
102  else if (rate_&&rate_->getEvalType()!="mrs_string") {
103  MRSWARN("Expr:: Repetition rate expression must evaluate to string: "+rate_->getEvalType());
104  if (rept_) rept_->deref(); rept_=NULL;
105  if (rate_) rate_->deref(); rate_=NULL;
106  }
107  initialized_=false;
108 }

159 void
160 ExFile::read(std::string fname)
161 {
162  std::ifstream from(fname.c_str());
163  if (!from) { MRSWARN("ExFile::read Cannot open file: "+fname); return; }
164  std::string data;
165  char buffer[256];
166  // flags correspond to the 5 possible blocks, no block can be declared more than once
167  bool flags[6] = { false };
168  int pos=-1;
169  std::string line;
170  while (from.getline(buffer,256)) {
171  if (buffer[0]=='#') {
172  if (buffer[1]=='E'&&buffer[2]=='x') { // #ExInit: | #ExExpr:
173  if (buffer[3]=='I'&&buffer[4]=='n'&&buffer[5]=='i'&&buffer[6]=='t'&&buffer[7]==':') {
174  store(pos,data); data=""; pos=1;
175  if (flags[pos]) { MRSWARN("ExFile::read Double declaration of #ExInit: block"); }
176  }
177  else if (buffer[3]=='E'&&buffer[4]=='x'&&buffer[5]=='p'&&buffer[6]=='r'&&buffer[7]==':') {
178  store(pos,data); data=""; pos=2;
179  if (flags[pos]) { MRSWARN("ExFile::read Double declaration of #ExExpr: block"); }
180  }
181  }
182  else if (buffer[1]=='R'&&buffer[2]=='p') {
183  if (buffer[3]=='E'&&buffer[4]=='x'&&buffer[5]=='p'&&buffer[6]=='r'&&buffer[7]==':') {
184  // #RpExpr:
185  store(pos,data); data=""; pos=3;
186  if (flags[pos]) { MRSWARN("ExFile::read Double declaration of #RpExpr: block"); }
187  }
188  else if (buffer[3]=='R'&&buffer[4]=='a'&&buffer[5]=='t'&&buffer[6]=='e'&&buffer[7]==':') {
189  // #RpRate:
190  store(pos,data); data=""; pos=4;
191  if (flags[pos]) { MRSWARN("ExFile::read Double declaration of #RpRate: block"); }
192  }
193  }
194  else {
195  MRSWARN("ExFile::read Unknown macro # in ExFile");
196  }
197  flags[pos]=true;
198  }
199  else { data=data+buffer; }
200  }
201  if (!data.empty()) { store(pos,data); }
202  from.close();
203  file_read_=true;
204 }
