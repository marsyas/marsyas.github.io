162  for (int i=0; i < tempo_candidates.getSize()/2; i++)
163  {
164  tempos(i) = tempo_candidates(2*i+1) / factor_;
165  tempo_scores(i) = tempo_candidates(2*i);
166  }
167 
168  // normalize to pdf
169  tempo_scores /= tempo_scores.sum();
170 
171  // holds the tempo scores based on cross-correlation with pulse train
172  mrs_realvec onset_scores;
173  onset_scores.create(tempo_scores.getSize());
174  // holds the best matching phase for each tempo candidate
175  mrs_realvec tempo_phases;
176  tempo_phases.create(tempo_scores.getSize());
177 
178  /*
179  // make sure the tempo candidates are reasonable
180  for (int k=0; k < tempos.getSize(); k++)
181  {
182  if (tempos(k) < 50.0)
183  tempos(k) = 0;
184  if (tempos(k) > 200)
185  tempos(k) = 0;
186 
187  }
188  */
189 
190  // The winSize and hopSize of the onset strength function
191  // needed to output correct beat location times
192  // mrs_natural bwinSize = ctrl_bwinSize_->to<mrs_natural>();
193  mrs_natural bhopSize = ctrl_bhopSize_->to<mrs_natural>();
194 
195  MarControlAccessor acc(ctrl_beats_);
196  mrs_realvec& beats = acc.to<mrs_realvec>();
197 
198 
199  for (o=0; o < inObservations_; o++)
200  {
201  for (t = 0; t < inSamples_; t++)
202  {
203  out (o,t) = in(o,t);
204  beats (o,t) = 0.0;
205  }
206  }
207 
208  mrs_real tempo;
209  mrs_real period;
210  mrs_natural phase;
211  mrs_real cross_correlation = 0.0;
212  mrs_real max_crco=0.0;
213  mrs_natural max_phase = 0;
214  mrs_realvec phase_correlations;
215 
216 
217  // loop for cross-correlating
218  // pulse trains with onset strength function
219  // for each tempo shift the pulse train until the best match is found
220  // tempo_scores holds the values of the cross-correlation
221  // and onset_scores holds the variance of cross-correlation among different phases
222  // for a particular tempo candidate
223  for (o=0; o < inObservations_; o++)
224  {
225  for (int k=0; k < tempos.getSize(); k++)
226  {
227  max_crco = 0.0;
228 
229  tempo = tempos(k);
230  period = 2.0 * osrate_ * 60.0 / tempo; // flux hopSize is half the winSize
231  mrs_natural period_int = (mrs_natural)(period+0.5);
232 
233 
234  if (period_int > 1)
235  {
236  //cout<<tempo<<"\t"<<period_int;
237  phase_correlations.create( period_int );
238 
239  for (phase=inSamples_-1; phase > inSamples_-1-period_int; phase--)
240  {
241  cross_correlation = 0.0;
242  // correlate with pulse train with half-beats and double beats
243  for (int b=0; b < 4; b++)
244  {
245  mrs_natural temp_t;
246  temp_t = phase - b * period_int;
247 
248  // 4 beats
249  if (temp_t >= 0) {
250  cross_correlation += in(o, temp_t);
251  //cout<<"\t"<<temp_t;
252  }
253 
254  // slow down by 2.0
255  temp_t = phase - b * period_int * 2;
256  if (temp_t >= 0) {
257  cross_correlation += 0.5 * in(o, temp_t);
258  //cout<<"\t"<<temp_t;
259  }
260 
261  // slow down by 3
262  temp_t = phase - b * period_int * 3 / 2;
263  if (temp_t >= 0) {
264  cross_correlation += 0.5 * in(o, temp_t);
265  //cout<<"\t"<<temp_t;
266  }
267  //cout<<endl;
268 
269 
270  }
271 
272  // quarter beats
273  /* for (int b = 0; b < 8; b++)
274  {
275  mrs_natural temp_t;
276  temp_t = phase - b * 0.25 * period;
277  if (temp_t >= 0)
278  {
279  cross_correlation += (0.2 * in(o, temp_t));
280  }
281  }
282  */
283 
284  phase_correlations(inSamples_-1-phase) = cross_correlation;
285  if (cross_correlation > max_crco)
286  {
287  max_crco = cross_correlation;
288  max_phase = phase;
289  }
290 
291  }
292  onset_scores(k) = phase_correlations.var();
293  tempo_scores(k) = max_crco;
294  tempo_phases(k) = max_phase;
295  beats.setval(0.0);
296 
297  //printf("\t%f\t%f", max_crco, onset_scores(k));
298  //cout<<endl;
299  }
300  }
301  }
302 
303 
304  // renormalize scores
305  onset_scores /= onset_scores.sum();
306  tempo_scores /= tempo_scores.sum();
307 
308  // combine the cross-correlation score with the variance
309  for (int i=0; i < tempo_scores.getSize(); i++)
310  tempo_scores(i) = onset_scores(i) + tempo_scores(i);
311 
312  // renormalize
313  tempo_scores /= tempo_scores.sum();
314 
315  // pick the maximum scoring tempo candidate
316  mrs_real max_score = 0.0;
317  int max_i=0;
318  for (int i= 0; i < tempos.getSize(); i++)
319  {
320  if (tempo_scores(i) > max_score)
321  {
322  max_score = tempo_scores(i);
323  max_i = i;
324  }
325  }
326 
327  // return the best tempo candidate and the
328  // corresponding score in both the tempo vector
329  // as well as the control phase_tempo
330  mrs_real swap_tempo = tempos(0);
331  mrs_real swap_score = tempo_scores(0);
332  tempos(0) = tempos(max_i);
333  tempo_scores(0) = tempo_scores(max_i);
334 
335  if (max_i != 0) {
336  tempos(max_i) = swap_tempo;
337  tempo_scores(max_i) = swap_score;
338  }
339 
340  max_score = 0.0;
341  max_i=0;
342  for (int i= 1; i < tempos.getSize(); i++)
343  {
344  if (tempo_scores(i) > max_score)
345  {
346  max_score = tempo_scores(i);
347  max_i = i;
348  }
349  }
350 
351  swap_tempo = tempos(1);
352  swap_score = tempo_scores(1);
353  tempos(1) = tempos(max_i);
354  tempo_scores(1) = tempo_scores(max_i);
355  if (max_i != 1) {
356  tempos(max_i) = swap_tempo;
357  tempo_scores(max_i) = swap_score;
358  }
359 
360  ctrl_phase_tempo_->setValue(tempos(max_i));
361 
362 
363  // select a tempo for the beat locations
364  // with doubling heuristic if tempo < 75 BPM
365  tempo = tempos(0);
366  if (tempo < 70.0)
367  tempo = tempo * 2;
368  mrs_real beat_length = 60.0 / tempo;
369 
370  if (tempo >= 50)
371  period = 2.0 * osrate_ * 60.0 / tempo; // flux hopSize is half the winSize
372  else
373  period = 0;
374 
375  period = (mrs_natural)(period+0.5);
376  // Place the beats in the right location in the onset detection function
377  for (int b=0; b < 4; b++) {
378  mrs_natural temp_t = (mrs_natural) (tempo_phases(max_i) - b * period);
379  if (temp_t >= 0) {
380  beats(0,temp_t) = -0.5;
381  }
382  }
383 
384 
385  //mrs_natural prev_sample_count;
386  //prev_sample_count = sampleCount_;
387 
388  // Output all the detected beats to it's own MarControl
389  int total_beats = 0;
390  MarControlAccessor beatOutputAcc(ctrl_beatOutput_);
391  mrs_realvec& beatOutput = beatOutputAcc.to<mrs_realvec>();
392  for (t = 0; t < inSamples_; t++)
393  {
394  beatOutput(t) = 0.0;
395  }
396 
397  // output the beats
398  mrs_real beat_location;
399  for (int t = inSamples_-1-2*bhopSize; t < inSamples_; t++)
400  {
401  if (beats(0,t) == -0.5)
402  {
403  beat_location = (sampleCount_ + t -(inSamples_-1 -bhopSize)) / (2.0 * osrate_);
404  if ((beat_location > current_beat_location_)&&((beat_location - current_beat_location_) > beat_length * 0.75))
405  {
406  if (ctrl_verbose_->isTrue()) {
407  MRSMSG(beat_location << "\t"
408  << beat_location + 0.02 << " b");
409  }
410  beatOutput(total_beats) = beat_location;
411  current_beat_location_ = beat_location;

413  }
414  }
415  }
