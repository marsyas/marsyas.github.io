78  // The output from PZFC (and HCL) is a realvec with the first half
79  // of observations being the signal, and the second half being the

96  }

146 void
147 AimLocalMax::myProcess(realvec& in, realvec& out)
148 {
149  mrs_natural o,t;
150 
151  // sness - Need this because we don't have a SignalBuffer class like AIM-C has, so we
152  // have to keep track of the strobes ourselves.
153  strobe_count_.clear();
154  strobe_count_.resize(channel_count_, 0.0);
155  last_strobe_.clear();
156  last_strobe_.resize(channel_count_, 0); // vector of ints
157 
158  // sness - Hmm, the original code isn't doing this, but I don't see
159  // how this could possibly work if you don't reset these samples at
160  // the beginning of a buffer. Otherwise, you get carryover from the
161  // last step that causes the line after the condition
162  // "strobe_count_[o] > 0" to coredump. Requires further
163  // investigation.
164  prev_sample_.clear();
165  prev_sample_.resize(channel_count_, 10000.0);
166  curr_sample_.clear();
167  curr_sample_.resize(channel_count_, 5000.0);
168  next_sample_.clear();
169  next_sample_.resize(channel_count_, 0.0);
170 
171  // Skip over the signals and centre frequencies from PZFC (and HCL)
172  mrs_natural skip_channels = channel_count_ + channel_count_;
173 
174  for (t = 0; t < ctrl_inSamples_->to<mrs_natural>(); t++) {
175  for (o = 0; o < channel_count_; o++) {
176  // Initialize the strobe
177  out(o + skip_channels,t) = 0.0;
178 
179  // curr_sample is the sample at time (i - 1)
180  prev_sample_[o] = curr_sample_[o];
181  curr_sample_[o] = next_sample_[o];
182  next_sample_[o] = in(o, t);
183 
184  // If the current sample is above threshold, the threshold is raised to
185  // the level of the current sample, and decays from there.
186  if (curr_sample_[o] >= threshold_[o]) {
187  threshold_[o] = curr_sample_[o];
188  decay_constant_[o] = threshold_[o] / strobe_decay_samples_;
189  // If the current sample is also a peak, then it is a potential strobe
190  // point.
191  if (prev_sample_[o] < curr_sample_[o]
192  && next_sample_[o] < curr_sample_[o]) {
193  // If there are no strobes so far in this channel, then the sample
194  // is definitely a strobe (this means that the timeout is not
195  // respected across frame boundaries. This is a minor bug, but I
196  // don't believe that it's serious enough to warrant updating the
197  // samples since last strobe all the time.)
198  if (strobe_count_[o] > 0) {
199  // If there are previous strobes, then calculate the time since
200  // the last one. If it's long enough, then this is a strobe point,
201  // if not, then just move on.
202  int samples_since_last = (t - 1) - last_strobe_[o];
203  if (samples_since_last > strobe_timeout_samples_) {
204  out(o + skip_channels, t-1) = 1.0;
205  strobe_count_[o]++;
206  last_strobe_[o] = t;
207  }
208  } else {
209  out(o + skip_channels, t-1) = 1.0;
210  strobe_count_[o]++;
211  last_strobe_[o] = t;
212  }
213  }
214  }
215 
216  // Update the threshold, decaying as necessary
217  if (threshold_[o] > decay_constant_[o])
218  threshold_[o] -= decay_constant_[o];
219  else
220  threshold_[o] = 0.0;
221  }
222  }
223 
224  // Now that we've added strobes as a second set of observations
225  // after the first, copy the the data from the input to the output.
226  for (t = 0; t < ctrl_inSamples_->to<mrs_natural>(); t++) {
227  for (o = 0; o < skip_channels; o++) {
228  out(o,t) = in(o,t);
229  }
230  }
231 }
