150 TranscriberExtract::getPitchesFromRealvecSink(MarSystem* rvSink,
151  const mrs_real srate)
152 {
153  realvec data = rvSink->getctrl("mrs_realvec/data")->to<mrs_realvec>();
154  rvSink->updControl("mrs_bool/done", true);
155 
156  realvec pitchList(data.getSize()/2);
157  mrs_real pitchOutput;
158  for (mrs_natural i=0; i<pitchList.getSize(); ++i)
159  {
160  // on linux (but not OSX), we have pitchOutput of 0.5 if the pitch
161  // detection can't decide on a pitch.
162  pitchOutput = data(2*i+1);
163  if (pitchOutput > 1)
164  pitchList(i) = samples2hertz( pitchOutput, srate);
165  else
166  pitchList(i) = 0;
167  }
168  return pitchList;
