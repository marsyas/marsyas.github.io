84 void
85 Spectrum2Mel::myUpdate(MarControlPtr sender)
86 {
87  mrs_natural t,o;
88  (void) sender; //suppress warning of unused parameter(s)
89 
90  ctrl_onSamples_->setValue(ctrl_inSamples_, NOUPDATE);
91  ctrl_onObservations_->setValue(ctrl_melBands_, NOUPDATE);
92  ctrl_osrate_->setValue(ctrl_israte_, NOUPDATE);
93 
94  // allocate memory for melMap_
95  // I'm not familiar with this algorithm so this might be
96  // wrong! I'm just closing some memory faults! -gp
97  // size based on its usage in ::myProcess
98  onObservations_ = ctrl_onObservations_->to<mrs_natural>();
99 
100 
101  if (pmelBands_ != ctrl_melBands_->to<mrs_natural>())
102  {
103  pmelBands_ = ctrl_melBands_->to<mrs_natural>();
104  ostringstream oss;
105  for (mrs_natural n=0; n < pmelBands_; n++)
106  {
107  oss << "MelBand_" << n << ",";
108  }
109  ctrl_onObsNames_->setValue(oss.str(), NOUPDATE);
110  }
111 
113  // calculate the Mel map
114  // based in the fft2melmx.m MATLAB script by Dan Ellis
115  // http://labrosa.ee.columbia.edu/projects/coversongs/
117  if(tinObservations_ != inObservations_ ||
118  tonObservations_ != onObservations_ ||
119  tisrate_ != israte_ ||
120  pbandWidth_ != ctrl_bandWidth_->to<mrs_real>() ||
121  pbandLowEdge_ != ctrl_bandLowEdge_->to<mrs_real>() ||
122  pbandHighEdge_ != ctrl_bandHighEdge_->to<mrs_real>() ||
123  phtkMel_ != ctrl_htkMel_->to<mrs_bool>() ||
124  pconstAmp_ != ctrl_constAmp_->to<mrs_bool>())
125  {
126  melMap_.allocate(onObservations_, inObservations_);
127 
128  mrs_natural nfilts = ctrl_melBands_->to<mrs_natural>();
129  bool htkmel = ctrl_htkMel_->to<mrs_bool>();
130  mrs_natural N2 = inObservations_;// we get N/2+1 spectrum points at the input...
131  mrs_natural N = (N2-1)*2; //fft size
132 
133  //get the original audio sampling rate
134  mrs_real srate = israte_*N;
135 
136  if(ctrl_bandHighEdge_->to<mrs_real>() == -1.0)
137  ctrl_bandHighEdge_->setValue(srate/2.0, NOUPDATE);
138 
139  pbandWidth_ = ctrl_bandWidth_->to<mrs_real>();
140  pbandLowEdge_ = ctrl_bandLowEdge_->to<mrs_real>();
141  pbandHighEdge_ = ctrl_bandHighEdge_->to<mrs_real>();
142  phtkMel_ = ctrl_htkMel_->to<mrs_bool>();
143  pconstAmp_ = ctrl_constAmp_->to<mrs_bool>();
144 
145  //calculate the frequencies (in Hz) for each spectrum bin
146  realvec fftfreqs(N2);
147  for(o=0; o < N2; ++o) {
148  fftfreqs(o) = o / (float)N * srate;
149  }
150 
151  // 'Center freqs' of mel bands - uniformly spaced between limits
152  mrs_real minmel = hertz2mel(ctrl_bandLowEdge_->to<mrs_real>(), htkmel);
153  mrs_real maxmel = hertz2mel(ctrl_bandHighEdge_->to<mrs_real>(), htkmel);
154  realvec binfrqs(nfilts+2);
155  realvec binbin(nfilts+2);
156  for(t=0; t < nfilts+2; ++t)
157  {
158  binfrqs(t) = mel2hertz(minmel + t/(nfilts+1.0)*(maxmel-minmel) , htkmel);
159  binbin(t) = floor(binfrqs(t)/srate*(N-1) + 0.5);
160  }
161 
162  realvec fs(3);
163  mrs_real width = ctrl_bandWidth_->to<mrs_real>();
164  realvec loslope(N2);
165  realvec hislope(N2);
166 
167  for(mrs_natural i=0; i < nfilts; ++i)
168  {
169  for(t=0; t<3; ++t) {
170  fs(t) = binfrqs(i+t);
171  }
172  //cout<<endl;
173 
174  //scale by width
175  for(t=0; t<3; ++t)
176  fs(t) = fs(1)+ width*(fs(t) - fs(1));
177 
178  //lower and upper slopes for all bins
179  for(t=0; t < N2; ++t)
180  {
181  loslope(t) = (fftfreqs(t) - fs(0)) / (fs(1) - fs(0));
182  hislope(t) = (fs(2) - fftfreqs(t)) / (fs(2) - fs(1));
183  }
184 
185  // .. then intersect them with each other and zero
186  for(t=0; t < N2; ++t)
187  melMap_(i,t) = max((mrs_real)0.0, (mrs_real)min(loslope(t), hislope(t)));
188  }
189 
190  if(ctrl_constAmp_->to<mrs_bool>() == false)
191  {
192  //Slaney-style mel is scaled to be approx constant E per channel
193  mrs_real diagMatrix;
194  for(o = 0; o < nfilts; ++o)
195  {
196  diagMatrix = 2.0 / (binfrqs(o+2) - binfrqs(o));
197  for(t=0; t< N2; ++t)
198  melMap_(o,t) *= diagMatrix;
199  }
200  }
201  }
202 }
