62 void APDelayOsc::myUpdate(MarControlPtr sender)
63 {
64  MRSDIAG("APDelayOsc.cpp - APDelayOsc:myUpdate");
65 
66  frequency_ = (getctrl("mrs_real/frequency")->to<mrs_real>());
67  noteon_ = (getctrl("mrs_bool/noteon")->to<mrs_bool>());
68  type_ = (getctrl("mrs_natural/type")->to<mrs_natural>());
69 
70  israte_ = (getctrl("mrs_real/israte")->to<mrs_real>());
71 
72  // loweset frequency on a piano is 27.5Hz ... sample rate/27.5 for commute
73  // this is the longest delay line required
74  if (delaylineSize_ == 0)
75  {
76  delaylineSize_ = (mrs_natural) (israte_/10);
77  delayline_.create((mrs_natural)delaylineSize_);
78 
79  for (mrs_natural t = 0; t < delaylineSize_; t++)
80  {
81  delayline_(t) = 0.0;
82  }
83  }
84 
85  if (noteon_)
86  {
87  // Allpass coefficient
88  // Chosen through experimentation
89  //ap1.delay(0.7);
90 
91  // The amount of delay to generate the correct pitch
92  mrs_real d = (israte_/frequency_);
93  N_ = (mrs_natural)floor(d);
94  frac_ = d - N_;
95  ap2.delay(frac_);
96 
97  // Choses the differences between generating saw
98  // or square waves.
99  switch (type_)
100  {
101  case 0: // Saw
102  dc_ = frequency_ / israte_;
103  neg_ = 1;
104  break;
105  case 1: // Square
106  dc_ = 0;
107  neg_ = -1;
108  N_ = N_/2; // Because this is bipolar
109  // we need to halve the period
110  break;
111  }
112 
113  // Initialize the delay line
114  for (mrs_natural t = 0; t < N_; t++)
115  {
116  delayline_(t) = 0;
117  }
118  // initialize the system with an impulse
119  delayline_(1) = 0.95;
120 
121  // Initialize our read and write pointers
122  rp_ = 1;
123  rpp_ = 0; // Not used, but could be used to do pitch shifting
124  wp_ = N_-1;
125  }
127  MarSystem::myUpdate(sender);
128 }

130 void APDelayOsc::myProcess(realvec& in, realvec& out)
131 {
132  (void) in;
133 
134  mrs_natural t;
135  mrs_real y, x = 0;
136 
137  for (t = 0; t < inSamples_; t++)
138  {
139  if (noteon_)
140  {
141  // rp holds the current sample
142  x = delayline_(rp_);
143 
144  y = ap2(ap1(x));
145 
146  // Write current sample back into the delay buffer.
147  // If square each sample is negated.
148  delayline_(wp_) = y * neg_;
149 
150  // The leaky Integrator is used to apply an
151  // exponential decay as frequencies increase.
152 
153  out(0,t) = dcb(le1(y - dc_));
154 
155  // Increment the delay line pointers

157  rp_ = (rp_ + 1) % N_;
158 
159  }
