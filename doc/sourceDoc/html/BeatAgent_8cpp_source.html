252

255  phase_ = (mrs_natural) agentControl_(myIndex_, 2);
256  periodFraction_ = ((mrs_real)period_ / (mrs_real)maxPeriod_);
257 
258  outterWinLft_ = (mrs_natural) ceil(period_ * lftOutterMargin_); //(% of IBI)
259  outterWinRgt_ = (mrs_natural) ceil(period_ * rgtOutterMargin_); //(% of IBI)

261  //innerWin_ = (mrs_natural) min(4.0, ceil(period_ * innerMargin_));

305

311  max = in(t);

313  }
314  }
315

318

321  //Classification of last beat based on above evaluation:
322 
323  //cout << identity_ << " -> Score: " << score_ << endl;

325  score_ = calcDScoreCorrSquare(in);
326
