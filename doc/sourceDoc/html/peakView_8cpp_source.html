53 mrs_natural
54 peakView::getFrameNumPeaks(const mrs_natural frame, const mrs_natural group) const
55 {
56  if(group==-1)//ignore group information
57  {
58  //count the number of detected peaks in the frame
59  //
60  //"peaks" with freq == 0 are considered non-existent
61  //so just count peaks until we get the first "peak" with freq!=0
62  mrs_natural p;
63  for(p=0; p < frameMaxNumPeaks_; ++p)
64  {
65  if((*this)(p, pkFrequency, frame) == 0)
66  return p;
67  }
68  return frameMaxNumPeaks_;
69  }
70  else //count peaks from "group" that exist in "frame"
71  {
72  mrs_natural numPeaks = 0;
73  for(mrs_natural p = 0; p < frameMaxNumPeaks_; ++p)
74  {
75  if((*this)(p, pkFrequency, frame) == 0)
76  return numPeaks;
77 
78  if((*this)(p, pkGroup, frame) == group)
79  numPeaks++;
80  }
81  return numPeaks;
82  }
83 }

190 void
191 peakView::toTable(realvec& vecTable)
192 {
193  //resize and initialize the table (assuming the largest possible number of peaks + the header for now...)
194  vecTable.create(frameMaxNumPeaks_*numFrames_+1, nbPkParameters);
195 
196 
197  //In Table format, the 1st row is a "header row"
198  vecTable(0, 0) = -1;
199  vecTable(0, 1) = fs_;
200  vecTable(0, 2) = frameSize_;
201  vecTable(0, 3) = frameMaxNumPeaks_;
202  vecTable(0, 4) = numFrames_;
203  vecTable(0, 5) = -1;
204  vecTable(0, pkGroup) = -2;
205  for (mrs_natural i = pkGroup+1 ; i < nbPkParameters ; ++i) //i = pkGroup or i = pkGroup+1 [?]
206  vecTable(0, i)=0;
207 
208 
209  //fill the table with peak data
210  mrs_natural l = 1; //l = peak index for table format (i.e. row index)
211  for (mrs_natural j=0 ; j < vec_.getCols(); ++j) //j = frame index
212  for (mrs_natural i=0 ; i < frameMaxNumPeaks_; ++i) //i = peak index
213  {
214  //just output existing peaks at each frame (i.e. freq != 0)
215  if(vec_(i, j) != 0.0)
216  {
217  for(mrs_natural k = 0; k < nbPkParameters; k++)// k = parameter index
218  vecTable(l, k) = (*this)(i, pkParameter(k), j);
219  l++;
220  }
221  }
222 
223  //resize the table to the correct (i.e. possibly smaller)
224  //number of peaks (i.e. nr of rows)
225 
226  vecTable.stretch(l, nbPkParameters);
227 }

229 void
230 peakView::fromTable(const realvec& vecTable)
231 {
232  //get data from header
233  fs_ = vecTable(0, 1);
234  frameSize_ = (mrs_natural)vecTable(0, 2);
235  frameMaxNumPeaks_ = (mrs_natural)vecTable(0, 3);
236  numFrames_ = (mrs_natural)vecTable(0, 4);
237 
238  //get the first frame in table (it may not be 0!!!)
239  mrs_natural frame = (mrs_natural)vecTable(1, pkFrame); // start frame [!]
240 
241  //resize (and set to zero) vec_ for new peak data
242  //(should accommodate empty frames before the first frame in table!)
243  vec_.create(frameMaxNumPeaks_*nbPkParameters, numFrames_+frame); //[!]
244 
245  mrs_natural p = 0; // peak index inside each frame
246  mrs_natural r = 1;//peak index in table (i.e. row index) - ignore header row
247 
248  //check in case realvec has less parameters than nbPkParameters
249  mrs_natural actualNbPkParams = (mrs_natural)min((mrs_real)nbPkParameters, (mrs_real)vecTable.getCols());
250 
251  //iterate over table rows (i.e. peaks) - excluding header row
252  while(r < vecTable.getRows()-1)
253  {
254  //get parameters for this peak
255  for(mrs_natural prm = 0; prm < actualNbPkParams; ++prm)
256  {
257  (*this)(p, pkParameter(prm), frame) = vecTable(r, prm);
258  }
259  ++r; //move on to next table row (i.e. peak)
260  p++;
261  //if the next row in table is form a different frame,
262  //reset peak index and get new frame index
263  if(vecTable(r, pkFrame) != frame)
264  {
265  frame = (mrs_natural)vecTable(r, pkFrame);
266  p = 0;
267  }
268  }
269 }
