122 bool
123 ExRecord::params_compare(std::string a, std::string b)
124 {
125  std::string::size_type p1=0;
126  std::string::size_type p2=0;
127  std::string::size_type len1 = a.length();
128  std::string::size_type len2 = b.length();
129  std::string::size_type start=1;
130 
131  while (p1<len1&&p2<len2) {
132  if (a[p1]!=b[p2]) {
133  if (a[p1]==','&&b[p2]=='|') {
134  while (p2<len2) {
135  if (b[p2]==',') break;
136  if (b[p2]==')') return false;
137  p2++;
138  }
139  p1++;
140  p2++;
141  }
142  else if (a[p1]==')'&&b[p2]=='|') {
143  while (p2<len2) {
144  if (b[p2]==',') return false;
145  if (b[p2]==')') return true;
146  p2++;
147  }
148  return false;
149  }
150  else {
151  while(p2<len2) {
152  if (b[p2]=='|') break;
153  if (b[p2]==','||b[p2]==')') return false;
154  p2++;
155  }
156  p1=start; p2++;
157  }
158  }
159  else {
160  if (a[p1]==',') start=p1+1;
161  p1++; p2++;
162  }
163  if (a[p1]==')'&&b[p2]==')') return true;
164  }
165  return false;
166 }

208 ExRecord*
209 ExRecord::find_sym(std::string path)
210 {
211  // get leading name in group: a|b|c.d|e|f.g|h|i => a|b|c d|e|f.g|h|i
212  std::string ghd, gtl; split_on(path,'.',ghd,gtl);
213  if (gtl==""&&ghd[0]!='(') {
214  split_on(ghd,'(',ghd,gtl,true);
215  }
216  if (gtl!="") { // then ghd is the variable name
217  // get leading name in name: a|b|c => a b|c
218  std::map<std::string, ExRecord*>::iterator gi=syms_.find(ghd);
219  if (gi==syms_.end()) {
220  std::map<std::string,std::string>::iterator it = syms_aliases_.find(ghd);
221  // a possible error condition since the path given diverges from the possible paths
222  if (it==syms_aliases_.end())
223  return NULL;
224  else
225  return syms_[it->second]->find_sym(gtl);
226  }
227 
228  return (gi->second)->find_sym(gtl);
229  }
230  if (path[0]=='(') {
231  ExRecord* answer=NULL;
232  std::map<std::string,ExRecord*>::iterator is=syms_.begin();
233  while (is!=syms_.end()&&!answer) {
234  if (params_compare((is->first),path)) { answer=is->second; }
235  else is++;
236  }
237  if (!answer) {
238  std::map<std::string,std::string>::iterator it=syms_aliases_.begin();
239  while (it!=syms_aliases_.end()&&!answer) {
240  if (params_compare((it->first),path))
241  answer=syms_[it->second];
242  else
243  ++it;
244  }
245  }
246  return answer;
247  }
248  std::map<std::string,ExRecord*>::iterator gi=syms_.find(ghd);
249  if (gi==syms_.end()) {
250  // name not found, so see if it is an alias
251  std::map<std::string,std::string>::iterator ni=syms_aliases_.find(ghd);
252  if (ni==syms_aliases_.end()) { return NULL; }
253  return syms_[ni->second];
254  }
255  return (gi->second);
256 }

270 void
271 ExRecord::addRecord(std::string path, ExRecord* sym)
272 {
273  // get leading name in group: a|b|c.d|e|f.g|h|i => a|b|c d|e|f.g|h|i
274  std::string ghd; std::string gtl; split_on(path,'.',ghd,gtl);
275  if (gtl!="") { // then ghd is the variable name
276  // now split on '|' symbol, get leading name in name: a|b|c => a b|c
277  std::string nhd, ntl; split_on(ghd,'|',nhd,ntl);
278  std::string name_=nhd;
279  std::map<std::string, ExRecord*>::iterator gi=syms_.find(name_);
280  ExRecord* syms=NULL;
281  // for first split, if name doesn't exist, add it as a record
282  if (gi==syms_.end()) { syms=new ExRecord(T_LIB); syms_[name_]=syms; syms->inc_ref(); }
283  else { syms=(gi->second); }
284 
285  // the rest of the names are aliases
286  addAliases(ntl,name_);
287  syms->addRecord(gtl,sym);
288  }
289  else {
290  // discover parameters
291  std::string params;
292  if (ghd[0]!='(') split_on(ghd,'(',ghd,params,true);
293  // get leading name in name: a|b|c => a b|c, becomes the master name
294  std::string nhd, ntl; split_on(ghd,'|',nhd,ntl);
295  std::map<std::string, ExRecord*>::iterator gi=syms_.find(nhd);
296  if (gi==syms_.end()) { // not found, so must be a new name
297  if (params!="") {
298  ExRecord* e=new ExRecord(T_FUN); e->inc_ref();
299  syms_[nhd]=e;
300  e->addRecord(params,sym);
301  }
302  else {
303  syms_[nhd]=sym;
304  sym->inc_ref();
305  }
306  }
307  else if (params!="") {
308  (gi->second)->addRecord(params,sym);
309  }
310  else {
311  MRSWARN("ExRecord::addRecord '"+nhd+"' already refers to a symbol");
312  sym->deref();
313  return;
314  }
315  addAliases(ntl,nhd);
316  }

318 ExRecord*
319 ExRecord::rmvRecord(std::string path)
320 {
321  // get leading name in group: a|b|c.d|e|f.g|h|i => a|b|c d|e|f.g|h|i
322  std::string ghd, gtl; split_on(path,'.',ghd,gtl);
323  std::map<std::string, ExRecord*>::iterator rec_i;
324  std::map<std::string,std::string>::iterator alias_i;
325 
326  if (gtl==""&&ghd[0]!='(') {
327  split_on(ghd,'(',ghd,gtl,true);
328  }
329  rec_i=syms_.find(ghd);
330  if (rec_i==syms_.end()) {
331  alias_i=syms_aliases_.find(ghd);
332  // a possible error condition since the path given diverges from the possible paths
333  if (alias_i==syms_aliases_.end()) { return NULL; }
334  ghd=alias_i->second;
335  rec_i=syms_.find(ghd); // let's assume that gi is valid
336  }
337  ExRecord* r;
338  bool del_rec=false;
339  if (gtl!="") r=(rec_i->second)->rmvRecord(gtl);
340  else { r=rec_i->second; del_rec=true; }
341  if (r!=NULL) {
342  std::vector<std::string> names_;
343  // eliminate aliases to r by first getting their names
344  for (alias_i=syms_aliases_.begin(); alias_i!=syms_aliases_.end(); ++alias_i) {
345  if (alias_i->second==ghd) names_.push_back(alias_i->first);
346  }
347  if (names_.size()>0) {
348  // delete aliases from syms_aliases_
349  std::vector<std::string>::iterator nmi;
350  for (nmi=names_.begin(); nmi!=names_.end(); ++nmi) {
351  syms_aliases_.erase(*nmi);
352  }
353  }
354  // eliminate the record from syms_
355  syms_.erase(rec_i);
356  if (del_rec&&r->size()>0) r->deref();
357  }
358  return r;
359 }

372 ExRecord::setValue(ExVal& v, std::string path, int elem_pos)
373 {
374  if (path!="") {
375  // get leading name in group: a|b|c.d|e|f.g|h|i => a|b|c d|e|f.g|h|i
376  std::string name; split_on(path,'.',name,path);
377  if (path!="") { // then ghd is the variable name
378  // get leading name in name: a|b|c => a b|c
379  std::map<std::string, ExRecord*>::iterator iter=syms_.find(name);
380  ExRecord* syms=NULL;
381  if (iter==syms_.end()) { syms=new ExRecord(); syms_[name]=syms; syms->inc_ref(); }
382  else { syms=iter->second; }
383  syms->setValue(v,path);
384  return;
385  }
386  else {
387  // get leading name in name: a|b|c => a b|c
388  std::map<std::string, ExRecord*>::iterator iter=syms_.find(name);
389  if (iter==syms_.end()) { // not in table, so add it
390  ExRecord* r=new ExRecord(T_VAR,name,v,false); r->inc_ref();
391  syms_[name]=r;
392  return;
393  }
394  else {
395  (iter->second)->setValue(v);
396  return;
397  }
398  }
399  }
400  if (getKind()!=T_VAR) {
401  MRSWARN("ExRecord::setValue Attempting assignment to non-variable");
402  }
403  else if (elem_pos>=0) {
404  if (getElemType()!=v.getType()) {
405  MRSWARN("ExRecord::setValue Type mismatch in assignment of element: "+getElemType()+" << "+v.getType());
406  }
407  else value_.setSeqElem(elem_pos, v);
408  }
409  else if (getType()!=v.getType()) {
410  MRSWARN("ExRecord::setValue Type mismatch in assignment: "+getType()+" << "+v.getType());
411  } else value_=v;
412 }
