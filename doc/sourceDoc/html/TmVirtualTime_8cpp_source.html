61 mrs_natural
62 TmVirtualTime::readTimeSrc()
63 {
64  if (read_src_==NULL) {
65  MRSWARN("TmVirtualTime::readTimeSrc() time source is not defined.");
66  return 0;
67  }
68  // this is the width of the last tick - elapsed time since last tick
69  mrs_natural ret = previous_tick_interval_;
70  mrs_real srate = srate_->to<mrs_real>();
71  if (srate<1.0)
72  return 0;
73  mrs_real interval_width = nsamples_->to<mrs_natural>() / srate;
74  mrs_real microseconds = (interval_width * 1000000.0) + error_term_;
75  error_term_ = microseconds - ((long)microseconds);
76 // mrs_natural count = (mrs_natural)microseconds;
77 // cout << "us=" << ((mrs_natural)microseconds) << endl;
78  previous_tick_interval_ = (mrs_natural)microseconds;
79  return ret;
80 }
