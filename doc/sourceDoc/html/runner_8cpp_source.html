66  int min_priority = sched_get_priority_min( policy );

68  int relative_priority = (int) ((max_priority - min_priority) * 0.6);
69  int priority = min_priority + relative_priority;

235 Control * Runner::create_control( const std::string & control_path )
236 {
237  if (isRunning()) {
238  MRSERR("Marsyas::Thread::System:: can not start tracking controls while running.");
239  return 0;
240  }
241 
242  Marsyas::MarControlPtr sys_control = m_system->getControl( control_path );
243  if (sys_control.isInvalid()) {
244  MRSERR("Marsyas::Thread::System:: Can not track control - invalid path: " << control_path);
245  return 0;
246  }
247 
248  AtomicControl *atomic_control;
249 
250  std::string sys_control_type = sys_control->getType();
251  if(sys_control_type == "mrs_bool")
252  atomic_control = new AtomicControlT<bool>(sys_control);
253  else if(sys_control_type == "mrs_real")
254  atomic_control = new AtomicControlT<mrs_real>(sys_control);
255  else if(sys_control_type == "mrs_natural")
256  atomic_control = new AtomicControlT<mrs_natural>(sys_control);
257  else if(sys_control_type == "mrs_string")
258  atomic_control = new AtomicControlT<mrs_string>(sys_control);
259  else if(sys_control_type == "mrs_realvec")
260  atomic_control = new AtomicControlT<mrs_realvec>(sys_control);
261  else {
262  MRSERR(
263  "Marsyas::Thread::System:: Can not track control - unsupported type: "
264  << sys_control_type.c_str()
265  );
266  return 0;
267  }
268 
269  Control * control = new Control(this, control_path, atomic_control);
270 
271  m_shared->controls.insert(std::pair<std::string, Control*>(control_path, control));
272

296 void RunnerThread::run()
297 {
298  //cout << "MarSystemThread: running" << endl;
299 
300  init_audio_recursive(m_system);
301 
302  m_system->updControl("mrs_bool/active", true);
303 
304  MarControlPtr done_control = m_system->getControl("mrs_bool/done");
305  function<bool()> not_system_done;
306  if (done_control.isInvalid())
307  not_system_done = [](){return true;};
308  else
309  not_system_done = [&done_control](){return !done_control->to<mrs_bool>();};
310 
311  int ticks_done = 0;
312  int ticks_left = m_ticks;
313 
314  while ( !m_stop &&
315  ticks_left &&
316  not_system_done() )
317  {
318  //cout << "tick" << endl;
319 
320  m_shared->controller->run();
321 
322  m_system->tick();
323 
324  for (const auto & mapping : m_shared->controls)
325  mapping.second->push();
326 
327  if (ticks_left > 0)
328  --ticks_left;
329  ++ticks_done;
330  }
331 
332 #if 0
333  if (done_control->to<mrs_bool>())
334  {
335  cout << "done in " << ticks_done << " ticks " << endl;
336  }
337  else if (ticks_left == 0)
338  {
339  cout << "timeout!" << endl;
340  }
341 #endif
342 
343  m_system->updControl("mrs_bool/active", false);
344 
345  // make sure not to miss latest updates:
346  m_shared->controller->run();
347 }
