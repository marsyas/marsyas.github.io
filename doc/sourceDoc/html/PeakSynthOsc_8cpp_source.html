91 void
92 PeakSynthOsc::myProcess(realvec& in, realvec& out)
93 {
94  out.setval(0);
95  silence_ = true;
96 
97  pkGrp2Synth_ = ctrl_peakGroup2Synth_->to<mrs_natural>();
98  Nb_ = in.getSize()/peakView::nbPkParameters ; //number of peaks in the current frame
99  nbH_ = ctrl_harmonize_->to<mrs_realvec>().getSize();
100 
101  if(nbH_)
102  for(mrs_natural j=0 ; j<(nbH_-1)/2 ; j++)
103  {
104  mulF_ = ctrl_harmonize_->to<mrs_realvec>()(1+j*2);
105  mulA_ = ctrl_harmonize_->to<mrs_realvec>()(2+j*2);
106  //cout << "mulF_" << mulF_ << "mulA_" << mulA_ << endl;
107  for (mrs_natural i=0; i < Nb_; ++i)
108  {
109  // either synthesize peaks with a corresponding GroupID or all with a group ID >= 0
110  mrs_bool synthMe = (pkGrp2Synth_ < 0)? (in(i+peakView::pkGroup*Nb_) >= 0) : (in(i+peakView::pkGroup*Nb_) == pkGrp2Synth_);
111  if( synthMe )
112  {
113  sine(out, in(i)*mulF_, in(i+Nb_)*mulA_, in(i+2*Nb_));
114  silence_ = false;
115  }
116  }
117  }
118  else
119  for (mrs_natural i=0; i < Nb_; ++i)
120  {
121  // either synthesize peaks with a corresponding GroupID or all with a group ID >= 0
122  mrs_bool synthMe = (pkGrp2Synth_ < 0)? (in(i+peakView::pkGroup*Nb_) >= 0) : (in(i+peakView::pkGroup*Nb_) == pkGrp2Synth_);
123  if( synthMe )
124  {
125  sine(out, in(i), in(i+Nb_), in(i+2*Nb_));
126  silence_ = false;
127  }
128  }
129 
130  //signal if at least one peak was synthesized or not
131  ctrl_isSilence_->setValue(silence_);
132 }
