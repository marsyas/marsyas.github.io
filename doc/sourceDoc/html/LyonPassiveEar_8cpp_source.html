49 class LyonAgc: public MarSystem
50 {
51 private:
52  static mrs_real lyonEpsilonFromTauFS (mrs_real tau, mrs_real fs)
53  {
54  return 1.0 - exp (-1/tau/fs);
55  }
56  // copied and reformatted from auditory toolbox
57  static void agc (const mrs_realvec input, mrs_realvec &output, mrs_realvec &state, mrs_real epsilon, mrs_real target, mrs_natural n)
58  {
59  mrs_natural i;
60  mrs_real f,
61  stateLimit = 1. - 1e-1;
62  mrs_real oneMinusEpsOverThree = (1.0 - epsilon)/3.0;
63  mrs_real epsOverTarget = epsilon/target;
64  mrs_real prevState = state(0);
65 
66  for ( i = 0; i < n-1; ++i)
67  {
68  output(i) = fabs (input(i) * (1.0 - state(i)));
69  f = output(i) * epsOverTarget + oneMinusEpsOverThree * (prevState + state(i) + state(i+1));
70 
71  if (f > stateLimit)
72  f = stateLimit;
73 
74  prevState = state(i);
75  state(i) = f;
76  }
77 
78  // do the last iteration with a different state input
79  output(i) = fabs (input(i) * (1.0 - state(i)));
80  f = output(i) * epsOverTarget + oneMinusEpsOverThree * (prevState + state(i) + state(i));
81 
82  if (f > stateLimit)
83  f = stateLimit;
84 
85  state(i) = f;
86  };

331 void
332 LyonPassiveEar::myUpdate(MarControlPtr sender)
333 {
334  (void) sender; //suppress warning of unused parameter(s)
335 
336  if (!setParametersIntern ())
337  {
338  this->updateControlsIntern ();
339  return;
340  }
341 
342  MRSDIAG("LyonPassiveEar.cpp - LyonPassiveEar:myUpdate");
343 
344  //FilterBank creation
345  // variable names more or less directly from matlab implementation...
346  const mrs_real Eb = 1000.0;
347  const mrs_real EarZeroOffset = 1.5;
348  const mrs_real EarSharpness = 5.0;
349  const mrs_real EarPremphCorner = 300.0;
350  mrs_natural i,numChans;
351  ostringstream name;
352  mrs_realvec a(3),
353  b(3);
354  mrs_real lowFreq,
355  topFreq = .5 * getctrl ("mrs_real/israte")->to<mrs_real>();
356  Cascade *filterBank = 0;
357 
358  //% Find top frequency, allowing space for first cascade filter.
359  // topf = fs/2.0;
360  // topf = topf - (sqrt(topf^2+Eb^2)/EarQ*StepFactor*EarZeroOffset)+ ...
361  // sqrt(topf^2+Eb^2)/EarQ*StepFactor;
362  topFreq = topFreq -
363  (sqrt (sqr(topFreq) + sqr(Eb)) / earQ_ * stepFactor_ * EarZeroOffset) +
364  sqrt (sqr(topFreq) + sqr(Eb)) / earQ_ * stepFactor_;
365 
366  //% Find place where CascadePoleQ < .5
367  // lowf = Eb/sqrt(4*EarQ^2-1);
368  // NumberOfChannels = floor((EarQ*(-log(lowf + sqrt(lowf^2 + Eb^2)) + ...
369  // log(topf + sqrt(Eb^2 + topf^2))))/StepFactor);
370  lowFreq = Eb / sqrt (4 * sqr(earQ_) - 1);
371  numChans = (mrs_natural)(floor ((earQ_ * (-log (lowFreq + sqrt (sqr(lowFreq) + sqr(Eb))) +
372  log (topFreq + sqrt (sqr(Eb) + sqr(topFreq)))))/stepFactor_) + .1); // add .1 to ensure correct cast...
373 
374  //% Now make an array of CenterFreqs..... This expression was derived by
375  // % Mathematica by integrating 1/EarBandwidth(cf) and solving for f as a
376  // % function of filterctrl number.
377  // cn = 1:NumberOfChannels;
378  // CenterFreqs = (-((exp((cn*StepFactor)/EarQ)*Eb^2)/ ...
379  // (topf + sqrt(Eb^2 + topf^2))) + ...
380  // (topf + sqrt(Eb^2 + topf^2))./exp((cn*StepFactor)/EarQ))/2;
381  centerFreqs_.create (numChans);
382  for (i = 0; i < numChans; ++i)
383  centerFreqs_(i) = (-((exp(((i+1)*stepFactor_)/earQ_) * sqr(Eb))/
384  (topFreq + sqrt(sqr(Eb) + sqr(topFreq)))) + (topFreq + sqrt(sqr(Eb) + sqr(topFreq)))/ exp(((i+1)*stepFactor_)/earQ_))*.5;
385 
386  // free memory if necessary...
387  if (passiveEar_)
388  {
389  delete passiveEar_;
390  passiveEar_ = 0;
391  }
392  // now create the processing chain
393  passiveEar_ = new Series ("LyonPassiveEar");
394  passiveEar_->addMarSystem (filterBank = new Cascade("LyonFilterBank"));
395  passiveEar_->addMarSystem (new HalfWaveRectifier("hwr1"));
396  passiveEar_->addMarSystem (new LyonZeroOutPreEmph("ZeroOut"));
397  if (agcActive_)
398  passiveEar_->addMarSystem (new LyonAgc("agc"));
399  if (channelDiffActive_)
400  {
401  passiveEar_->addMarSystem (new LyonChannelDiff("differ"));
402  passiveEar_->addMarSystem (new HalfWaveRectifier("hwr2"));
403  }
404 
405  if (decimFactor_ > 1)
406  {
407  b(0) = b(1) = 0;
408  b(2) = a(0) = 1;
409  a(1) = -2.0*(1-LyonAgc::lyonEpsilonFromTauFS (decimFactor_/fs_*decimTauFactor_, fs_));
410  a(2) = sqr(a(1)/(-2.0));
411  b *= lyonSetGain (b, a, 1.0, 0, fs_);
412  }
413  else
414  {
415  // otherwise set the filter to "bypass"
416  b(0) = a(0) = 1;
417  b(1) = b(2) = a(1) = a(2) = 0;
418  }
419 
420  passiveEar_->addMarSystem (lyonCreateFilter (b,a,"decim"));
421 
422  //% Finally, let's design the front filters.
423  // front(1,:) = SetGain([0 1 -exp(-2*pi*EarPremphCorner/fs) 0 0], 1, fs/4, fs);
424  // topPoles = SecondOrderFilter(topf,CascadePoleQ(1), fs);
425  // front(2,:) = SetGain([1 0 -1 topPoles(2:3)], 1, fs/4, fs);
426  b(0) = a(1) = a(2) = 0;
427  b(1) = a(0) = 1;
428  b(2) = -exp(-TWOPI*EarPremphCorner/fs_);
429  b *= lyonSetGain (b, a, 1.0, fs_*.25, fs_);
430 
431  name.str("");
432  name << "front_" << 0;
433  filterBank->addMarSystem(lyonCreateFilter (b, a, name.str())); // preemphasis 1
434 
435  b(0) = 1;
436  b(1) = 0;
437  b(2) = -1;
438  a = lyonSecondOrderFilter (topFreq, centerFreqs_(0)/(sqrt(sqr(centerFreqs_(0)) + sqr(Eb))/earQ_), fs_);
439  b *= lyonSetGain (b, a, 1.0F, fs_*.25F, fs_);
440 
441  name.str("");
442  name << "front_" << 1;
443  filterBank->addMarSystem(lyonCreateFilter (b, a, name.str())); // preemphasis 2
444 
445  for (i = 0; i < numChans; ++i)
446  {
447  mrs_real EarBandwidth, CascadeZeroCF, CascadeZeroQ, CascadePoleCF, CascadePoleQ;
448 
449  //% OK, now we can figure out the parameters of each stage filter.
450  // EarBandwidth = sqrt(CenterFreqs.^2+Eb^2)/EarQ;
451  // CascadeZeroCF = CenterFreqs + EarBandwidth * StepFactor * EarZeroOffset;
452  // CascadeZeroQ = EarSharpness*CascadeZeroCF./EarBandwidth;
453  // CascadePoleCF = CenterFreqs;
454  // CascadePoleQ = CenterFreqs./EarBandwidth;
455  EarBandwidth = sqrt(sqr(centerFreqs_(i)) + sqr(Eb))/earQ_;
456  CascadeZeroCF = centerFreqs_(i) + EarBandwidth * stepFactor_ * EarZeroOffset;
457  CascadeZeroQ = EarSharpness * CascadeZeroCF / EarBandwidth;
458  CascadePoleCF = centerFreqs_(i);
459  CascadePoleQ = centerFreqs_(i)/EarBandwidth;
460 
461  // compute filter coeffs
462  b = lyonSecondOrderFilter (CascadeZeroCF, CascadeZeroQ, fs_);
463  a = lyonSecondOrderFilter (CascadePoleCF, CascadePoleQ, fs_);
464 
465  //% Now we can set the DC gain of each stage.
466  // dcgain(2:NumberOfChannels)=CenterFreqs(1:NumberOfChannels-1)./ ...
467  // CenterFreqs(2:NumberOfChannels);
468  // dcgain(1) = dcgain(2);
469  // for i=1:NumberOfChannels
470  // filters(i,:) = SetGain(filters(i,:), dcgain(i), 0, fs);
471  // end
472  b *= (i == 0) ? lyonSetGain (b, a, centerFreqs_(i)/centerFreqs_(i+1), 0, fs_) :
473  lyonSetGain (b, a, centerFreqs_(i-1)/centerFreqs_(i), 0, fs_);
474 
475  // create the filter and add it
476  name.str("");
477  name << "filter_" << i;
478  filterBank->addMarSystem(lyonCreateFilter (b, a, name.str()));
479  }
480 
481  numFilterChannels_ = numChans + 2; // plus two front filters...
482 
483  this->updateControlsIntern ();
484 
485  // alloc some internal memory
486  tmpOut_.create (numFilterChannels_, inSamples_);
487  decimOut_.create (numFilterChannels_-2, inSamples_/decimFactor_); // integer division intentionally;
488 }

490 void
491 LyonPassiveEar::myProcess(realvec& in, realvec& out)
492 {
493  if(getctrl("mrs_bool/mute")->to<mrs_bool>()) return;
494 
495  // check number of out observations
496  mrs_natural i,
497  currCount = -currDecimState_-1,
498  numOutSamples = (inSamples_+currDecimState_)/decimFactor_;// integer division intended
499 
500  MRSASSERT(currDecimState_ <= decimFactor_);
501 
502  // update the number of output samples if necessary
503  if (onSamples_ != numOutSamples)
504  updControl ("mrs_natural/onSamples", numOutSamples);
505 
506  decimOut_.stretch (numFilterChannels_-2, numOutSamples);
507 
508  // process the series
509  passiveEar_->process(in, tmpOut_);
510 
511  // post-"processing": do the sample rate decimation and remove the pre-emphasis filters
512  for (i = 0; i < numOutSamples; ++i)
513  {
514  mrs_realvec decimTmp(numFilterChannels_-2,1);
515  currCount = currCount + decimFactor_;
516  MRSASSERT(currCount <= inSamples_);
517  tmpOut_.getSubMatrix (2, currCount, decimTmp);
518  decimOut_.setSubMatrix (0, i, decimTmp);
519  }
520  currDecimState_ = inSamples_ - currCount-1;
521  out = decimOut_;
522 }

595 mrs_realvec LyonPassiveEar::lyonSecondOrderFilter (mrs_real midFreq, mrs_real q, mrs_real sRate)
596 {
597  //function filts = SecondOrderFilter(f,q,fs)
598  //cft = f'/fs;
599  //rho = exp(- pi * cft ./ q');
600  // theta = 2 * pi * cft .* sqrt(1-1 ./(4*q'.^2));
601  // filts = [ ones(size(rho)) -2*rho.*cos(theta) rho.^2 ];
602  mrs_realvec result (3);
603  mrs_real cft = midFreq / sRate,
604  rho = exp (-PI * cft / q);
605 
606  result(0) = 1;
607  result(1) = -2 * rho * cos (TWOPI * cft * sqrt (1 - 1.0/(4*sqr(q))));
608  result(2) = sqr(rho);
609  return result;
610 }

613 mrs_real LyonPassiveEar::lyonFreqResp (mrs_realvec firCoeffs, mrs_realvec iirCoeffs, mrs_real freq, mrs_real sRate, mrs_bool inDb)
614 {
615  //function mag=FreqResp(filter,f,fs)
616  // cf = exp(i*2*pi*f/fs);
617  //mag = (filter(3) + filter(2)*cf + filter(1)*cf.^2) ./ ...
618  // (filter(5) + filter(4)*cf + cf.^2);
619  //mag = 20*log10(abs(mag));
620  mrs_complex cf = mrs_complex (cos (TWOPI * freq / sRate), sin (TWOPI * freq / sRate));
621  mrs_complex mag = (firCoeffs(2) + firCoeffs(1) * cf + firCoeffs(0) * (cf*cf)) / (iirCoeffs(2) + iirCoeffs(1) * cf + (cf*cf));
622  mrs_real res = sqrt (sqr(mag.real ()) + sqr (mag.imag ()));
623 
624  return inDb? 20.0/log(10.0) * log (res) : res;
