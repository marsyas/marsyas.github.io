98  }
99 
100  mode_ = getctrl("mrs_string/mode")->to<mrs_string>();
101 
102  myfft_->rfft(channel1, inSamples_/2, FFT_FORWARD);
103  myfft_->rfft(channel2, inSamples_/2, FFT_FORWARD);
104 
105  if (mode_ == "general")
106  {
107 
108  //Compress the magnitude spectrum and zero
109  // the imaginary part.
110  for (t=1; t < inSamples_/2; t++)
111  {
112  re1 = channel1[2*t];
113  im1 = channel1[2*t+1];
114 
115  re2 = channel2[2*t];
116  im2 = channel2[2*t+1];
117 
118  //(re1 + j im1)*(re2 - j im2) = (re1*re2 + im1*im2) + j(im1re2 - im2re1)
119 
120  re = re1*re2 + im1*im2;
121  im = re2*im1 - re1*im2;
122 
123  channelOut[2*t] = re;
124  channelOut[2*t+1] = im;
125  }
126  }
127  else if (mode_ == "phat")
128  {
129 
130  //Compress the magnitude spectrum and zero
131  // the imaginary part.
132  for (t=1; t < inSamples_/2; t++)
133  {
134  re1 = channel1[2*t];
135  im1 = channel1[2*t+1];
136 
137  re2 = channel2[2*t];
138  im2 = channel2[2*t+1];
139 
140  //(re1 + j im1)*(re2 - j im2) = (re1*re2 + im1*im2) + j(im1re2 - im2re1)
141 
142  re = re1*re2 + im1*im2;
143  im = re2*im1 - re1*im2;
144 
145  // divide by absolute value (PHAT)
146  abs_out = sqrt(re*re + im*im);
147 
148  re = re/abs_out;
149  im = im/abs_out;
150 
151  channelOut[2*t] = re;
152  channelOut[2*t+1] = im;
153 
154  }
155 
156  }
157  else if (mode_ == "ml")
158  {
159  // Maximum Likelihood - robust solution for time difference calculation
160  //
161  // Reference:
162  // MAXIMUM LIKELIHOOD TIME DELAY ESTIMATION WITH PHASE DOMAIN ANALYSIS
163  // IN THE GENERALIZED CROSS CORRELATION FRAMEWORK
164 
165  mrs_real *sub_channel1 = sub_scratch1_.getData();
166  mrs_real *sub_channel2 = sub_scratch2_.getData();

169  mrs_natural sub_window_size, sub_hop, sub_count, sub_start, sub_end;
170 
171  mrs_real re_q1_1,re_q2_2,re_q1_2,im_q1_2;
172 
173  mrs_natural window_size = inSamples_;
174 
175  mrs_realvec q1_1(window_size);
176  mrs_realvec q2_2(window_size);
177  mrs_realvec q1_2(window_size);
178  mrs_realvec mu1_2(window_size);
179  mrs_realvec v1_2(window_size);
180 
181 
182  //-------------------------------------------------------------------------------
183  // SUB LOOP -
184  // Used to calculate expected correlation mean, phase mean and phase variation
185  // based on subsets of the data
186 
187  sub_window_size = window_size/4;
188  sub_hop = window_size/8;
189  sub_start = 0;
190  sub_end = sub_start + sub_window_size;
191  sub_count = 1;
192 
193  // Set to zero
194  for (t=0; t < inSamples_; t++)
195  {
196  q1_1(t) = 0;
197  q2_2(t) = 0;
198  q1_2(t) = 0;
199  mu1_2(t) = 0;
200  v1_2(t) = 0;
201  }
202 
203  while (sub_end < window_size)
204  {
205  for (t=0; t < sub_window_size; t++)
206  {
207  sub_scratch1_(t) = 0;
208  sub_scratch2_(t) = 0;
209 
210  sub_scratch1_(t) = in(o,t+sub_start);
211  sub_scratch2_(t) = in(o+1,t+sub_start);
212  }
213 
214  // zeropad
215  for (t=sub_window_size; t < window_size; t++)
216  {
217  sub_scratch1_(t) = 0;
218  sub_scratch2_(t) = 0;
219  }
220 
221  myfft_->rfft(sub_channel1, inSamples_/2, FFT_FORWARD);
222  myfft_->rfft(sub_channel2, inSamples_/2, FFT_FORWARD);
223 
224  for (t=0; t < inSamples_/2; t++)
225  {
226  sub_re1 = sub_channel1[2*t];
227  sub_im1 = sub_channel1[2*t+1];
228 
229  sub_re2 = sub_channel2[2*t];
230  sub_im2 = sub_channel2[2*t+1];
231 
232  //(re1 + j im1)*(re2 - j im2) = (re1*re2 + im1*im2) + j(im1re2 - im2re1)
233 
234  // autocorrelations
235  q1_1(2*t) = q1_1(2*t) + (sub_re1*sub_re1 + sub_im1*sub_im1);
236  q1_1(2*t+1) =0;
237 
238  q2_2(2*t) = q2_2(2*t) + (sub_re2*sub_re2 + sub_im2*sub_im2);
239  q2_2(2*t+1) =0;
240 
241  // cross correlation
242  q1_2(2*t) = q1_2(2*t) + (sub_re1*sub_re2 + sub_im1*sub_im2);
243  q1_2(2*t+1) = q1_2(2*t+1) + (sub_re2*sub_im1 - sub_re1*sub_im2);
244  }
245 
246  sub_start = sub_start + sub_hop;
247  sub_end = sub_end + sub_hop;
248  sub_count = sub_count + 1;
249  }
250 
251 
252  for (t=0; t < inSamples_/2; t++)
253  {
254  // q = q/sub_count; (for averaging)
255  re_q1_1 = q1_1(2*t)/sub_count;
256  re_q2_2 = q2_2(2*t)/sub_count;
257 
258  re_q1_2 = q1_2(2*t)/sub_count;
259  im_q1_2 = q1_2(2*t+1)/sub_count;
260 
261  // mu1_2 = abs(q1_2)/sqrt(q2_2*q1_1)

263  mu1_2(2*t+1) = 0;
264

267  }
