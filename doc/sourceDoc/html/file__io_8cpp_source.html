91 bool FileWriter::write_record( const Record & record )
92 {
93  std::vector<const realvec*> data;
94 
95  for (const auto & descriptor : m_descriptors)
96  {
97  const realvec * output;
98 
99  try
100  {
101  const Record::Entry & record_entry = record.entries().at(descriptor.path);
102  output = &record_entry.output;
103  }
104  catch (const std::out_of_range &)
105  {
106  std::cerr << "Marsyas::Debug::FileWriter: Record has no entry for path: "
107  << descriptor.path << std::endl;
108  return false;
109  }
110 
111  if (output->getRows() != descriptor.out_rows ||
112  output->getCols() != descriptor.out_columns)
113  {
114  std::cerr << "Marsyas::Debug::FileWriter: Record entry format mismatch for path: "
115  << descriptor.path << std::endl;
116  return false;
117  }
118 
119  data.push_back(output);
120  }
121 
122  for (const realvec * d : data)
123  {
124  const realvec & vector = *d;
125  for (int i = 0; i < vector.getSize(); ++i)
126  {
127  char bytes[sizeof(mrs_real)];
128  to_bytes<mrs_real>( vector(i), bytes );
129  m_file.write( bytes, sizeof(mrs_real) );
130  }
131  }
132 
133  return true;
134 }

187 bool FileReader::read_header()
188 {
189  while (!m_file.eof())
190  {
191  SystemDescriptor descriptor;
192 
193  m_file >> descriptor.path;
194  if (descriptor.path == ".")
195  break;
196 
197  m_file >> descriptor.out_columns;
198  if (m_file.fail())
199  return false;
200 
201  m_file >> descriptor.out_rows;
202  if (m_file.fail())
203  return false;
204 
205  m_descriptors.push_back(descriptor);
206 
207  m_record_size += descriptor.out_columns * descriptor.out_rows;
208  }
209 
210  if (!m_descriptors.size())
211  {
212  std::cerr << "Marsyas::Debug::FileReader: No header entries!" << std::endl;
213  return false;
214  }
215 
216  m_file.ignore( std::numeric_limits<std::streamsize>::max(), '\n' );
217 
218  m_start_pos = m_file.tellg();
219 
220  return true;
221 }

223 bool FileReader::read_record( Record & record )
224 {
225  if (!m_record_size || m_start_pos == (pos_t)-1 || m_file.eof())
226  {
227  std::cerr << "Marsyas::Debug::FileReader: Nothing to read." << std::endl;
228  return false;
229  }
230 
231  for (const auto & descriptor : m_descriptors)
232  {
233  realvec data(descriptor.out_rows, descriptor.out_columns);
234 
235  for (int i = 0; i < data.getSize(); ++i)
236  {
237  char bytes[sizeof(mrs_real)];
238  m_file.read( bytes, sizeof(mrs_real) );
239  from_bytes<mrs_real>( data.getData()[i], bytes );
240  }
241  if (m_file.fail())
242  return false;
243 
244  Record::Entry entry;
245  entry.output = data;
246  record.insert(descriptor.path, entry);
247  }
248 
249  return true;
250 }
