104 void
105 PowerSpectrum::myProcess(realvec& in, realvec& out)
106 {
107  mrs_natural t,o;
108  for (t=0; t < inSamples_; ++t)
109  {
110  for (o=0; o < N2_; o++)
111  {
112  if (o==0) //DC bin (i.e. 0)
113  {
114  re_ = in(0,t);
115  im_ = 0.0;
116  }
117  else if (o == N2_-1) //Nyquist bin (i.e. N/2)
118  {
119  re_ = in(1,t);
120  im_ = 0.0;
121  }
122  else //all other bins
123  {
124  re_ = in(2*o, t);
125  im_ = in(2*o+1, t);
126  }
127 
128  switch (ntype_)
129  {
130  case PSD_POWER:
131  out(o, t) = re_*re_ + im_*im_;
132  break;
133  case PSD_MAG:
134  out(o,t) = sqrt(re_ * re_ + im_ * im_);
135  break;
136  case PSD_DB:
137  // TODO(sness) - Check validity of this magic number. Should probably be FLT_MIN
138  dB_ = (mrs_real)(10*log10(re_ * re_ + im_ * im_ + 0.000000001));
139  out(o,t) = dB_;
140  break;
141  case PSD_WDB:
142  // NOTE : FIXME!!!
143  // 20*log10() seems to work better for toy_with_onsets
144  // This is not good, and should be fixed inside the code that
145  // looks for onsets.
146  dB_ = (mrs_real)(20*log10(re_ * re_ + im_ * im_ + 0.000000001));
147  if (dB_ < -100) dB_ = -100;
148  out(o,t) = dB_;
149  break;
150  case PSD_PD:
151  pwr_ = re_ * re_ + im_ * im_;
152  out(o,t) = (mrs_real)(2.0 * pwr_) / N2_;
153  break;
154  case PSD_LOGMAG2:
155  out(o,t) = (mrs_real)log10(1+sqrt(re_ * re_ + im_ * im_));
156  break;
157  case PSD_LOGMAG:
158  out(o,t) = log(1+1000.0 * sqrt(re_ * re_ + im_ * im_));
159  // out(o,t) = pow(sqrt(re_ * re_ + im_ * im_), 0.5);
160  // out(o,t) = asin(sqrt(re_ * re_ + im_ * im_));
161 
162  }
163  }
164  }
