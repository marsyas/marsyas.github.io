108 void
109 ArffFileSink::prepareOutput()
110 {
111  // If internal file name differs from the file name we should be writing to:
112  // close previous file, open a new file and write header.
113  if (filename_ != ctrl_filename_->to<mrs_string>())
114  {
115  // Close current file.
116  closeOutput();
117  // Open a new output stream for the new file.
118  filename_ = ctrl_filename_->to<mrs_string>();
119  os_ = new ofstream;
120  os_->open(filename_.c_str());
121  if (os_->fail()) {
122  // \todo make this cleaner, e.g. by creating and using a dedicated
123  // exception for IO/file errors.
124  ostringstream oss;
125  oss << "[Error in " << __FILE__ << ":" << __LINE__ << "] "
126  << "could not open file '" << filename_ << "' for writing.";
127  throw ios_base::failure(oss.str());
128  }
129  // Write ARFF header.
130  writeArffHeader();
131  }
132 }

135 void
136 ArffFileSink::writeArffHeader()
137 {
138  // General header stuff.
139  (*os_) << "% ARFF file Created by Marsyas (ArffFileSink)" << endl;
140  (*os_) << "@relation " << filename_ << endl;
141 
142  // Print the attributes.
143  // Get and output the observation names.
144  mrs_string onObsNames = ctrl_onObsNames_->to<mrs_string>();
145  vector<mrs_string> attributeNames = stringSplit(onObsNames, ",");
146  const mrs_natural onObservations = ctrl_onObservations_->to<mrs_natural>();
147  // Print the observation names, if available.
148  for (mrs_natural i = 0; i < onObservations; ++i)
149  {
150  if (i < (mrs_natural)attributeNames.size() && !attributeNames[i].empty())
151  {
152  (*os_ ) << "@attribute " << attributeNames[i] << " real" << endl;
153  }
154  else
155  {
156  (*os_ ) << "@attribute " << "untitled" << i << " real" << endl;
157  }
158  }
159 
160  // End of header, we are ready now for outputting the data.
161  (*os_) << "\n@data" << endl;
162 }

164 void
165 ArffFileSink::myProcess(realvec& in, realvec& out)
166 {
167  mrs_natural o,t;
168  // Make sure we can write to the output stream.
169  prepareOutput();
170 
171  // Copy input to output.
172  for (t = 0; t < inSamples_; t++)
173  {
174  for (o = 0; o < inObservations_; o++)
175  {
176  out(o, t) = in(o, t);
177  }
178  }
179 
180  if (!ctrl_mute_->isTrue()) {
181  // Write data to file.
182  for (t = 0; t < inSamples_; t++)
183  {
184  if (decimationCounter % decimationFactor_ == 0)
185  {
186  for (o = 0; o < inObservations_; o++)
187  {
188  (*os_) << fixed << setprecision(floatPrecision_) << out(o, t);
189  if (o < inObservations_ - 1)
190  {
191  (*os_) << ",";
192  }
193  }
194  (*os_) << endl;
195  }
196  decimationCounter ++;
197  }
198  }
199 }
