60 {
61 
62  // fundamental frequency estimate
63  mrs_real hF;
64 
65  // Use the lowest in frequency highest amplitude
66  // peak of the frames under consideration
67  hF = min(peak1SetFreqs(0), peak2SetFreqs(0));
68 
69  // Original HWPS using the considered peaks for folding
70  // hF = min(peak1Freq, peak2Freq);
71 
72  // mrs_real mhF = min(hF, abs(peak1Freq-peak2Freq));
73 
74  // shift frequencies
75  peak1SetFreqs -= peak1Freq;
76  peak2SetFreqs -= peak2Freq;
77 
78 
79  /*
80  MATLAB_PUT(peak1SetFreqs, "P1");
81  MATLAB_PUT(peak2SetFreqs, "P2");
82  MATLAB_EVAL("clf ; subplot(3, 1, 1); hold ; stem(P1, A1); stem(P2, A2, 'r')");
83  */
84 
85  // wrap frequencies around fundamental freq estimate
86  peak1SetFreqs /= hF;
87  peak2SetFreqs /= hF;
88 
89  for (mrs_natural k=0 ; k<peak1SetFreqs.getSize() ; k++)
90  {
91  peak1SetFreqs(k)=fmod(peak1SetFreqs(k), 1);
92  //if(peak1SetFreqs(k)<0)
93  while(peak1SetFreqs(k)<0)//replacing "if" in case of strongly negative (=> multiple wraps)
94  peak1SetFreqs(k)+=1;
95  }
96  for (mrs_natural k=0 ; k<peak2SetFreqs.getSize() ; k++)
97  {
98  peak2SetFreqs(k)=fmod(peak2SetFreqs(k), 1);
99  //if(peak2SetFreqs(k)<0)

101  peak2SetFreqs(k)+=1;
102  }

142 void
143 HWPS::myProcess(realvec& in, realvec& out)
144 {
145  mrs_natural o;
146  //get the two stacked vectors from the input
147  for(o=0; o < inObservations_/2; ++o)
148  {
149  vec_i_(o) = in(o,0);
150  vec_j_(o) = in(o+inObservations_/2,0);
151  }
152 
153  //parse data from the input vectors.
154  //format:
155  //[peakFreq, frameNumPeaks, pkSetFreq0,...,pkSetFreqN, pkSetAmp0,..., pkSetAmpN, dummyPad, dummyPad, ...]'
156  //where N = frameNumPeaks-1, and dummyPad are dummy values used for filling the vector when frameNumPeaks < maxFrameNumPeaks.
157  pk_i_freq_ = vec_i_(HWPS::pkFreqIdx); //peak i frequency
158  pk_j_freq_ = vec_j_(HWPS::pkFreqIdx); //peak j frequency
159  i_frameNumPeaks_ = (mrs_natural)vec_i_(HWPS::frameNumPeaksIdx); //peakSet i frameNumPeaks
160  j_frameNumPeaks_ = (mrs_natural)vec_j_(HWPS::frameNumPeaksIdx); //peakSet j frameNumPeaks
161 
162  //get i peaksets freqs and amplitudes
163  pkSet_i_Freqs_.stretch(i_frameNumPeaks_);
164  pkSet_i_Amps_.stretch(i_frameNumPeaks_);
165  for(o=0; o < i_frameNumPeaks_; ++o)
166  {
167  pkSet_i_Freqs_(o) = vec_i_(o+pkSetFeatsIdx);
168  pkSet_i_Amps_(o) = vec_i_(o+pkSetFeatsIdx+i_frameNumPeaks_);
169  }
170  //get j peaksets freqs and amplitudes
171  pkSet_j_Freqs_.stretch(j_frameNumPeaks_);
172  pkSet_j_Amps_.stretch(j_frameNumPeaks_);
173  for(o=0; o < j_frameNumPeaks_; ++o)
174  {
175  pkSet_j_Freqs_(o) = vec_j_(o+pkSetFeatsIdx);
176  pkSet_j_Amps_(o) = vec_j_(o+pkSetFeatsIdx+j_frameNumPeaks_);
177  }
178 
179  //perform harmonic wrapping
180  pkSet_i_WrapFreqs_ = pkSet_i_Freqs_;
181  pkSet_j_WrapFreqs_ = pkSet_j_Freqs_;
182  harmonicWrap(pk_i_freq_, pk_j_freq_, pkSet_i_WrapFreqs_, pkSet_j_WrapFreqs_);
183 
184  /*
185  MATLAB_PUT(pkSet_i_WrapFreqs_, "P1");
186  MATLAB_PUT(pkSet_j_WrapFreqs_, "P2");
187  MATLAB_PUT(pkSet_i_Amps_, "A1");
188  MATLAB_PUT(pkSet_j_Amps_, "A2");
189  MATLAB_EVAL("subplot(3, 1, 2); hold ; stem(P1, A1); stem(P2, A2, 'r')");
190  */
191 
192  //create histograms for both peaks
193  histSize_ = ctrl_histSize_->to<mrs_natural>();
194  discretize(pkSet_i_WrapFreqs_, pkSet_i_Amps_, histSize_, histogram_i_);
195  discretize(pkSet_j_WrapFreqs_, pkSet_j_Amps_, histSize_, histogram_j_);
196 
197  /*
198  MATLAB_PUT(histogram_i_, "H1");
199  MATLAB_PUT(histogram_j_, "H2");
200  MATLAB_EVAL("subplot(3, 1, 3); bar([H1; H2]')");
201  */
202 
203  if(ctrl_calcDistance_->isTrue())
204  {
205  //return the cosine DISTANCE between the two histograms and we get the HWPDistance!
206  out(0) = NumericLib::cosineDistance(histogram_i_, histogram_j_);
207  }
208  else
209  {
210  //return the cosine SIMILARITY between the two histograms and we get the HWPSimilarity!
211  out(0) = 1.0 - NumericLib::cosineDistance(histogram_i_, histogram_j_);
212  }
213 }
