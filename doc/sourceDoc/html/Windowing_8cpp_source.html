75 Windowing::myUpdate(MarControlPtr sender)

83 
84  // Add prefix to the observation names.
85  mrs_string inObsNames = ctrl_inObsNames_->to<mrs_string>();
86  mrs_string prefix = mrs_string("Win") + type + mrs_string("_");
87  ctrl_onObsNames_->setValue(obsNamesAddPrefix(inObsNames, prefix), NOUPDATE);
88 
89  //if zeroPadding control changed...
90  if (ctrl_zeroPadding_->to<mrs_natural>() != onSamples_- inSamples_)
91  {
92  //zero padding should always be a positive or zero value
93  if (ctrl_zeroPadding_->to<mrs_natural>() < 0)
94  {
95  ctrl_zeroPadding_->setValue(0, NOUPDATE);
96  }
97  ctrl_size_->setValue(ctrl_inSamples_->to<mrs_natural>() +
98  ctrl_zeroPadding_->to<mrs_natural>(), NOUPDATE);
99  onSamples_ = ctrl_size_->to<mrs_natural>();
100  }
101  //if size control changed...
102  if (ctrl_size_->to<mrs_natural>() != onSamples_)
103  {
104  //size should never be smaller than inSamples
105  if (ctrl_size_->to<mrs_natural>() < inSamples_)
106  {
107  ctrl_size_->setValue(inSamples_, NOUPDATE);
108  }
109  ctrl_zeroPadding_->setValue(ctrl_size_->to<mrs_natural>() -
110  ctrl_inSamples_->to<mrs_natural>(), NOUPDATE);
111  }
112 
113  ctrl_onSamples_->setValue(ctrl_size_, NOUPDATE);
114 
115  //check if zero phasing should be performed
116  if (ctrl_zeroPhasing_->isTrue())
117  {
118  delta_ = inSamples_/2+1;
119  }
120  else
121  {
122  delta_=0;
123  }
124 
125 
126  tmp_.create(inSamples_);
127 
128  // Precalculate the envelope.
129  // \todo only do this when the type or size is changed.
130  envelope_.create(inSamples_);
131 
132  if (type == "Hamming")
133  {
134  windowingFillHamming(envelope_);
135  }
136  else if (type == "Hanning" || type == "Hann")
137  {
138  windowingFillHanning(envelope_);
139  }
140  else if (type == "Triangle")
141  {
142  windowingFillTriangle(envelope_);
143  }
144  else if (type == "Bartlett")

147  }

151  }
152  else if (type == "Blackman")
153  {
154  windowingFillBlackman(envelope_, 0.16);
155  }
156  else if (type == "Blackman-Harris")
157  {
158  windowingFillBlackmanHarris(envelope_);
159  }

161  {

163  }

189 void
190 Windowing::myProcess(realvec& in, realvec& out)
191 {
192  out.setval(0.0);
193  mrs_natural o,t;
194 
195  for (o=0; o < inObservations_; o++)
196  {
197  //shift windowed data in case zeroPhasing is selected
198  if (ctrl_zeroPhasing_->isTrue())
199  {
200  //apply the window to the input data
201  for (t = 0; t < inSamples_; t++)
202  {
203  tmp_(t) = in(o,t)*envelope_(t); // /(norm_);
204  }
205  for (t = 0; t < inSamples_/2; t++)
206  {
207  out(o,t)=tmp_((t+delta_)%inSamples_);
208  }
209  for (t = inSamples_/2; t < inSamples_; t++)
210  {
211  out(o,t+(onSamples_-inSamples_))=tmp_((t+delta_)%inSamples_);
212  }
213  }
214  else
215  {
216  for (t=0; t< inSamples_; ++t)
217  {
218  out(o,t) = in(o,t) * envelope_(t);
219  }
220  }
221  }
222 }

297 void Marsyas::windowingFillGaussian(realvec& envelope, mrs_real sigma)
298 {
299  mrs_natural N = envelope.getSize();
300  mrs_real tmp;
301  for (mrs_natural t = 0; t < N; t++)
302  {
303  tmp = (t - (N - 1.0) / 2.0) / (sigma * (N - 1.0) / 2.0);
304  envelope(t) = exp(-0.5*tmp*tmp);
305  }

315 void Marsyas::windowingFillBlackman(realvec& envelope, mrs_real alpha)
316 {
317  mrs_natural N = envelope.getSize();
318  mrs_real a0 = (1.0 - alpha) / 2.0;
319  mrs_real a2 = alpha / 2.0;
320  for (mrs_natural t = 0; t < N; t++)
321  {
322  envelope(t) = a0 - 0.5 * cos(2.0 * PI * t / (N - 1.0))
323  + a2 * cos(4.0 * PI * t / (N - 1.0));
324  }
325 }
