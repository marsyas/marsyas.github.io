286 ExFun::setParams(ExNode* ps)
287 {
288  bool evaluatable=true;
289  num_params=0;
290  if (ps!=NULL) {
291  ExNode* p=ps;
292  for (; p!=NULL; num_params++,p=p->next);
293  params=new ExNode*[num_params];
294  int i=0;
295  while(i<num_params) {
296  ExNode* param=ps; ps=ps->next; param->next=NULL; // sever the linked list to avoid eval() list
297  std::string pt_i=param_types[i];
298  std::string pst=param->getType();
299  if (pt_i!=pst) {
300  if (pt_i=="mrs_real") {
301  if (pst=="mrs_natural") { param=new ExNode_NaturalToReal(param); }
302  }
303  else if (pt_i=="mrs_natural") {
304  if (pst=="mrs_real") { param=new ExNode_RealToNatural(param); }
305  }
306  }
307  params[i]=param;
308  if (param->getKind()!=T_CONST) { evaluatable=false; }
309  ++i;
310  }
311  }
312  const_params=evaluatable;
313 }
