102 
103  //pair indexes = peak amplitude

105 
106  for (ki=0; ki < k; ++ki)
107  {
108  if (newmax > out(o, 2*ki))
109  {
110  mrs_real oldmax = out(o, 2*ki);
111  mrs_real oldmax_i = out(o,2*ki+1);
112  out(o,2*ki) = newmax;
113  out(o,2*ki+1) = newmax_i;
114  newmax = oldmax;
