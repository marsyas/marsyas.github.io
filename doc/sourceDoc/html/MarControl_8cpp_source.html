164 bool
165 MarControl::linkTo(MarControlPtr ctrl, bool update)
166 {
167  if (ctrl.isInvalid())
168  {
169  ostringstream oss;
170  oss << "MarControl::linkTo() - Linking to an invalid control ";
171  oss << "(" << ctrl->cname_ << " with " << cname_ << ").";
172  MRSWARN(oss.str());
173  return false;
174  }
175 
176  //check if these controls are already linked
177  //(i.e. they own the same MarControlValue)
178  if(value_ == ctrl->value_)
179  {
180  return true;//already linked! :-)
181  }
182 
183  if (ctrl->value_->type_ != value_->type_)
184  {
185  ostringstream oss;
186  oss << "MarControl::linkTo() - Linking controls of different types ";
187  oss << "(" << ctrl->cname_ << " with " << cname_ << ").";
188  MRSWARN(oss.str());
189  return false;
190  }
191 
192  //unlink this control (but keeping all links to it)
193  //before linking it again to the passed control
194  this->unlinkFromTarget();
195 
196  //store a pointer to the (soon to be old) MarControlValue object,
197  //so we can later delete it from memory
198  MarControlValue* oldvalue = value_;
199  //and get a pointer to the new value
200  MarControlValue* newvalue = ctrl->value_;
201 
202  //get all the links of our current MarControlValue so we can also
203  //re-link them to the passed ctrl
204  vector<pair<MarControl*, MarControl*> >::iterator lit;
205  for(lit=oldvalue->links_.begin(); lit!=oldvalue->links_.end(); ++lit)
206  {
207  //make each linked control now point to the "passed" MarControlValue
208  lit->first->value_ = newvalue;
209 
210  // check if this is the root link
211  if(lit->first == lit->second)
212  {
213  //make it "link to" the passed control
214  newvalue->links_.push_back(pair<MarControl*, MarControl*>(lit->first, ctrl()));
215  }
216  else //if not a root link, just copy the table entry unchanged into the new MarControlValue
217  newvalue->links_.push_back(*lit);
218  }
219 
220  //old MarControlValue can and should now be safely deleted from memory
221  delete oldvalue;
222 
223  //check if it's needed to call update()
224  if(update)
225  value_->callMarSystemsUpdate();//newvalue->callMarSystemsUpdate();
226 
227  return true;
228 }

336 void
337 MarControl::unlinkFromAll()
338 {
339  //first unlink this control from all the controls to which
340  //it links to
341  this->unlinkFromTarget();
342 
343  //get a list of all the controls that target this controls...
344  vector<pair<MarControl*, MarControl*> >::iterator lit;
345  vector<MarControl*> linkedControls;
346  for(lit=value_->links_.begin(); lit!=value_->links_.end(); ++lit)
347  {
348  if(lit->second == this && lit->first != lit->second)
349  linkedControls.push_back(lit->first);
350  }
351  //... and now unlink them all from this
352  for(mrs_natural i=0; i < (mrs_natural)linkedControls.size(); ++i)
353  linkedControls[i]->unlinkFromTarget();
354 }

356 void
357 MarControl::unlinkFromTarget()
358 {
359  vector<pair<MarControl*, MarControl*> >::iterator lit;
360 
361  //check if this MarControl is linked
362  //(i.e. more than one MarControl linking
363  //to the MarControlValue).
364  //if not, no point doing unlink - just return.
365  if(value_->links_.size() <= 1)
366  return;
367 
368  MarControlValue* oldvalue = value_;
369  MarControlValue* newvalue = oldvalue->clone();
370 
371  vector<pair<MarControl*, MarControl*> >* inSet = new vector<pair<MarControl*, MarControl*> >;
372  vector<pair<MarControl*, MarControl*> >* outSet = new vector<pair<MarControl*, MarControl*> >;
373 
374  size_t toProcess = oldvalue->links_.size();
375  bool* processed = new bool[toProcess];
376  for(size_t i=0; i < toProcess; ++i)
377  processed[i] = false;
378 
379  //iterate over all the links
380  MarControl* oldRootLink = NULL;
381  //vector<pair<MarControl*, MarControl*> >::iterator lit;
382  lit = oldvalue->links_.begin();
383  mrs_natural idx = 0;
384  while(toProcess > 0)
385  {
386  //avoid processing processed links
387  if(!processed[idx])
388  {
389  //check if this is the old root link and send it to the outSet (also as a root link)
390  if(lit->first == lit->second)
391  {
392  oldRootLink = lit->first;
393  outSet->push_back(*lit);
394  toProcess --;
395  processed[idx]=true;
396  }
397  //check if this is this same control (i.e. the control to unlink at)
398  //and set it as the root link of the inSet
399  else if(lit->first == this)
400  {
401  lit->first->value_ = newvalue; //"this" is already locked for writing
402  inSet->push_back(pair<MarControl*, MarControl*>(lit->first, lit->first));
403  toProcess--;
404  processed[idx]=true;
405  }
406  //check if this is directly linked to the old root link and if so,
407  //send it to the outSet
408  else if(lit->second == oldRootLink)
409  {
410  outSet->push_back(*lit);
411  toProcess --;
412  processed[idx]=true;
413  }
414  //if this control links directly to the control we want to unlink, send it to the inSet
415  else if(lit->second == this)
416  {
417  lit->first->value_ = newvalue;
418  inSet->push_back(*lit);
419  toProcess--;
420  processed[idx]=true;
421  }
422  //This control is not directly connected to any root link, so we have
423  //to check to which set belongs the control it links to and assign it to
424  //the same set
425  else
426  {
427  bool found = false;
428  vector<pair<MarControl*, MarControl*> >::iterator sit;
429  //do a copy of the inSet because we are using iterators
430  //and they become invalid if we change the size of the inSet object
431  //inside the for loop below
432  vector<pair<MarControl*, MarControl*> > inSet2 = *inSet;
433  //start by searching in the inSet...
434  for(sit = inSet2.begin(); sit != inSet2.end(); ++sit)
435  {
436  if(lit->second == sit->first)
437  {
438  lit->first->value_ = newvalue;
439  inSet->push_back(*lit);
440  toProcess--;
441  processed[idx]=true;
442  found = true;
443  }
444  }
445  //if not found there, look for it in the outSet...
446  if(!found)
447  {
448  vector<pair<MarControl*, MarControl*> > outSet2 = *outSet;
449  for(sit = outSet2.begin(); sit != outSet2.end(); ++sit)
450  {
451  if(lit->second == sit->first)
452  {
453  outSet->push_back(*lit);
454  toProcess--;
455  processed[idx]=true;
456  }
457  }
458  }
459  }
460  }
461 
462  //iterate until all links were processed!
463  if(lit!=oldvalue->links_.end())
464  {
465  lit++;
466  idx++;
467  }
468  else
469  {
470  lit=oldvalue->links_.begin();
471  idx = 0;
472  }
473  }
474 
475  delete [] processed;
476 
477  //set the two unlinked tables
478  oldvalue->links_ = *outSet;
479  newvalue->links_ = *inSet;
480 
481  delete inSet;
482  delete outSet;
483 
484  //check if newValue has in fact any control
485  //linking to it
486  if(newvalue->links_.size() == 0)
487  delete newvalue;
488 }
