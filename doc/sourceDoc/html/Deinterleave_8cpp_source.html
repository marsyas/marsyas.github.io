74  MarSystem::myUpdate(sender);

76  mrs_natural numSets = ctrl_numSets_->to<mrs_natural>();
77 
78  outindex_.stretch(inObservations_);
79  inindex_.stretch(inObservations_);
80  weights_.stretch(inObservations_,inObservations_);
81 
82 
83  mrs_natural rest=inObservations_%numSets;
84  mrs_natural part=inObservations_/numSets;
85  mrs_natural count=0;
86 
87  for (mrs_natural t = 0; t < rest; t++)
88  {
89  for (mrs_natural n = 0; n <1+part ; n++)
90  {
91  //get Reordering indices for all rows for which there is a rest (they get one output more)
92  outindex_(count)=n+part*t;
93  inindex_(count)=numSets*n+t;
94  count++;
95  }
96  }
97  for (mrs_natural t = rest; t < numSets; t++)
98  {
99  for (mrs_natural n = 0; n < part; n++)
100  {
101  //get Reordering indices for all remaining rows
102  outindex_(count)=n+rest+part*t;
103  inindex_(count)=numSets*n+t;
104  count++;
105  }
106  }
107 
108  //Weights deinterleave n
109  for(mrs_natural i=0; i<inindex_.getSize(); ++i)
110  {
111  weights_((mrs_natural)(outindex_(i)),(mrs_natural)(inindex_(i)))=1.0;
112  }
113 
114  pat_->setctrl("mrs_realvec/weights",weights_);
115 
116 
117 }
