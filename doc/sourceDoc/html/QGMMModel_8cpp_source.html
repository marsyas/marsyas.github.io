158 QGMMModel::deltaBIC(realvec C1, mrs_natural N1, realvec C2, mrs_natural N2, realvec C, mrs_real lambda)
159 {
160  //matrices should be square and equal sized
161  if(C1.getCols() != C2.getCols() && C1.getCols() != C.getCols() &&
162  C1.getCols()!= C1.getRows())
163  {
164  MRSERR("QGMMModel:deltaBIC: matrices should all be squared and equal sized...");
165  return MAXREAL; //just a way to sinalize the above error... [!]
166  }
167 
168  mrs_real res;
169  mrs_real N = (mrs_real)(N1 + N2);
170  mrs_real d = (mrs_real)C1.getCols();
171 
172 
173  res = N * log(C.det());
174  res -= (mrs_real)N1 * log(C1.det());
175  res -= (mrs_real)N2 * log(C2.det());
176  res *= 0.5f;
177 
178  res -= 0.5f * lambda * (d + 0.5f*d*(d+1.0f))* log(N);
179 
180  return res;
181 }
