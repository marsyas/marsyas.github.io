79 void
80 ResampleNearestNeighbour::myProcess(realvec& in, realvec& out)
81 {
82  mrs_natural o,t;
83  mrs_real tp;
84  mrs_natural tl, tr;
85  mrs_real alpha = ctrl_stretch_->to<mrs_real>();
86 
87 
88  for (o=0; o < onObservations_; o++)
89  {
90  for (t = 0; t < onSamples_; t++)
91  {
92  tp = t / alpha;
93  tl= (mrs_natural)tp;
94  tr = tl + 1;
95  if (tl<inSamples_)
96  {
97 
98  out(o,t) = (tr-tp) * in(o,tl) + (tp-tl) * in(o,tr);
99  if((tp-tl)>(tr-tp)) //on equality default to the leftmost value
100  {
101  out(o,t) =in(o,tr);
102  }
103  else
104  {
105  out(o,t) =in(o,tl);
106  }
107  }
108  else // "reflect" on boundary
109  {
110  out(o,t) = in(o,inSamples_-1); // alternative possibilities would include:
111 
112  //periodic on boundary:
113  //out(o,t) = in(o,0);
