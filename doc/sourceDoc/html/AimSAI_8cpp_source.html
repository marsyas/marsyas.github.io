259 void
260 AimSAI::myProcess(realvec& in, realvec& out)
261 {
262  // cout << "AimSAI::myProcess" << endl;
263  size_t _max_concurrent_strobes = (size_t) std::max((mrs_natural) 0, ctrl_max_concurrent_strobes_->to<mrs_natural>());
264  mrs_real _israte = ctrl_israte_->to<mrs_real>();
265  // mrs_natural _inSamples = ctrl_inSamples_->to<mrs_natural>();
266 
267  // Grab the centre frequencies from the input
268  for (int o = 0; o < channel_count_; ++o) {
269  centre_frequencies_[o] = in(o + channel_count_);
270  // cout << "centre_frequencies_[" << o << "]=" << centre_frequencies_[o] << endl;
271  }
272 
273  findStrobes(in);
274 
275  // Reset the next strobe times
276  next_strobes_.clear();
277  next_strobes_.resize(channel_count_, 0);
278 
279  // Offset the times on the strobes from the previous buffer
280  for (int o = 0; o < channel_count_; ++o) {
281  active_strobes_[o].ShiftStrobes(ctrl_inSamples_->to<mrs_natural>());
282  }
283 
284  // Loop over samples to make the SAI
285  for (int t = 0; t < ctrl_inSamples_->to<mrs_natural>(); ++t)
286  {
287  double decay_factor = pow(sai_decay_factor_, fire_counter_);
288 
289  // Loop over channels
290  for (int o = 0; o < channel_count_; ++o)
291  {
292  // cout << "o=" << o << endl;
293  // cout << "\tch=" << o << endl;
294  // Local convenience variables
295  StrobeList &active_strobes = active_strobes_[o];
296  unsigned int next_strobe_index = next_strobes_[o];
297 
298  // Update strobes
299  // If we are up to or beyond the next strobe...
300  if (next_strobe_index < strobes_[o].size()) {
301  // cout << "(next_strobe_index < strobes_[o].size())" << endl;
302  // cout << "\t\tnext_strobe_index" << next_strobe_index << endl;
303  // cout << "\t\tinput.strobe(ch, next_strobe_index)=" << strobes_[o][next_strobe_index] << endl;
304 
305  if (t == strobes_[o][next_strobe_index]) {
306  // cout << "(t == strobes_[o][next_strobe_index])" << endl;
307  // A new strobe has arrived.
308  // If there are too many strobes active, then get rid of the
309  // earliest one
310  if (active_strobes.strobe_count() >= _max_concurrent_strobes) {
311  active_strobes.DeleteFirstStrobe();
312  }
313 
314  // Add the active strobe to the list of current strobes and
315  // calculate the strobe weight
316  double weight = 1.0;
317  if (active_strobes.strobe_count() > 0) {
318  int last_strobe_time = active_strobes.Strobe(
319  active_strobes.strobe_count() - 1).time;
320 
321  // If the strobe occured within 10 impulse-response
322  // cycles of the previous strobe, then lower its weight
323  // cout << "t=" << t << "\tlast_strobe_time=" << last_strobe_time << "_israte=" << _israte << " cf=" << centre_frequencies_[o] << endl;
324  weight = (t - last_strobe_time) / _israte
325  * centre_frequencies_[o] / 10.0;
326  if (weight > 1.0)
327  weight = 1.0;
328  }
329  active_strobes.AddStrobe(t, weight);
330  next_strobe_index++;
331 
332  // Update the strobe weights
333  double total_strobe_weight = 0.0;
334  for (size_t si = 0; si < active_strobes.strobe_count(); ++si) {
335  // cout << "si=" << si << "\tw=" << active_strobes.Strobe(si).weight << "\tw2=" << strobe_weights_[active_strobes.strobe_count() - si - 1] << endl;
336  total_strobe_weight += (active_strobes.Strobe(si).weight
337  * strobe_weights_[active_strobes.strobe_count() - si - 1]);
338  }
339  for (size_t si = 0; si < active_strobes.strobe_count(); ++si) {
340  // cout << "si=" << si << "\tw1=" << active_strobes.Strobe(si).weight << "\tw2=" << strobe_weights_[active_strobes.strobe_count() - si - 1] << " tot=" << total_strobe_weight << endl;
341  active_strobes.SetWorkingWeight(si,
342  (active_strobes.Strobe(si).weight
343  * strobe_weights_[active_strobes.strobe_count() - si - 1])
344  / total_strobe_weight);
345  }
346  }
347  }
348 
349  // Remove inactive strobes
350  while (active_strobes.strobe_count() > 0) {
351  // Get the relative time of the first strobe, and see if it exceeds
352  // the maximum allowed time.
353  if ((t - active_strobes.Strobe(0).time) > max_strobe_delay_idx_) {
354  active_strobes.DeleteFirstStrobe();
355  } else {
356  // cout << "break" << endl;
357  break;
358  }
359  }
360 
361  // Update the SAI buffer with the weighted effect of all the active
362  // strobes at the current sample
363  for (size_t si = 0; si < active_strobes.strobe_count(); ++si) {
364  // cout << "si=" << si << endl;
365  // Add the effect of active strobe at correct place in the SAI buffer
366  // Calculate 'delay', the time from the strobe event to now
367  int delay = t - active_strobes.Strobe(si).time;
368 
369  // If the delay is greater than the (user-set)
370  // minimum strobe delay, the strobe can be used
371  if (delay >= min_strobe_delay_idx_ && delay < max_strobe_delay_idx_) {
372  // The value at be added to the SAI
373  double sig = in(o, t);
374  // cout << "sig=" << sig << "\tww=" << active_strobes.Strobe(si).working_weight << "\tdecay=" << decay_factor << endl;
375 
376  // Weight the sample correctly
377  sig *= active_strobes.Strobe(si).working_weight;
378 
379  // Adjust the weight acording to the number of samples until the
380  // next output frame
381  sig *= decay_factor;
382 
383  // Update the temporary SAI buffer
384  // cout << "output1(" << o << "," << delay << ")=" << out(o, delay) + sig << endl;
385  sai_temp_(o, delay) = sai_temp_(o, delay) + sig;
386  }
387  }
388 
389  next_strobes_[o] = next_strobe_index;
390  } // End loop over channels
391 
392  fire_counter_--;
393 
394  // cout << "fire_counter_=" << fire_counter_ << endl;
395 
396  // Check to see if we need to output an SAI frame on this sample
397  if (fire_counter_ <= 0) {
398  // cout << "(fire_counter_ <= 0)" << endl;
399  // Decay the SAI by the correct amount and add the current output frame
400  double decay = pow(sai_decay_factor_, frame_period_samples_);
401 
402  for (int o = 0; o < channel_count_; ++o) {
403  // for (int t = 0; t < _inSamples; ++t ) {
404  for (int t = 0; t < frame_period_samples_; ++t ) {
405  // cout << "output2(" << o << "," << t << ")=" << sai_temp_(o,t) + out(o,t) * decay << endl;
406  out(o, t) = sai_temp_(o,t) + out(o,t) * decay;
407  }
408  }
409 
410  // Zero the temporary signal
411  for (int o = 0; o < sai_temp_.getRows(); ++o) {
412  for (int t = 0; t < sai_temp_.getCols(); ++t) {
413  sai_temp_(o, t) = 0.0;
414  }
415  }
416  fire_counter_ = frame_period_samples_ - 1;
417  }
418  } // End loop over samples
419

421  // for (int t = 0; t < ctrl_inSamples_->to<mrs_natural>(); ++t) {
422  // for (int o = channel_count_; o < channel_count_ * 2; ++o) {
