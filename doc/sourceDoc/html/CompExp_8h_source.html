47 /* How to use the Compressor/Expander Marsystem
48 
49  The system can be setup using the marsystem manager
50  series->addMarSystem(mng.create("CompExp", "compexp"));
51 
52  Options: Threshold can be set to any value between -1.0 and 1
53  Compressor -1.0 <= thresh < 0
54  Expander 0 <= thresh <= 1.0
55  series->updctrl("CompExp/compexp/mrs_real/thresh", thresh);
56 
57  Attack time can be calculated using the following formula: at = 1 - exp(-2.2*T/t_AT)
58  where at = attack time, T = sampling period, t_AT = time parameter 0.00016 < t_AT < 2.6 sec
59  series->updctrl("CompExp/compexp/mrs_real/rolloff", t_AT);
60 
61  Release time can be calculated similar to at time: rt = 1 - exp(-2.2*T/t_RT)
62  where rt = release time, T = sampling period, t_RT = time parameter 0.001 < t_RT < 5.0 msec
63  series->updctrl("CompExp/compexp/mrs_real/rt", t_RT);
64 
65  Slope factor: 0 < slope <= 1.0
66  series->updctrl("CompExp/compexp/mrs_real/rt", slope);
67 */
