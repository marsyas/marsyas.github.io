117 //
118 // With do_log, the signal is first scaled up so that values <1.0 become
119 // negligible. This just rescales the sample values to fill the range of a
120 // 16-bit signed integer, then we lose the bottom bit of resolution. If the
121 // signal was sampled at 16-bit resolution, there shouldn't be anything to
122 // speak of there anyway. If it was sampled using a higher resolution, then
123 // some data will be discarded.
124 //
125 void
126 AimHCL2::myProcess(realvec& in, realvec& out)
127 {
128  mrs_natural o,t;
129  mrs_real _israte = ctrl_israte_->to<mrs_real>();
130  mrs_natural _inObservations = ctrl_inObservations_->to<mrs_natural>();
131  mrs_natural _inSamples = ctrl_inSamples_->to<mrs_natural>();
132  mrs_natural _onSamples = ctrl_onSamples_->to<mrs_natural>();
133  mrs_natural _lowpass_order = ctrl_lowpass_order_->to<mrs_natural>();
134  mrs_bool _do_lowpass = ctrl_do_lowpass_->to<mrs_bool>();
135  mrs_bool _do_log = ctrl_do_log_->to<mrs_bool>();
136 
137  mrs_natural _num_channels = _inObservations;
138  double b = exp(-1.0 / (_israte * time_constant_));
139  double gain = 1.0 / (1.0 - b);
140 
141  for (o = 0; o < _num_channels; ++o) {
142  for (t = 0; t < _inSamples; ++t) {
143  // cout << "in(o,t)=" << in(o,t) << endl;
144  if (in(o,t) < 0.0) {
145  // cout << "s=zero" << endl;
146  out(o, t) = 0.0;
147  } else {
148  double s = in(o,t);
149  if (_do_log) {
150  s *= pow(2.0, 15);
151  if (s < 1.0) s = 1.0;
152  s = 20.0 * log10(s);
153  }
154  // cout << "s=" << s << endl;
155  out(o, t) = s;
156  }
157  }
158  if (_do_lowpass) {
159 
160  for (int j = 0; j < _lowpass_order; j++) {
161  for (int k = 0; k < _onSamples; ++k) {
162  xn_ = out(o,k);
163  yn_ = xn_ + b * yns_[o][j];
164  yns_[o][j] = yn_;
165  // cout << "gain=" << gain << endl;
166  // cout << "yn_=" << yn_ << endl;
167  out(o, k) = yn_ / gain;
168  }
169  }
170  }
171  }
