87 bool
88 SoundFileSink::updateBackend()
89 {
90  delete backend_;
91  backend_ = 0;
92 
93  if (filename_.empty())
94  return false;
95 
96  // check if file exists
97  bool file_exists;
98  {
99  FILE *file = fopen(filename_.c_str(), "wb");
100  file_exists = file != 0;
101  if (file)
102  fclose(file);
103  }
104  if (!file_exists)
105  {
106  MRSWARN("SoundFileSink: Failed to open file for writing: " << filename_);
107  return false;
108  }
109 
110  // try to open file with appropriate format
111  mrs_string::size_type pos = filename_.rfind(".", filename_.length());
112  mrs_string ext;
113  if (pos != mrs_string::npos)
114  ext = filename_.substr(pos);
115 
116  if (ext == ".au")
117  {
118  backend_ = new AuFileSink(getName());
119  }
120  else if (ext == ".wav")
121  {
122  backend_ = new WavFileSink(getName());
123  }
124 #ifdef MARSYAS_LAME
125  else if (ext == ".mp3")
126  {
127  backend_ = new MP3FileSink(getName());
128  }
129 #endif
130  else
131  {
132  MRSWARN("Unknown file extension: " << filename_);
133  return false;
134  }
135 
136  return true;
137 }
