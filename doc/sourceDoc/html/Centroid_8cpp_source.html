54 void
55 Centroid::myProcess(realvec& in, realvec& out)
56 {
57  mrs_natural o,t;
58  // computer centroid of observations for each time sample
59  // using zero and first-order moments
60  for (t = 0; t < inSamples_; t++)
61  {
62  m0_ = 0.0;
63  m1_ = 0.0;
64  for (o=0; o < inObservations_; o++)
65  {
66  m1_ += o * in(o,t);
67  m0_ += in(o,t);
68  }
69  if (m0_ != 0.0)
70  out(0,t) = (m1_ / m0_) / inObservations_;
71  else
72  out(0,t) = 0.5;
73  }

75 }
