276 template<class T>
277 void
278 MarControlValueT<T>::callMarSystemsUpdate()
279 {
280  //must keep a copy of the current value in case
281  //this control is "toggled" in the following update calls
282  //so it can be "reinjected" into all MarSystem::update() methods
283  //(otherwise, only the first MarSystem in the loop below would
284  //get the current value - all the remaining ones would get the value
285  //"toggled" bu the first MarSystem update() call)
286  T tempValue_ = value_;
287 
288  //iterate over all the MarControls that own this MarControlValue
289  //and call any necessary MarSystem updates after this value change
290  for(link_vector::iterator it = links_.begin(); it != links_.end(); ++it)
291  {
292  value_ = tempValue_; //make sure to use the current value, not a "toggled" one
293  MarControl * linked_control = it->first;
294  updateMarSystemFor(linked_control);
295  }
296 }
