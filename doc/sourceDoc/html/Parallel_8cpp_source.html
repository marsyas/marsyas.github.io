162 void Parallel::myProcess(realvec& in, realvec& out)
163 {
164  mrs_natural t,o;
165  mrs_natural inIndex = 0;
166  mrs_natural outIndex = 0;
167  mrs_natural localIndex = 0;
168  child_count_t child_count = marsystems_.size();
169 
170  if (child_count == 1)
171  {
172  marsystems_[0]->process(in, out);
173  }
174  else if (child_count > 1)
175  {
176  for (child_count_t i = 0; i < child_count; ++i)
177  {
178  localIndex = childInfos_[i].inObservations;
179  for (o = 0; o < localIndex; o++)
180  {
181  // avoid reading unallocated memory
182  if ((inIndex + o) < in.getRows()) {
183  for (t = 0; t < inSamples_; t++) //lmartins: was t < onSamples [!]
184  {
185  (*(slices_[2*i]))(o,t) = in(inIndex + o,t);
186  }
187  } else {
188  for (t = 0; t < inSamples_; t++) {
189  (*(slices_[2*i]))(o,t) = 0;
190  }
191  }
192  }
193  inIndex += localIndex;
194  marsystems_[i]->process(*(slices_[2*i]), *(slices_[2*i+1]));
195  localIndex = childInfos_[i].onObservations;
196  for (o = 0; o < localIndex; o++)
197  {
198  for (t = 0; t < onSamples_; t++)
199  {
200  out(outIndex + o,t) = (*(slices_[2*i+1]))(o,t);
201  }
202  }
203  outIndex += localIndex;
204  }
205  }
206  else if(child_count == 0) //composite has no children!
207  {
208  MRSWARN("Parallel::process: composite has no children MarSystems - passing input to output without changes.");
209  out = in;
210  }
211 }
