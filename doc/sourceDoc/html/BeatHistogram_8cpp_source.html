115  for (mrs_natural o=0; o < inObservations_; o++)
116  {
117  for (mrs_natural t = 1; t < inSamples_; t++)
118  {
119  bin = (mrs_natural)(( (2*srate) * 60.0 * factor_ / (t+1)) + 0.5);

123 
124  // optional tempo weight
125 
126  if (getctrl("mrs_bool/tempoWeighting")->to<mrs_bool>())
127  {

131  tempo_weight = exp(0.5 * tempo_weight * tempo_weight);
132  }
133  else
