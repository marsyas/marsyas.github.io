81 void
82 EnhADRess::myProcess(realvec& in, realvec& out)
83 {
84  out.setval(0.0);
85 
86  for(mrs_natural t=0; t < inSamples_; ++t)
87  {
88  for (mrs_natural k=0; k < N4_; k++)
89  {
90  //get left channel spectrum bin
91  if (k==0) //DC bin (i.e. 0)
92  {
93  rel_ = in(0,t);
94  iml_ = 0.0;
95  }
96  else if (k == N4_-1) //Nyquist bin (i.e. N/2)
97  {
98  rel_ = in(1, t);
99  iml_ = 0.0;
100  }
101  else //all other bins
102  {
103  rel_ = in(2*k, t);
104  iml_ = in(2*k+1, t);
105  }
106 
107  //get right channel spectrum bin
108  if (k==0) //DC bin (i.e. 0)
109  {
110  rer_ = in(N2_,t);
111  imr_ = 0.0;
112  }
113  else if (k == N4_-1) //Nyquist bin (i.e. N/2)
114  {
115  rer_ = in(N2_+1, t);
116  imr_ = 0.0;
117  }
118  else //all other bins
119  {
120  rer_ = in(N2_ + 2*k, t);
121  imr_ = in(N2_ + 2*k+1, t);
122  }
123 
124  phaseL_ = atan2(iml_, rel_); //left channel phases
125  phaseR_ = atan2(imr_, rer_); //right channel phases
126 
127  deltaPhase_ = abs(phaseL_ - phaseR_);
128 
129  //wrap phase into the 0~2*PI range
130  deltaPhase_ = (mrs_real)fmod((double)deltaPhase_, (double)2*PI);
131 
132  //left amplitude value
133  Lk_ = sqrt(rel_*rel_ + iml_*iml_);
134 
135  //right amplitude value
136  Rk_ = sqrt(rer_*rer_ + imr_*imr_);
137 
138  if(deltaPhase_ < PI/2)
139  {
140  minLk_ = Lk_ * sin(deltaPhase_);
141  minRk_ = Rk_ * sin(deltaPhase_);
142 
143  if(Lk_ < Rk_) // i.e. minLk < minRk --> sound panned right
144  {
145  //store magnitude to output
146  mrs_real mag = Rk_ - minLk_;
147  // bins with amplitude inferior to -100dB are discarded
148  if(20.0*log10(mag*mag+0.000000001) > -100.0)
149  {
150  out(k,t) = mag;
151 
152  //store phase to output
153  out(k+N4_,t) = phaseR_;
154 
155  //store azimuth to output
156  out(k+N4_*2,t) = 1.0 - Lk_ * cos(deltaPhase_) / Rk_ ; //-1.0-> L; 0.0-> C;
157  }
158  }
159  else if(Lk_ > Rk_) // i.e. minLk > minRk --> sound panned left
160  {
161  //store magnitude to output
162  mrs_real mag = Lk_ - minRk_;
163  // bins with amplitude inferior to -100dB are discarded
164  if(20.0*log10(mag*mag+0.000000001) > -100.0)
165  {
166  out(k,t) = mag;
167 
168  //store phase to output
169  out(k+N4_,t) = phaseL_;
170 
171  //store azimuth to output
172  out(k+N4_*2,t) = Rk_ * cos(deltaPhase_) / Lk_ -1.0; //0.0 -> C; 1.0-> R;
173  }
174  }
175  else if(Lk_ == Rk_) //sound panned at the CENTER
176  {
177  //store magnitude to output
178  mrs_real mag = Lk_ - minRk_;
179  // bins with amplitude inferior to -100dB are discarded
180  if(20.0*log10(mag*mag+0.000000001) > -100.0)
181  {
182  out(k,t) = mag;
183 
184  //store phase to output
185  out(k+N4_,t) = phaseL_; //could have been phaseR_...
186 
187  //store azimuth to output
188  out(k+N4_*2,t) = 0.0; //0.0 -> C;
189  }
190  }
191  }
192  else
193  {
194  if(20.0*log10(Lk_*Lk_+0.000000001) < -100.0)
195  Lk_ = 0.0;
196  if(20.0*log10(Rk_*Rk_+0.000000001) < -100.0)
197  Rk_ = 0.0;
198 
199  if(Lk_ > Rk_)
200  {
201  out(k,t) = Lk_;
202  out(k+N4_,t) = phaseL_;
203  out(k+N4_*2,t) = 0.0; //-1.0;
204  }
205  else if(Rk_ > Lk_)
206  {
207  out(k,t) = Rk_;
208  out(k+N4_,t) = phaseR_;
209  out(k+N4_*2,t) = 0.0; //1.0;
210  }
211  else if(Lk_ == Rk_ && Lk_ != 0.0)
212  {
213  out(k,t) = Lk_;
214  out(k+N4_,t) = phaseL_;
215  out(k+N4_*2,t) = 0.0;
216  }
217  }
218  }
219  }
