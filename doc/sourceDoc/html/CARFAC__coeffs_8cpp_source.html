39 std::vector<double> filter_state_class::FilterStep(CF_class &CF, double input_waves, std::vector<double> &filterstep_detect)
40 {
41  if (!init) {
42  filterstep_inputs.resize(CF.n_ch);
43  filterstep_zA.resize(CF.n_ch);
44  filterstep_zB.resize(CF.n_ch);
45  filterstep_zY.resize(CF.n_ch);
46  filterstep_z1.resize(CF.n_ch);
47  filterstep_z2.resize(CF.n_ch);
48  }
49 
50  // Use each stage previous Y as input to next.
51  filterstep_inputs[0] = input_waves;
52 
53  for (unsigned int i=0; i < zY_memory.size()-1; i++) {
54  filterstep_inputs[i+1] = zY_memory[i];
55  }
56 
57  // AGC interpolation.
58  for (int i=0; i < CF.n_ch; i++) {
59  zB_memory[i] = zB_memory[i] + dzB_memory[i];
60  double filterstep_r = CF.filter_coeffs.r_coeffs[i] - CF.filter_coeffs.c_coeffs[i] * (zA_memory[i] + zB_memory[i]);
61 
62  // Now reduce filter_state by r and rotate with the fixed cos/sin
63  // coeffs.
64  double z1_mem = z1_memory[i];
65  z1_memory[i] = ((filterstep_r *
66  (CF.filter_coeffs.a_coeffs[i] * z1_memory[i] -
67  CF.filter_coeffs.c_coeffs[i] * z2_memory[i]))
68  + filterstep_inputs[i]);
69  z2_memory[i] = filterstep_r * (CF.filter_coeffs.c_coeffs[i] * z1_mem +
70  CF.filter_coeffs.a_coeffs[i] * z2_memory[i]);
71  }
72 
73  // Update the "velocity" for cubic nonlinearity, into zA.
74  for (int i=0; i<CF.n_ch; i++) {
75  zA_memory[i] = pow(((z2_memory[i] - z2_memory[i]) * CF.filter_coeffs.velocity_scale), 2);
76  }
77 
78  for (int i=0; i<CF.n_ch; i++) {
79  // Simulate Sigmoidal OHC effect on damping.
80  zA_memory[i] = (1 - pow((1 - zA_memory[i]), 4)) / 4; // soft max at 0.25
81  // Get outputs from inputs and new z2 values.
82  zY_memory[i] = CF.filter_coeffs.g_coeffs[i] * (filterstep_inputs[i] + CF.filter_coeffs.h_coeffs[i] * z2_memory[i]);
83  }
84 
85  // TODO(dicklyon): Generalize to a detection nonlinearity.
86  double maxval = 0.0;
87  for (int i=0; i<CF.n_ch; i++) {
88  filterstep_detect[i] = zY_memory[i] > maxval ? zY_memory[i] : maxval;
89  }
90 
91  for (int i=0; i<CF.n_ch; i++) {
92  double output = zY_memory[i];
93  double detect = output > 0.0 ? output : 0.0;
94  filterstep_detect[i] = detect;
95  detect_accum[i] += detect;
96  }
97 
98  return filterstep_detect;
99 }

356 // function CF = CARFAC_Design(fs, CF_filter_params, ...
357 // CF_AGC_params, ERB_min_BW, ERB_Q)
358 //
359 // This function designs the CARFAC (Cascade of Asymmetric Resonators with
360 // Fast-Acting Compression); that is, it take bundles of parameters and
361 // computes all the filter coefficients needed to run it.
362 //
363 // fs is sample rate (per second)
364 // CF_filter_params bundles all the PZFC parameters
365 // CF_AGC_params bundles all the AGC parameters
366 //
367 // See other functions for designing and characterizing the CARFAC:
368 // [naps, CF] = CARFAC_Run(CF, input_waves)
369 // transfns = CARFAC_Transfer_Functions(CF, to_channels, from_channels)
370 //
371 // All args are defaultable; for sample/default args see the code; they
372 // make 96 channels at default fs = 22050, 114 channels at 44100.

375 // TODO(snessnet) - Have CARFAC_Design take parameters like the original function
376 void CF_class::CARFAC_Design(double _fs, double _ERB_break_freq, double _ERB_Q)
377 {
378  AGC_coeffs.detect_scale = CF_AGC_params.detect_scale;
379  AGC_coeffs.AGC_stage_gain = CF_AGC_params.AGC_stage_gain;
380  AGC_coeffs.AGC_mix_coeff = CF_AGC_params.AGC_mix_coeff;
381 
382  // TODO(snessnet) - We should get this from israte, but this won't
383  // be setup until we load the soundfile. For now require 22050Hz
384  // sound files.
385  if (_fs == -1) {
386  fs = 22050;
387  }
388 
389  // first figure out how many filter stages (PZFC/CARFAC channels):
390  double pole_Hz = CF_filter_params.first_pole_theta * fs / (2 * PI);
391  n_ch = 0;
392  while (pole_Hz > CF_filter_params.min_pole_Hz) {
393  n_ch = n_ch + 1;
394  pole_Hz = pole_Hz - CF_filter_params.ERB_per_step * ERB_Hz(pole_Hz, _ERB_break_freq, _ERB_Q);
395  }
396 
397  pole_freqs.assign(n_ch,0);
398  pole_Hz = CF_filter_params.first_pole_theta * fs / (2 * PI);
399  for(int ch = 0; ch < n_ch; ch++) {
400  pole_freqs[ch] = pole_Hz;
401  pole_Hz = pole_Hz - CF_filter_params.ERB_per_step * ERB_Hz(pole_Hz, _ERB_break_freq, _ERB_Q);
402  }
403 
404  // Design the AGC and Filters
405  CARFAC_DesignFilters();
406  CARFAC_DesignAGC();
407 }

409 // Calculate the Equivalent Rectangular Bandwith of a given frequency
410 double CF_class::ERB_Hz(double CF_Hz, double ERB_break_freq, double ERB_Q)
411 {
412  if (ERB_Q == -1) {
413  ERB_Q = 1000/(24.7*4.37); // 9.2645
414  }
415 
416  if (ERB_break_freq == -1) {
417  ERB_break_freq = 1000/4.37; // 228.833
418  }

421  return ERB;

424 // From the input filter_params, sampling frequency and
425 // pole_frequencies, create all the filter coefficients.
426 void CF_class::CARFAC_DesignFilters()
427 {
428  int n_ch = (int) pole_freqs.size();
429 
430  filter_coeffs.init(CF_filter_params.velocity_scale, n_ch);
431 
432  // zero_ratio comes in via h. In book's circuit D, zero_ratio is
433  // 1/sqrt(a), and that a is here 1 / (1+f) where h = f*c.
434  // solve for f: 1/zero_ratio^2 = 1 / (1+f)
435  // zero_ratio^2 = 1+f => f = zero_ratio^2 - 1
436  double f = pow(CF_filter_params.zero_ratio,2) - 1; // nominally 1 for half-octave
437 
438  // Make pole positions, s and c coeffs, h and g coeffs, etc., which
439  // mostly depend on the pole angle theta.
440  std::vector<double> theta(n_ch);
441  for (unsigned int i = 0; i < theta.size(); i++) {
442  theta[i] = pole_freqs[i] * (2 * PI / fs);
443  }
444 
445  // Different possible interpretations for min-damping r:
446  // r = exp(-theta * CF_filter_params.min_zeta);
447  std::vector<double> r(n_ch);
448  for (unsigned int i = 0; i < r.size(); i++) {
449  r[i] = (1 - (sin(theta[i]) * CF_filter_params.min_zeta)); // higher Q at highest thetas
450  }
451  filter_coeffs.r_coeffs = r;
452 
453  // Undamped coupled-form coefficients:
454  for (unsigned int i = 0; i < theta.size(); i++) {
455  filter_coeffs.a_coeffs[i] = cos(theta[i]);
456  filter_coeffs.c_coeffs[i] = sin(theta[i]);
457  }
458 
459  // The zeros follow via the h_coeffs
460  std::vector<double> h(n_ch);
461  for (unsigned int i = 0; i < theta.size(); i++) {
462  h[i] = sin(theta[i]) * f;
463  }
464  filter_coeffs.h_coeffs = h;
465 
466  // Aim for unity DC gain at min damping, here; or could try r^2
467  std::vector<double> r2 = r;
468  for (unsigned int i = 0; i < theta.size(); i++) {
469  filter_coeffs.g_coeffs[i] =
470  1 / (1 + h[i] * r2[i] * sin(theta[i]) / (1 - 2 * r2[i] * cos(theta[i]) + pow(r2[i], 2)));
471  }
472 }

474 // Calculate the parameters for the AGC step
475 void CF_class::CARFAC_DesignAGC()
476 {
477  std::vector<double> AGC1_scales = CF_AGC_params.AGC1_scales;
478  std::vector<double> AGC2_scales = CF_AGC_params.AGC2_scales;
479 
480  int n_AGC_stages = CF_AGC_params.n_stages;
481  AGC_coeffs.AGC_epsilon.assign(n_AGC_stages, 0);
482  AGC_coeffs.AGC1_polez.assign(n_AGC_stages, 0);
483  AGC_coeffs.AGC2_polez.assign(n_AGC_stages, 0);
484  int decim = CF_AGC_params.decimation;
485 
486  for (int stage = 0; stage < n_AGC_stages; stage++) {
487  double tau = CF_AGC_params.time_constants[stage];
488 
489  // Epsilon is how much new input to take at each update step.
490  AGC_coeffs.AGC_epsilon[stage] = 1 - exp(-decim / (tau * fs));
491 
492  // And these are the smoothing scales and poles for decimated rate.
493  double ntimes = tau * (fs / decim); // Effective number of times the smoothing is done
494 
495  // Divide the spatial variance by effective number of smoothings:
496  double t = (pow(AGC1_scales[stage],2)) / ntimes; // Adjust scale per step for diffusion
497  AGC_coeffs.AGC1_polez[stage] = 1 + 1/t - sqrt(pow(1+1/t,2) - 1);
498  t = (pow(AGC2_scales[stage],2)) / ntimes; // Adjust scale per step for diffusion.
499  AGC_coeffs.AGC2_polez[stage] = 1 + 1/t - sqrt(pow(1+1/t,2) - 1);
500  }
501 }
