138 // read all the raw data into data[]
139 //
140 void RawFileSource::readData(unsigned long index)//[!]
141 {
142  mrs_natural i;
143  mrs_natural length = bufferSize_;
144 
145  // Read samples into data[].
146  if (fseek(sfp_, index, SEEK_SET) == -1) {
147  MRSWARN("ERROR(fseek): could not read raw file data.");
148  return;
149  }
150 
151  if (fread(buffer_, length, 2, sfp_) != 2 ) {
152  MRSWARN("ERROR(fread): could not read raw file data");
153  return;
154  }
155 
156  // if we are on a little endian machine.
157  byteSwap_ = true;
158  if ( byteSwap_ ) {
159  short* ptr = buffer_;
160  for ( i=length; i>=0; i-- )
161  swap16((unsigned char *)(ptr++));
162  }
163 
164  for ( i=length-1; i>=0; i-- ) {
165  data_(i) = buffer_[i];
166  }
167 
168  // repeat last sample frame for interpolation
169  data_(length) = data_(length-1);
170 
171  // find the peak
172  mrs_real max = 0.0;
173  for (i=0; i < length; ++i) {
174  if (fabs(data_(i)) > max)
175  max = (mrs_real) fabs((double) data_(i));
176  }
177 
178  // now normalize according to the peak.
179  if (max > 0.0) {
180  max = (mrs_real)(1.0 / max);
181  max *= 1.0; // constant factor for now.
182  for ( i=0; i <= length; ++i ) {
183  data_(i) *= max;
184  }
185  }
186 
187 }
