165 void
166 GStreamerSource::init_pipeline()
167 {
168  GstElement *conv;
169 
170  /* Initialize GStreamer */
171  if(!gst_is_initialized() && !gst_init_check(NULL, NULL, NULL)) {
172  MRSERR("GStreamer could not be initialized!");
173  return;
174  }
175 
176  /* Create pipeline and elements */
177  pipe_ = gst_pipeline_new("pipeline");
178  dec_ = gst_element_factory_make("uridecodebin", NULL);
179  conv = gst_element_factory_make("audioconvert", NULL);
180  sink_ = gst_element_factory_make("appsink", NULL);
181 
182  /* Unfloat floating references we wish to access later */
183  gst_object_ref_sink(pipe_);
184  gst_object_ref_sink(dec_);
185  gst_object_ref_sink(sink_);
186 
187  /* Configure pipeline and elements */
188  gst_pipeline_use_clock(GST_PIPELINE(pipe_), NULL);
189 
190  /* Set up the capabilities filtering */
191  GstCaps *caps = gst_caps_new_simple("audio/x-raw",
192  "format", G_TYPE_STRING, g_strdup_printf("F%luLE", sizeof(mrs_real)*8),
193  NULL);
194  gst_app_sink_set_caps(GST_APP_SINK(sink_), caps);
195  gst_caps_unref(caps);
196 
197  /* Add everything to the pipeline and link up static pads */
198  gst_bin_add_many(GST_BIN(pipe_), dec_, conv, sink_, NULL);
199  gst_element_link(conv, sink_);
200 
201  /* Set up callback to link the decoder's dynamic pad to the converter */
202  g_signal_connect(dec_, "pad_added", G_CALLBACK(on_pad_added), conv);
203 
204  pipe_created_ = true;
205 }

208 void
209 GStreamerSource::getHeader(mrs_string filename)
210 {
211  if(!pipe_created_) {
212  init_pipeline();
213  if(!pipe_created_) {
214  // Pipe creation failed
215  size_ = -1;
216  return;
217  }
218  }
219 
220  /* (Re)Set the uridecodebin's filename */
221  gst_element_set_state(pipe_, GST_STATE_NULL);
222  if(gst_uri_is_valid(filename.c_str())) {
223  /* "filename" is already a valid uri, just use it as-is */
224  g_object_set(G_OBJECT(dec_), "uri", filename.c_str(), NULL);
225  } else {
226  g_object_set(G_OBJECT(dec_), "uri", gst_filename_to_uri(filename.c_str(), NULL), NULL);
227  }
228 
229  /* Set to PAUSED so we can get a preroll buffer */
230  GstStateChangeReturn ret = gst_element_set_state(pipe_, GST_STATE_PAUSED);
231  if(ret == GST_STATE_CHANGE_FAILURE) {
232  MRSERR("GStreamer pipeline failed to change state. This could be due to an invalid filename");
233  }
234  GstSample *sample = gst_app_sink_pull_preroll(GST_APP_SINK(sink_));
235 
236  /* Grab the sample's caps so we can get some useful info */
237  GstCaps *samp_caps = gst_sample_get_caps(sample);
238  GstStructure *samp_struct = gst_caps_get_structure(samp_caps, 0);
239 
240  /* Get sample rate from buffer */
241  gint rate;
242  if(gst_structure_get_int(samp_struct, "rate", &rate)) {
243  osrate_ = (mrs_real)rate;
244  }
245 
246  /* Get number of channels from buffer */
247  gint channels;
248  if(gst_structure_get_int(samp_struct, "channels", &channels)) {
249  onObservations_ = channels;
250  }
251 
252  /* Get the rough size of the audio. Use GST_FORMAT_TIME since it works with most file types */
253  gint64 duration;
254  GstFormat format = GST_FORMAT_TIME;
255 
256  // Force blocking on state change completion
257  gst_element_get_state(pipe_, NULL, NULL, GST_SECOND);
258  if (gst_element_query_duration(GST_ELEMENT(pipe_), format, &duration) && format==GST_FORMAT_TIME) {
259  size_ = (mrs_natural)(duration*1e-9*osrate_ + 0.5);
260  } else {
261  /* GStreamer can't tell us the size of the stream */
262  MRSWARN("Query Duration failed:");
263  size_ = -1;
264  }
265 
266  /* Reset these in case this isn't our first song */
267  pos_ = 0;
268  hasData_ = true;
269  lastTickWithData_ = false;
270 
271  /* Clean up */
272  gst_caps_unref(samp_caps);
273 
274  /* Set Controls */
275  //setctrl("mrs_real/israte", osrate_);
276  setctrl("mrs_real/osrate", osrate_);
277  setctrl("mrs_natural/onObservations", onObservations_);
278  //setctrl("mrs_natural/inObservations", onObservations_);
279  setctrl("mrs_natural/size", size_);
280  setctrl("mrs_natural/pos", pos_);
281  ctrl_hasData_->setValue(hasData_);
282  ctrl_lastTickWithData_->setValue(lastTickWithData_);
283  ctrl_currentHasData_->setValue(hasData_);
284  ctrl_currentLastTickWithData_->setValue(lastTickWithData_);
285 
286 
287  /* Start playing, so queue fills with buffers [Should we do this?] */
288  gst_element_set_state(pipe_, GST_STATE_PLAYING);
289  playing_ = true; // Should we check if set_state worked before doing this?
290 }

295  realvec& vec,

311 mrs_bool
312 GStreamerSource::pull_buffer() {
313  GstSample *sample = gst_app_sink_pull_sample(GST_APP_SINK(sink_));
314 
315  if(sample == NULL) {
316  if(gst_app_sink_is_eos(GST_APP_SINK(sink_))) {
317  /* EOS, stop pulling buffers */
318  hasData_ = false;
319  lastTickWithData_ = true;
320 
321  ctrl_hasData_->setValue(hasData_);
322  ctrl_lastTickWithData_->setValue(lastTickWithData_);
323  ctrl_currentHasData_->setValue(hasData_);
324  ctrl_currentLastTickWithData_->setValue(lastTickWithData_);
325  playing_ = false;
326  return false;
327  } else {
328  /* Something messed up. Should we try pulling again? */
329  MRSERR("GStreamer Error");
330  return false;
331  }
332  } else {
333  buffer_ = gst_sample_get_buffer(sample);
334  buffer_size_ = gst_buffer_get_size(buffer_) / sizeof(mrs_real) / onObservations_;
335  buffer_left_ = buffer_size_;
336  }
337 
338  return true;
339 }

342 void
343 GStreamerSource::myProcess(realvec& in, realvec& out)
344 {
345  (void)in; // Suppress warning from unused parameter
346 
347  // TODO Should output_size be calculated from output.getCols() or in/onSamples_?
348  mrs_natural output_size = inSamples_;
349  mrs_natural output_left = output_size;
350  //mrs_real *output = out.getData();
351 
352  while(output_left > 0) {
353  if(buffer_left_ == 0) {
354  /* Current buffer empty, so pull a new one */
355  if(!pull_buffer()) {
356  break;
357  }
358  }
359 
360  if(buffer_left_ >= output_left) {
361  /* Copy enough to fill the output */
362  /* GSt_BUFFER_DATA() returns guint8*, i.e. a byte-pointer, while output it a double-pointer */
363  // memcpy(output + output_size - output_left,
364  // GST_BUFFER_DATA(buffer_) + (buffer_size_ - buffer_left_)*sizeof(mrs_real),
365  // output_left*sizeof(mrs_real));
366  copyFromBuffer(buffer_, buffer_size_-buffer_left_, out, output_size-output_left, output_left);
367  buffer_left_ -= output_left;
368  output_left = 0;
369  } else {
370  /* Copy what's left in the buffer */
371  // memcpy(output + (output_size - output_left),
372  // GST_BUFFER_DATA(buffer_) + (buffer_size_ - buffer_left_)*sizeof(mrs_real),
373  // buffer_left_*sizeof(mrs_real));
374  copyFromBuffer(buffer_, buffer_size_-buffer_left_, out, output_size-output_left, buffer_left_);
375  output_left -= buffer_left_;
376  buffer_left_ = 0;
377  }
378 
379  if(buffer_left_ == 0 && buffer_ != NULL) {
380  /* Buffer empty, unref it */
381  gst_buffer_unref(buffer_);
382  }
383  }
384 
385  // /* Make sure unfilled output is zeroed (TODO is this necessary?) */
386  // mrs_natural i;
387  // for(i = output_size-1; i >= output_size-output_left; --i) {
388  // out(0,i) = 0.0;
389  // }
390 
391  /* Update current position */
392  pos_ += output_size - output_left;
393  setctrl("mrs_natural/pos", pos_);
394 }

396 mrs_bool
397 GStreamerSource::seek()
398 {
399  GstSeekFlags flags = (GstSeekFlags)(GST_SEEK_FLAG_FLUSH | GST_SEEK_FLAG_KEY_UNIT);
400  gint64 seek_to = (gint64)(newpos_/osrate_*1e9);
401  if(seek_to >= 0) {
402  gst_element_seek_simple(GST_ELEMENT(pipe_), GST_FORMAT_TIME, flags, seek_to);
403  } else {
404  MRSWARN("Tried to seek to negative position");
405  return false;
406  }
407 
408  GstStateChangeReturn res = gst_element_get_state(pipe_, NULL, NULL, GST_SECOND);
409  if(res != GST_STATE_CHANGE_SUCCESS) {
410  MRSERR("Seek failed!");
411  return false;
412  }
413 
414  gint64 cur;
415  GstFormat format = GST_FORMAT_TIME;
416  if(!gst_element_query_position(GST_ELEMENT(pipe_), format, &cur)) {
417  MRSERR("Position query failed!");
418  return false;
419  }
420 
421  pos_ = cur*1e-9*osrate_;
422  buffer_left_ = 0;
423  return true;
424 }
