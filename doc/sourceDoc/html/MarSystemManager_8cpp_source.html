1503 MarSystemManager::getMarSystem(istream& is, MarSystem *parent)
1504 {
1505  /* Create a MarSystem object from an input stream in .mpl format
1506  * ( this is the format created by MarSystem::put(ostream& o) )
1507  */
1508  mrs_string skipstr;
1509  mrs_string mcomposite;
1510  mrs_natural i;
1511  bool isComposite;
1512  mrs_string marSystem = "MarSystem";
1513  mrs_string marSystemComposite = "MarSystemComposite";
1514 
1515  /* first line looks like:
1516  * # marSystem(Composite)
1517  */
1518  is >> skipstr;
1519  is >> mcomposite;
1520  if (mcomposite == marSystem)
1521  isComposite = false;
1522  else if (mcomposite == marSystemComposite)
1523  isComposite = true;
1524  else
1525  {
1526  MRSERR("Unknown MarSystemType" << mcomposite);
1527  MRSERR("skipstr = " << skipstr);
1528  return 0;
1529  }
1530 
1531  /* next line looks like:
1532  * # Type = MarSystemSubclass
1533  */
1534  is >> skipstr >> skipstr >> skipstr;
1535  mrs_string mtype;
1536  is >> mtype;
1537 
1538 
1539  /* next line looks like:
1540  * # Name = mname
1541  */
1542  is >> skipstr >> skipstr >> skipstr;
1543  mrs_string mname;
1544  is >> mname;
1545 
1546  MarSystem* msys = getPrototype(mtype);
1547 
1548  if (msys == 0)
1549  {
1550  if (compositesMap_.find(mtype) == compositesMap_.end())
1551  {
1552  MRSERR("MarSystem::getMarSystem - MarSystem " << mtype << " is not yet part of Marsyas");
1553  return 0;
1554  }
1555  else
1556  {
1557  // lazy composite registration
1558  registerComposite(mtype);
1559  msys = getPrototype(mtype);
1560  }
1561  }
1562 
1563  msys->setName(mname);
1564  msys->setParent(parent);
1565 
1566  //delete all children MarSystems in a (prototype) Composite
1567  //and read and link (as possible) local controls
1568  is >> *msys;
1569 
1570  msys->update();
1571 
1572  workingSet_[msys->getName()] = msys; // add to workingSet
1573 
1574  //recreate the Composite destroyed above, relinking all
1575  //linked controls in its way
1576  if (isComposite)
1577  {
1578  /* If this is a composite system, we may have subcomponents
1579  * the number of subcomponents will be listed like this:
1580  * (blank line)
1581  * # nComponents = 3
1582  * (blank line)
1583  * Here, we read in the nComponents, then instantiate each component
1584  */

1587  is >> nComponents;
