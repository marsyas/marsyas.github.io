94 int
95 PeakClusterSelect::partition(realvec& rv, mrs_natural dimension, mrs_natural left, mrs_natural right, mrs_bool sortColumns)
96 {
97  // Not quite fair, but good enough random partitioning
98  int pivot_i = rand()%(right-left+1) + left;
99  // Place pivot val at the end of the series
100  swap(rv, pivot_i, right, sortColumns);
101 
102  mrs_real pivot_val;
103  if( sortColumns )
104  pivot_val = rv(dimension,right);
105  else
106  pivot_val = rv(right,dimension);
107 
108  int i = left-1;
109 
110  if( sortColumns )
111  {
112  for( int j=left ; j<right ; j++ )
113  {
114  if( rv(dimension,j) <= pivot_val )
115  {
116  ++i;
117  swap(rv, i, j, sortColumns);
118  }
119  }
120  // re-insert pivot val
121  swap(rv, i+1, right, sortColumns);
122  }
123  else
124  {
125  for( int j=left ; j<right ; j++ )
126  {
127  if( rv(j,dimension) <= pivot_val )
128  {
129  ++i;
130  swap(rv, i, j, sortColumns);
131  }
132  }
133  // re-insert pivot val
134  swap(rv, i+1, right, sortColumns);
135  }
136 
137  return i+1;
138 }

169 void
170 PeakClusterSelect::myProcess(realvec& in, realvec& out)
171 {
172  mrs_natural t;
173  mrs_natural numClustersToKeep = ctrl_numClustersToKeep_->to<mrs_natural>();
174  mrs_natural curNumClusters=-1, i, j, k, curClusterLabel;
175  mrs_natural numPeaks = ctrl_inSamples_->to<mrs_natural>();
176 
177  // Determine number of clusters in input by finding maximum index plus 1
178  for (t = 0; t < inSamples_; t++)
179  if( in(0,t) > curNumClusters )
180  curNumClusters = (mrs_natural)in(0,t);
181  curNumClusters += 1;
182 
183  mrs_realvec intraClusterSimilarities(3, curNumClusters);
184  mrs_realvec clusterSimilarity (curNumClusters, curNumClusters);
185  mrs_realvec clusterNorm (curNumClusters, curNumClusters);
186  mrs_realvec keepMe(curNumClusters);
187  mrs_real intraSimKeepThresh = .5;
188 
189  clusterSimilarity.setval (0.);
190  clusterNorm.setval (0.);
191  keepMe.setval (1.);
192 
193  for( i=0 ; i<curNumClusters ; ++i )
194  {
195  // Store cluster index in first row
196  // (so upon sorting of density values,
197  // we have reference to the cluster)
198  intraClusterSimilarities(0,i) = i;
199  // Keep track of the number of peaks in each cluster
200  intraClusterSimilarities(1,i) = 0;
201  // Finally, keep track of the sum of intra-cluster similarities
202  intraClusterSimilarities(2,i) = 0;
203  }
204 
205 
206  // compute the similarity of each peak to each each other
207  // accumulate per cluster pair
208  for( i=0 ; i<numPeaks ; ++i )
209  {
210  mrs_realvec numAcc(curNumClusters);
211  mrs_realvec similarity(curNumClusters);
212 
213  numAcc.setval (0.);
214  similarity.setval (0.);
215  for( j=0 ; j<numPeaks ; j++ )
216  {
217  if (i==j)
218  continue;
219  similarity((mrs_natural)(in(0,j)+.1)) += in (i+1, j);
220  clusterNorm ((mrs_natural)(in(0,i)+.1),(mrs_natural)(in(0,j)+.1)) += 1;
221  }
222  for (k = 0; k < curNumClusters; k++)
223  clusterSimilarity ((mrs_natural)(in(0,i)+.1), k) += similarity(k);
224  }
225 
226  for (i = 0; i < curNumClusters; i++)
227  for (j = 0; j < curNumClusters; j++)
228  clusterSimilarity(i,j) /= (clusterNorm(i,j)>0)? clusterNorm(i,j) : 1.;
229 
230  // compute something similar to silhouette coefficients
231  mrs_realvec silhouetteCoeffs (curNumClusters);
232  silhouetteCoeffs.setval(0.);
233 
234  for (k = 0; k < curNumClusters; k++)
235  {
236  mrs_real selfSim, maximum, minSim = 0;//1e37;
237  selfSim = clusterSimilarity(k,k);
238  for (i = 0; i < curNumClusters; i++)
239  {
240  if (i == k)
241  continue;
242  //minSim = (clusterSimilarity(k,i) < minSim)?clusterSimilarity(k,i) : minSim;
243  minSim += clusterSimilarity(k,i);
244  }
245  minSim /= (curNumClusters-1);
246 
247  if ((maximum = max(selfSim, minSim)) != 0)
248  silhouetteCoeffs(k) = (selfSim - minSim)/ maximum;
249  }
250 
251  // update output values
252  for (k = 0; k < curNumClusters; k++)
253  intraClusterSimilarities(2,k) = clusterSimilarity(k,k);
254 
255 #ifdef MARSYAS_MATLAB
256 #ifdef MTLB_DBG_LOG
257  //MATLAB_PUT(clusterSimilarity, "sim");
258  //MATLAB_PUT(curNumClusters, "numClust");
259  //MATLAB_EVAL ("figure(13),imagesc(0:(numClust-1),0:(numClust-1),sim,[0 1]),colorbar");
260  //MATLAB_PUT(silhouetteCoeffs, "sil");
261  //MATLAB_EVAL ("figure(14),plot(0:(length(sil)-1), sil),axis([0 length(sil)-1 0 1]), grid on");
262 #endif
263 #endif
264 
265 
266 
267  // (Quick) Sort by intra-cluster similarity density
268  sort( intraClusterSimilarities, 2, 0, curNumClusters-1 );
269 
270 
271 #ifdef DBG_FILE_OUT
272  if (outTextFile.good ())
273  {
274  if (curNumClusters < 6)
275  {
276  for (k = 0; k < 6-curNumClusters; k++)
277  outTextFile << 0 <<"\t";
278  }
279  for (k = 0; k < curNumClusters; k++)
280  outTextFile << silhouetteCoeffs(k) <<"\t"; ;
281  //outTextFile << intraClusterSimilarities(2,k) <<"\t"; ;
282 
283  outTextFile << std::endl;
284  }
285 #endif
286 
287  // ignore setting of numclusters2Keep and set it by threshold of the intraclustersimilarity
288  if (numClustersToKeep == 0)
289  {
290  const mrs_real intraThreshBounds[2] = {.3,.6};
291  const mrs_real silThreshBound = 1./curNumClusters;
292  intraSimKeepThresh = max((mrs_real)intraThreshBounds[0],min((mrs_real)intraThreshBounds[1],(mrs_real).5*((mrs_real)intraClusterSimilarities(2,0) + (mrs_real)intraClusterSimilarities(2, curNumClusters-1))));
293  for (k = 0; k < curNumClusters; k++)
294  {
295  // intra cluster similarity threshold
296  if (intraClusterSimilarities(2,k) < intraSimKeepThresh)
297  keepMe(k) = 0;
298 
299  // silhouette coefficient threshold
300  if (silhouetteCoeffs((mrs_natural)(intraClusterSimilarities(0,k)+.1)) < silThreshBound)
301  keepMe(k) = 0;
302  }
303  numClustersToKeep = (mrs_natural)(keepMe.sum () +.1);
304 
305  if (numClustersToKeep == curNumClusters)
306  keepMe(0) = 0; // always abandon one cluster!
307  //std::cout << numClustersToKeep << " " << std::endl;
308  }
309  else
310  {
311  for (k = 0; k < (curNumClusters - numClustersToKeep); k++)
312  keepMe(k) = 0;
313  }
314 
315  for (t = 0; t < inSamples_; t++)
316  {
317  curClusterLabel = (mrs_natural)in(0,t);
318  out(0,t) = curClusterLabel; //signals clusters to be synthesized
319 
320  for( k=0 ; k < curNumClusters; k++)
321  {
322  if( curClusterLabel == intraClusterSimilarities(0,k) )
323  {
324  if (keepMe(k) < .5)
325  out(0,t) = (curClusterLabel)? -curClusterLabel : -1; //negative value signals clusters to be discarded, zero is mapped to -1
326  break;
327  }
328  }
329  }
330 }
