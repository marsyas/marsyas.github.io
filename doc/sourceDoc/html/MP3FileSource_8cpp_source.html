756 void
757 MP3FileSource::fillStream( mrs_natural target )
758 {
759 
760  // fill the input buffer
761  if (stream.buffer == NULL || stream.error == MAD_ERROR_BUFLEN)
762  {
763 
764  register mrs_natural remaining = 0;
765  register mrs_natural chunk = INPUT_BUFFER_SIZE;
766 
767  // cout << "offset = " << offset << endl;
768 
769  // when called with the default parameter, carry on decoding...
770  if ( stream.next_frame != NULL ) {
771  offset = stream.next_frame - ptr_;
772  remaining = fileSize_ - offset;
773  } else if ( target != 0 ) {
774  // we have seeked somewhere in the file...
775  offset = target;
776  remaining = fileSize_ - offset;
777  } else if ( target == -1 ) {
778  // we rewound the track...
779  offset = 0;
780  remaining = fileSize_;
781  }
782 
783 
784  // there may not be enough to fill the buffer
785  if ( remaining < INPUT_BUFFER_SIZE ) {
786  chunk = remaining + MAD_BUFFER_GUARD;
787  }
788 
789  // if we have hit the end...
790  if ( offset >= fileSize_ ) {
791  hasData_ = false;
792  // MRSWARN("MP3FileSource: cannot seek to offset");
793  } else {
794  // fill the mad buffer
795  if (offset == -1)
796  offset = 1;
797 
798  mad_stream_buffer(&stream, ptr_ + offset, chunk);
799 
800  stream.error = MAD_ERROR_NONE;
801  }
802  }
803 }
