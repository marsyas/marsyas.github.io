283 mrs_real
284 ERB::gain(mrs_real x, mrs_real y)
285 {
286  mrs_real r1(2.0f*x*PI/fs);
287  mrs_real r2(sqrt(3.0f-sqrt(8.0f)));
288  mrs_real r3(sqrt(3.0f+sqrt(8.0f)));
289  mrs_real z1r(cos(2.0f*r1)),z1i(sin(2.0f*r1));
290  mrs_real z2r(cos(r1)),z2i(sin(r1));
291  mrs_real z3r(-2.0f*z1r/fs),z3i(-2.0f*z1i/fs);
292  mrs_real z4r(2.0f*exp(-y/fs)*z2r/fs),z4i(2.0f*exp(-y/fs)*z2i/fs);
293  return abs(z3r + z4r*(cos(r1)-r2*sin(r1)),z3i + z4i*(cos(r1)-r2*sin(r1)))*
294  abs(z3r + z4r*(cos(r1)+r2*sin(r1)),z3i + z4i*(cos(r1)+r2*sin(r1)))*
295  abs(z3r + z4r*(cos(r1)-r3*sin(r1)),z3i + z4i*(cos(r1)-r3*sin(r1)))*
296  abs(z3r + z4r*(cos(r1)+r3*sin(r1)),z3i + z4i*(cos(r1)+r3*sin(r1)))/
297  pow(abs(-2.0f/exp(2.0f*y/fs) - 2.0f*z1r + 2.0f*(1.0f + z1r)/exp(y/fs),2.0f*z1i*(-1.0f + 1.0f/exp(y/fs))),4);
298 }
