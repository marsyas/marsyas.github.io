62 void
63 MarSystemTemplateAdvanced::addControls()
64 {
65  someString_ = "";
66 
67  // register new custom control in MarControlManager
68  MarControlManager *mcm = MarControlManager::getManager();
69  if (!mcm->isRegistered("mrs_myheader"))
70  {
71  mcm->registerPrototype("mrs_myheader", new MyHeaderT());
72  // if we do not define a typedef for our custom control
73  // (see .h), we must then use a equivalent, but somewhat
74  // more verbose call:
75  //
76  // mcm->registerPrototype("mrs_myheader", new MarControlValueT<MyHeader>());
77  }
78 
79  // Ask MarControlManager to create a new instance
80  // of our "custom" myHeader control
81  ctrl_header_ = mcm->create("mrs_myheader");
82 
83  // create a temporary header and fill it
84  // with the desired parameter values
85  MyHeader myh;
86  myh.someString = "abcd";
87  myh.someValue = 50;
88  myh.someFlag = true;
89  myh.someVec.create(10);
90 
91  // Alternatively, we could ask our newly created control
92  // for a copy of it that we can then use for setting its parameters
93  // (no advantage to get a reference, because it would const...)
94  // MyHeader myh = ctrl_header_->to<MyHeader>();
95  // myh.someString = "abcd";
96  // myh.someValue = 50;
97  // myh.someFlag = true;
98  // myh.someVec.create(10);
99 
100  // use the temporary header object to set
101  // our custom control values
102  ctrl_header_->setValue(myh);
103 
104  // finally add it as control to this MarSystem, and remember
105  // to register its member variable (third parameter)
106  addctrl("mrs_myheader/hdrname", ctrl_header_, ctrl_header_);
107 
108  // it's also possible to set a custom control with state!
109  ctrl_header_->setState(true);
110 }
