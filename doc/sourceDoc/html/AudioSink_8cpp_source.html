241  RtAudio::Api rt_audio_api = RtAudio::UNSPECIFIED;
242  if (!backend.empty()) {
243  if (backend == "jack")
244  rt_audio_api = RtAudio::UNIX_JACK;
245  else if (backend == "alsa")
246  rt_audio_api = RtAudio::LINUX_ALSA;
247  else if (backend == "pulse")
248  rt_audio_api = RtAudio::LINUX_PULSE;
249  else if (backend == "oss")
250  rt_audio_api = RtAudio::LINUX_OSS;
251  else if (backend == "core-audio")
252  rt_audio_api = RtAudio::MACOSX_CORE;
253  else if (backend == "asio")
254  rt_audio_api = RtAudio::WINDOWS_ASIO;
255  else if (backend == "direct-sound")
256  rt_audio_api = RtAudio::WINDOWS_DS;
257  else
258  MRSWARN("AudioSink: audio backend '" << backend << "' not supported.");
259  }
260 
261  if (audio_ == NULL)
262  audio_ = new RtAudio(rt_audio_api);
263  else if (audio_->isStreamOpen())
264  audio_->closeStream();
265 
266  int device_id = (int) getctrl("mrs_natural/device")->to<mrs_natural>();
267  if (device_id == 0)
268  {
269  device_id = audio_->getDefaultOutputDevice();
270  }
271 
272  std::vector<unsigned int> supported_sample_rates;
273  supported_sample_rates = audio_->getDeviceInfo(device_id).sampleRates;
274 
275  unsigned int target_sample_rate = *(supported_sample_rates.begin());
276  for (std::vector<unsigned int>::iterator i = supported_sample_rates.begin();
277  i != supported_sample_rates.end(); i++)
278  {
279 
280  ostringstream msg;
281  msg << *i << "-" << *sample_rate;
282 
283  if (*i == (unsigned int) *sample_rate) {
284  target_sample_rate = *sample_rate;
285  break;
286  }
287  }
288 
289  if (target_sample_rate != (unsigned int) *sample_rate) {
290  ostringstream msg;
291  msg << "AudioSink: Audio device does not support sample rate " << *sample_rate << "Hz. Resampling to " << target_sample_rate << "Hz.";
292  MRSWARN(msg.str());
293  }
294  *sample_rate = (mrs_natural) target_sample_rate;
295 
296 
297  // expand mono to stereo
298  channel_count = std::max((unsigned int) 2, channel_count);
299 
300  RtAudio::StreamParameters output_params;
301  output_params.deviceId = device_id;
302  output_params.nChannels = channel_count;
303  output_params.firstChannel = 0;
304 
305  RtAudio::StreamOptions options;
306  options.streamName = "Marsyas";
307  options.numberOfBuffers = 0; // Use default.
308  options.flags = RTAUDIO_SCHEDULE_REALTIME;
309  options.priority = 70;
310  if (realtime)
311  options.flags |= RTAUDIO_MINIMIZE_LATENCY; // Use smallest possible 'numberOfBuffers'.
312 
313  RtAudioFormat format = (sizeof(mrs_real) == 8) ? RTAUDIO_FLOAT64 : RTAUDIO_FLOAT32;
314 
315  // Suppress useless warnings when both an AudioSource and
316  // an AudioSink are being opened using ALSA:
317  audio_->showWarnings(false);
318 
319  try
320  {
321  audio_->openStream(&output_params, NULL, format, *sample_rate,
322  block_size, &playCallback, (void *)&shared, &options);
323  }
324  catch (RtError& e)

326  MRSERR("AudioSink: RtAudio error:");
327  e.printMessage();

329  }

332 }

372 bool AudioSink::reformatBuffer(mrs_natural source_block_size,
373  mrs_natural dest_block_size,
374  mrs_natural channel_count,
375  bool realtime, bool resize)
376 {
377 
378  mrs_natural new_capacity = source_block_size + dest_block_size + 1;
379  if (!realtime)
380  new_capacity = std::max( new_capacity * 4, (mrs_natural) 2000 );
381 
382  if (resize)
383  {
384  assert(stopped_);
385  mrs_natural size = new_capacity * 2;
386  if (size != shared.buffer.samples() || channel_count != shared.buffer.observations())
387  {
388  bool do_clear_buffer = true;
389  shared.buffer.resize(channel_count, size, new_capacity, do_clear_buffer);
390  }
391  else
392  {
393  shared.buffer.set_capacity(new_capacity);
394  }
395  shared.watermark = realtime ? 0 : new_capacity / 2;
396  }
397  else
398  {
399  if (channel_count != shared.buffer.observations()
400  || new_capacity > shared.buffer.samples())
401  {
402  MRSERR("AudioSink: Can not set requested buffer capacity or channel count without"
403  " resizing the buffer!");
404  return false;
405  }
406 
407  //cout << "Changing capacity: " << new_capacity << "/" << shared.buffer.samples() << endl;
408  unsigned int new_watermark = realtime ? 0 : new_capacity / 2;;
409  if (new_capacity > shared.buffer.capacity())
410  {
411  // First increase capacity, then watermark.
412  shared.buffer.set_capacity(new_capacity);
413  shared.watermark = new_watermark;
414  }
415  else
416  {
417  // First decrease watermark, then capacity.
418  shared.watermark = new_watermark;
419  shared.buffer.set_capacity(new_capacity);
420  }
421  }
422 
423  return true;
424 }

426 void
427 AudioSink::myProcess(realvec& in, realvec& out)
428 {
429  // always pass input unchanged
430  out = in;
431 
432  //check if RtAudio is initialized
433  if (!isInitialized_)
434  return;
435 
436  // assure that RtAudio thread is running
437  // (this may be needed by if an explicit call to start()
438  // is not done before ticking or calling process() )
439  if ( stopped_ )
440  {
441  start();
442  }
443 
444  //check MUTE
445  if(ctrl_mute_->isTrue())
446  {
447  return;
448  }
449 
450  if (is_resampling_)
451  resampler_->process(in, resampler_output_);
452 
453  const realvec & source = is_resampling_ ? resampler_output_ : in;
454  mrs_natural out_samples = source.getCols();
455  mrs_natural out_observations = source.getRows();
456 
457  realvec_queue_producer producer(shared.buffer, out_samples);
458 
459  if ((mrs_natural) producer.capacity() < out_samples)
460  {
461  // cout << "Producer waiting..." << endl;
462 
463  auto resume_condition = [&]()
464  {
465  // cout << "Producer awake..." << endl;
466  bool ok = producer.reserve(out_samples);
467  if (shared.watermark > 0)
468  ok = ok && shared.buffer.write_capacity() >= shared.watermark;
469  return ok;
470  };
471 
472  std::unique_lock<std::mutex> locker(shared.mutex);
473 
474  shared.condition.wait ( locker, resume_condition );
475 
476  locker.unlock();
477  //cout << "Producer continuing..." << endl;
478  }
479 
480  for (mrs_natural t=0; t < out_samples; t++)
481  {
482  for (mrs_natural o=0; o < out_observations; o++)
483  producer(o,t) = source(o,t);
484  }
485 }

488 int
489 AudioSink::playCallback(void *outputBuffer, void *inputBuffer,
490  unsigned int nFrames, double streamTime,
491  unsigned int status, void *userData)
492 {
493  (void) inputBuffer;
494  (void) streamTime;
495  (void) status;
496 
497 
498  mrs_real* data = (mrs_real*)outputBuffer;
499  OutputData &shared = *((OutputData*)userData);
500  unsigned int nChannels = shared.channel_count;
501 
502  //cout << "-- Consuming " << nFrames << endl;
503 
504  if (shared.underrun)
505  shared.underrun = shared.buffer.read_capacity() <= shared.watermark;
506 
507  if (!shared.underrun)
508  {
509  // Limit scope of realvec_queue_consumer!
510  realvec_queue_consumer consumer(shared.buffer, nFrames);
511 
512  if (consumer.capacity() >= nFrames)
513  {
514  for (unsigned int t=0; t < nFrames; t++)
515  {
516  unsigned int t2 = t * 2;
517  if (nChannels == 1)
518  {
519  mrs_real val = consumer(0, t);
520  data[t2] = val;
521  data[t2+1] = val;
522  }
523  else
524  {
525  for (unsigned int ch=0; ch < nChannels; ch++) {
526  data[nChannels*t+ch] = consumer(ch, t);
527  }
528  }
529  }
530  //cout << "-- Consumed." << endl;
531  }
532  else
533  {
534  shared.underrun = true;
535  MRSWARN("AudioSink: buffer underrun!");
536  }
537  }
538 
539  if (shared.underrun)
540  {
541  // write silence:
542  nChannels = std::max(nChannels, (unsigned int) 2);
543  std::memset(outputBuffer, 0, nChannels * nFrames * sizeof(mrs_real));
544  }
545 
546  shared.mutex.lock();
547  shared.condition.notify_all();
548  shared.mutex.unlock();
549 
550  return 0;
551 }
