169 MP3FileSink::myProcess(realvec& in, realvec& out)
170 {
171 
172 
173  mrs_natural t,o;
174 
175  // copy input to output
176  for (o=0; o < inObservations_; o++)
177  for (t=0; t < inSamples_; t++)
178  {
179  out(o,t) = in(o,t);
180  }
181 
182  mrs_natural encodeCheck=-1;
183 
184  // fill left buffer
185  for (t=0; t < inSamples_; t++)
186  leftpcm_[t] = (short)(in(0,t) * PCM_MAXSHRT);
187  if (inObservations_>1) {
188 // fill right buffer
189  for (t=0; t < inSamples_; t++)
190  rightpcm_[t] = (short)(in(1,t) * PCM_MAXSHRT);
191 
192  encodeCheck = lame_encode_buffer(gfp_,
193  leftpcm_, rightpcm_,
194  inSamples_, mp3Buffer_, 0);
195  }
196  else
197  encodeCheck = lame_encode_buffer(gfp_,
198  leftpcm_, leftpcm_,
199  inSamples_, mp3Buffer_,0); // [ML] 0 do not check for buffer size, should be sizeof(mp3Buffer_) but check fails
200 
201 
202  /* was our output buffer big enough? */
203  if (encodeCheck < 0) {
204  if (encodeCheck == -1) {
205  MRSWARN("mp3 buffer is not big enough... \n");
206  }
207  else
208  MRSWARN("mp3 internal error");
209  }
210  mrs_natural owrite = (int) fwrite(mp3Buffer_, 1, encodeCheck, sfp_);
211  if (owrite != encodeCheck) {
212  MRSWARN("Writing MP3 data to "+filename_+" failed.");
213  }

216 
217 }
