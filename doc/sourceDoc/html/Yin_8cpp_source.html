89 double Yin::vec_quadint_min(realvec *x,unsigned int pos, unsigned int span) {
90  double step = 1./200.;
91  /* init resold to - something (in case x[pos+-span]<0)) */
92  double res, frac, s0, s1, s2, exactpos = (double)pos, resold = 100000.;
93  if ((pos > span) && (pos < x->getSize()-span)) {
94  //s0 = (*x)(0,pos-span);
95  //s1 = (*x)(0,pos);
96  //s2 = (*x)(0,pos+span);
97  s0 = (*x)(pos-span);
98  s1 = (*x)(pos);
99  s2 = (*x)(pos+span);
100  /* increase frac */
101  for (frac = 0.; frac < 2.; frac = frac + step) {
102  res = Yin::aubio_quadfrac(s0, s1, s2, frac);
103  if (res < resold) {
104  resold = res;
105  } else {
106  exactpos += (frac-step)*span - span/2.;
107  break;
108  }
109  }
110  }
111  return exactpos;
112 }

128 void
129 Yin::myProcess(realvec& in, realvec& out)
130 {
131 
132  // The tolerance for the yin algorithm
133  const mrs_real tol = ctrl_tolerance_->to<mrs_real>();
134 
135  // get pointers to avoid excessive (long,long) lookups
136  mrs_real *yin_buffer = yin_buffer_realvec_.getData();
137  const mrs_natural yin_buffer_size = yin_buffer_realvec_.getSize();
138  // ASSUME: only one channel
139  mrs_real *input = in.getData();
140 
141 
142  mrs_real pitch = -1.0;
143 
144 // cout << "yin.getSize()=" << yin.getSize() << endl;
145 // cout << "tol=" << tol << endl;
146 
147  const mrs_real freq_max = ctrl_frequency_max_->to<mrs_real>();
148  const mrs_real freq_min = ctrl_frequency_min_->to<mrs_real>();
149 
150  // yes, low_sample comes from the highest pitch
151  mrs_natural low_sample = 4;
152  if (freq_max > 0) {
153  low_sample = (mrs_natural) (israte_ / freq_max);
154  }
155  mrs_natural high_sample = yin_buffer_size;
156  if (freq_min > 0) {
157  high_sample = (mrs_natural) (israte_ / freq_min);
158  }
159 
160  // Calculate the pitch with the Yin method
161  //mrs_natural c=0;
162  mrs_real cmndf = 0.; // cumulative mean normalized difference function
163 
164 
165  std::fill(yin_buffer, yin_buffer + yin_buffer_size, 0.0);
166  yin_buffer[0] = 1.;
167 
168  //for (mrs_natural tau=1; tau < yin_size_; tau++)
169  for (mrs_natural tau=1; tau < high_sample; tau++)
170  {
171  // d_t( tau )
172  for (mrs_natural j=0; j < yin_buffer_size; j++)
173  {
174  const mrs_real delta = input[j] - input[j+tau];
175  yin_buffer[tau] += delta * delta;
176  }
177  cmndf += yin_buffer[tau];
178  yin_buffer[tau] *= tau / cmndf;
179  if (tau > low_sample) {
180  const mrs_natural period = tau-3;
181  //if(tau > 4 && (yin_buffer_(c,period) < tol) &&
182  // (yin_buffer_(c,period) < yin_buffer_(c,period+1)))
183  if((yin_buffer[period] < tol) &&
184  (yin_buffer[period] < yin_buffer[period+1])) {
185  pitch = vec_quadint_min(&yin_buffer_realvec_,period,1);
186  break;
187  }
188  }
189  }
190  if (pitch < 0) {
191  pitch = vec_quadint_min(&yin_buffer_realvec_,
192  vec_min_elem(&yin_buffer_realvec_),1);
193  }
194 
195  if (pitch !=0)
196  out(0,0) = ctrl_osrate_/pitch;
197  else
198  out(0,0) = 0.0;
199 
200 }
