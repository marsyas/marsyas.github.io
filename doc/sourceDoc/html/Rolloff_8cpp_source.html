72 void
73 Rolloff::myProcess(realvec& in, realvec& out)
74 {
75  mrs_natural o,t;
76  // computer rolloff of observations for each time sample
77  for (t = 0; t < inSamples_; t++)
78  {
79  sum_ = 0.0;
80  sumWindow_.setval(0.0);
81  for (o=0; o < inObservations_; o++)
82  {
83  sum_ += in(o,t);
84  sumWindow_(o) = sum_;
85  }
86  total_ = sumWindow_(inObservations_-1);
87  for (o=inObservations_-1; o>1; o--)
88  {
89  if (sumWindow_(o) < perc_ *total_)
90  {
91  out(0,t) = (mrs_real)o / inObservations_;
92  return;
93  }
94  }
95  out(0,t) = 1.0; // default
96  }
