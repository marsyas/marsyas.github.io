256 summaryStatistics ClassificationReport::computeSummaryStatistics(const realvec& mat)
257 {
258  MRSASSERT(mat.getCols()==mat.getRows());
259 
260  summaryStatistics stats;
261 
262  mrs_natural size = mat.getCols();
263 
264  vector<mrs_natural>rowSums(size);
265  for(int ii=0; ii<size; ++ii) rowSums[ii] = 0;
266  vector<mrs_natural>colSums(size);
267  for(int ii=0; ii<size; ++ii) colSums[ii] = 0;
268  mrs_natural diagonalSum = 0;
269 
270  mrs_natural instanceCount = 0;
271  for(mrs_natural row=0; row<size; row++)
272  {
273  for(mrs_natural col=0; col<size; col++)
274  {
275  mrs_natural num = (mrs_natural)mat(row,col);
276  instanceCount += num;
277 
278  rowSums[row] += num;
279  colSums[col] += num;
280 
281  if(row==col)
282  diagonalSum += num;
283  }
284  }
285  //printf("row1 sum:%d\n",rowSums[0]);
286  //printf("row2 sum:%d\n",rowSums[1]);
287  //printf("col1 sum:%d\n",colSums[0]);
288  //printf("col2 sum:%d\n",colSums[1]);
289  //printf("diagonal sum:%d\n",diagonalSum);
290  //printf("instanceCount:%d\n",instanceCount);
291 
292  mrs_natural N = instanceCount;
293  mrs_natural N2 = (N*N);
294  stats.instances = instanceCount;
295  stats.correctInstances = diagonalSum;
296 
297  mrs_natural sum = 0;
298  for(mrs_natural ii=0; ii<size; ++ii)
299  {
300  sum += (rowSums[ii] * colSums[ii]);
301  }
302  mrs_real PE = (mrs_real)sum / (mrs_real)N2;
303  mrs_real PA = (mrs_real)diagonalSum / (mrs_real)N;
304  stats.kappa = (PA - PE) / (1.0 - PE);
305 
306  mrs_natural not_diagonal_sum = instanceCount - diagonalSum;
307  mrs_real MeanAbsoluteError = (mrs_real)not_diagonal_sum / (mrs_real)instanceCount;
308  //printf("MeanAbsoluteError:%f\n",MeanAbsoluteError);
309  stats.meanAbsoluteError = MeanAbsoluteError;
310 
311  mrs_real RootMeanSquaredError = sqrt(MeanAbsoluteError);
312  //printf("RootMeanSquaredError:%f\n",RootMeanSquaredError);
313  stats.rootMeanSquaredError = RootMeanSquaredError;
314 
315  mrs_real RelativeAbsoluteError = (MeanAbsoluteError / 0.5) * 100.0;
316  //printf("RelativeAbsoluteError:%f%%\n",RelativeAbsoluteError);
317  stats.relativeAbsoluteError = RelativeAbsoluteError;
318

320  //printf("RootRelativeSquaredError:%f%%\n",RootRelativeSquaredError);
321  stats.rootRelativeSquaredError = RootRelativeSquaredError;
322 
323  return stats;
324 }//computeSummaryStatistics
