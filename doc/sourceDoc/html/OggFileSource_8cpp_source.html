132 void
133 OggFileSource::getHeader(mrs_string filename)
134 {
135  // if we have a file open already, close it
136  closeFile();
137  mrs_real duration = 0;
138  mrs_real israte = 22050.0;
139  mrs_natural nChannels = 1;
140  mrs_natural size = 0;
141  hasData_ = false;
142  mrs_natural bitRate = 128*1024;
143 
144  FILE* fp = fopen(filename.c_str(), "rb");
145  vf = new OggVorbis_File;
146 
147  /* Using ov_open_callbacks because ov_open fails under windows. */
148  if(fp && ov_open_callbacks(fp, vf, NULL, 0, OV_CALLBACKS_DEFAULT) == 0)
149  {
150  vi=ov_info(vf,-1);
151  size = ov_pcm_total(vf,-1);
152  duration = ov_time_total(vf,-1);
153  nChannels = vi->channels;
154  israte = vi->rate;
155  hasData_ = true;
156  bitRate = ov_bitrate(vf, -1);
157  }
158  else
159  {
160  (void) filename; // in case the macro is not expanded
161  MRSWARN(filename + " does not appear to be an Ogg bitstream.");
162  }
163  setctrl("mrs_natural/nChannels", nChannels);
164  setctrl("mrs_real/israte", israte);
165  setctrl("mrs_natural/size", size);
166  setctrl("mrs_bool/hasData", hasData_);
167  setctrl("mrs_natural/bitRate", bitRate);
168  updControl("mrs_real/duration", duration);
169 }
