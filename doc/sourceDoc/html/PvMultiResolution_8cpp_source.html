154  }
155  }

157  {
158 
159

161  for (t = 0; t < inSamples_; t++)
162  {
163  out(o,t) = in(o, t);
164  }
165 
166  /* use long window for frequencies lower than approx. 2000 Hz
167  and short window for higher frequencies */
168  for (o=inObservations_/2; o < inObservations_/2 + 200; o++)
169  for (t = 0; t < inSamples_; t++)

172  }
173

175  for (t = 0; t < inSamples_; t++)
176  {
177  out(2*o, t) = 2 * out(2*o,t);
178  }
179  }
180 
181  else if (mode == "transient_switch")
182  {
183 
184  // short window
185  for (o=0; o < inObservations_/2; o++)
186  for (t = 0; t < inSamples_; t++)
187  {
188  out(o,t) = in(o, t);
189  }
190 
191

193  for (o=0; o < onObservations_/2; o++)
194  for (t = 0; t < inSamples_; t++)
