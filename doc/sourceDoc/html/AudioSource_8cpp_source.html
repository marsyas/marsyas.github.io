90 
91  if (getctrl("mrs_bool/initAudio")->to<mrs_bool>())

96 
97  const bool do_resize_buffer = true;
98  reformatBuffer(source_block_size, dest_block_size, channel_count, realtime, do_resize_buffer);
99 
100  shared.sample_rate = sample_rate;
101  shared.channel_count = channel_count;
102  shared.overrun = false;
103 
104  isInitialized_ = true;
105 
106  //update bufferSize control which may have been changed
107  //by RtAudio (see RtAudio documentation)
108  setctrl("mrs_natural/bufferSize", (mrs_natural) source_block_size);
109 
110  setctrl("mrs_bool/initAudio", false);
111  }
112  else if (isInitialized_)

114  const bool do_not_resize_buffer = false;
115 
116  if (source_block_size != old_source_block_size_
117  || sample_rate != shared.sample_rate
118  || (realtime != (shared.watermark == 0))

120  dest_block_size,

122  realtime,
123  do_not_resize_buffer) )
124  {

126  // stop processing until re-initialized;
127  stop();
128  isInitialized_ = false;
129  }
130  }
131 
132  old_source_block_size_ = source_block_size;
133  old_dest_block_size_ = dest_block_size;
134 }

141  mrs_natural channel_count,

143 )

145  //marsyas represents audio data as float numbers

152  audio_->closeStream();
153  }
154 
155  RtAudio::StreamParameters source_params;

157  source_params.nChannels = channel_count;

159 
160  RtAudio::StreamOptions options;
161  options.streamName = "Marsyas";
162  options.numberOfBuffers = 0; // Use default.
163  options.flags = RTAUDIO_SCHEDULE_REALTIME;

169 
170  // Suppress useless warnings when both an AudioSource and
171  // an AudioSink are being opened using ALSA:
172  audio_->showWarnings(false);
173 
174  try
175  {
176  unsigned int uint_block_size = *block_size;
177  audio_->openStream(NULL, &source_params, source_format, sample_rate,
178  &uint_block_size, &recordCallback, (void *)&shared, &options);
179  *block_size = uint_block_size;

181  catch (RtError& e)

232 bool AudioSource::reformatBuffer(mrs_natural source_block_size,
233  mrs_natural dest_block_size,
234  mrs_natural channel_count,
235  bool realtime, bool resize)
236 {
237  mrs_natural new_capacity = source_block_size + dest_block_size + 1;
238  if (!realtime)
239  new_capacity = std::max( new_capacity * 4, (mrs_natural) 2000 );
240 
241  if (resize)
242  {
243  assert(stopped_);
244  mrs_natural size = new_capacity * 2;
245  if (size != shared.buffer.samples() || channel_count != shared.buffer.observations())
246  {
247  bool do_clear_buffer = true;
248  shared.buffer.resize(channel_count, size, new_capacity, do_clear_buffer);
249  }
250  else
251  {
252  shared.buffer.set_capacity(new_capacity);
253  }
254  shared.watermark = realtime ? 0 : new_capacity / 2;
255  }
256  else
257  {
258  if (channel_count != shared.buffer.observations()
259  || new_capacity > shared.buffer.samples())
260  {
261  MRSERR("AudioSource: Can not set requested buffer capacity or channel count without"
262  " resizing the buffer!");
263  return false;
264  }
265 
266  //cout << "Changing capacity: " << new_capacity << "/" << shared.buffer.samples() << endl;
267  mrs_natural new_watermark = realtime ? 0 : new_capacity / 2;;
268  if (new_capacity > shared.buffer.capacity())
269  {
270  // First increase capacity, then watermark.
271  shared.buffer.set_capacity(new_capacity);
272  shared.watermark = new_watermark;
273  }
274  else
275  {
276  // First decrease watermark, then capacity.
277  shared.watermark = new_watermark;
278  shared.buffer.set_capacity(new_capacity);
279  }
280  }
281 
282  return true;
283 }

285 int
286 AudioSource::recordCallback(void *outputBuffer, void *inputBuffer,
287  unsigned int nFrames,
288  double streamTime, unsigned int status,
289  void *userData)
290 {
291  (void) outputBuffer;
292  (void) streamTime;
293  (void) status;
294 
295  mrs_real* data = (mrs_real*)inputBuffer;
296  InputData &shared = *((InputData*) userData);
297  unsigned int nChannels = shared.channel_count;
298 
299  if (shared.overrun)
300  shared.overrun = shared.buffer.write_capacity() <= shared.watermark;
301 
302  if (!shared.overrun)
303  {
304  // Limit scope of realvec_queue_producer!
305  realvec_queue_producer producer( shared.buffer, nFrames );
306 
307  if (producer.capacity() == nFrames)
308  {
309  for (unsigned int t=0; t < nFrames; t++)
310  {
311  for (unsigned int ch=0; ch < nChannels; ch++) {
312  producer(ch, t) = data[nChannels*t+ch];
313  }
314  }
315  }
316  else
317  {
318  shared.overrun = true;
319  MRSWARN("AudioSource: buffer overrun!");
320  }
321  }
322 
323  shared.mutex.lock();
324  shared.condition.notify_all();
325  shared.mutex.unlock();
326 
327  return 0;
328 }

331 AudioSource::myProcess(realvec& in, realvec& out)
332 {
333  (void) in;
334 
335  //check if RtAudio is initialized
336  if (!isInitialized_)
337  {
338  return;
339  }
340 
341  //check MUTE
342  if(ctrl_mute_->isTrue())
343  {
344  return;
345  }
346 
347  //assure that RtAudio thread is running
348  //(this may be needed by if an explicit call to start()
349  //is not done before ticking or calling process() )
350  if ( stopped_ )
351  {
352  start();
353  }
354 
355  realvec_queue_consumer consumer(shared.buffer, onSamples_);
356 
357  if ((mrs_natural) consumer.capacity() < onSamples_)
358  {
359  std::unique_lock<std::mutex> locker(shared.mutex);
360 
361  shared.condition.wait (
362  locker,
363  [&consumer, this]()
364  {
365  bool ok = consumer.reserve(onSamples_);
366  if (shared.watermark > 0)
367  ok = ok && shared.buffer.read_capacity() >= shared.watermark;
368  return ok;
369  }
370  );
371 
372  locker.unlock();
373  }
374 
375  MRSASSERT((mrs_natural) consumer.capacity() == onSamples_);
376 
377  for (mrs_natural t=0; t < onSamples_; t++)
378  {
379  for (mrs_natural o=0; o < onObservations_; o++)
380  {
381  out(o,t) = consumer(o,t);
382  }
383  }
384 }
