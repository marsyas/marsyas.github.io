219 void
220 Collection::shuffle()
221 {
222  // Use a Fisher-Yates shuffle
223  // http://en.wikipedia.org/wiki/Fisher-Yates_shuffle
224  mrs_natural n = (mrs_natural)collectionList_.size();
225  while (n > 1)
226  {
227  // Generate a random index in the range [0, n).
228  mrs_natural k = (mrs_natural)(n * ((mrs_real)rand() / ((mrs_real)(RAND_MAX) + (mrs_real)1)));
229 
230  n--;
231  swap(collectionList_[n], collectionList_[k]);
232  if (hasLabels_)
233  swap(labelList_[n], labelList_[k]);
234  }
235 }
