130 void OscTransmitter::myUpdate( MarControlPtr handler )
131 {
132  if (handler.isInvalid())
133  return;
134 
135  auto subscribers_it = m_subscribers.find(handler());
136  if (subscribers_it == m_subscribers.end())
137  return;
138 
139  subscription & slot = subscribers_it->second;
140  if (slot.empty())
141  return;
142 
143  osc::OutboundPacketStream packet( m_buffer, m_buffer_size );
144  try
145  {
146  packet << osc::BeginMessage( slot.path.c_str() );
147  if (handler->hasType<bool>())
148  packet << handler->to<bool>();
149  else if (handler->hasType<mrs_natural>())
150  packet << ((int) handler->to<mrs_natural>());
151  else if (handler->hasType<mrs_real>())
152  packet << ((float) handler->to<mrs_real>());
153  else if (handler->hasType<mrs_string>())
154  packet << handler->to<mrs_string>().c_str();
155  else
156  throw std::runtime_error("Unsupported control type.");
157  packet << osc::EndMessage;
158  }
159  catch ( std::exception & e )
160  {
161  MRSWARN("OSC sender: " << e.what());
162  }
163 
164  for( OscSubscriber * subscriber : slot.subscribers )
165  {
166  subscriber->process(packet.Data(), packet.Size());
167  }
168 }
169 
170 }
