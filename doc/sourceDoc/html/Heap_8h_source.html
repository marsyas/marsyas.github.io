106  Type* pop() {
107  // on empty heap throw a const char* exception, specified in the contract
108  if (first==NULL) { throw "Heap::pop() empty heap exception."; }
109  // save top data
110  Type* data = first->data;
111  // if it's the root then clear the heap
112  if (is_root(last)) {
113  delete(last); first=NULL; last=NULL; node_count=0;
114  }
115  else {
116  // swap last element data into top position
117  first->data = last->data;
118  // extricate the node from its parent
119  if (is_lchild(last)) { last->parent->lchild=NULL; }
120  else { last->parent->rchild=NULL; }
121  // remove the last node and update the pointer to the new last
122  last=last->prev; delete(last->next); last->next=NULL;
123  // bubble down
124  Node* f = first;
125  while (1) {
126  Node* c = f->lchild;
127  // if lchild of c==NULL then c cannot be bubbled down further
128  if (c==NULL) { break; }
129  // make f point to the smallest of c's children
130  if (f->rchild!=NULL && (cmp(f->rchild->data,c->data))) { c=f->rchild; }
131  // use the template required Comparator to compare if the
132  // smallest child of c is less, if not leave
133  if (cmp(f->data,c->data)) { break; }
134  // swap data
135  Type* sw=c->data; c->data=f->data; f->data=sw;
136  // track the bubbling node
137  f=c;
138  }
139  // update node_count ensuring it never goes below 0
140  node_count = (node_count>0) ? node_count-1 : 0;
141  }
142  return data;
143  };

146  // could throw an exception here...hmm
147  if (data==NULL) { return; }
148  node_count++;
149 
150  Node* n = new Node(node_count,data);
151  if (first==NULL) {
152  first=n; last=n;
153  }
154  else {
155  // add node to end of list
156  last->next=n; n->prev=last;
157  // insert node into tree and update parent and parent_children
158  if (is_root(last)) {
159  n->parent=last;
160  last->lchild = n;
161  }
162  else if (is_lchild(last)) {
163  n->parent=last->parent;
164  last->parent->rchild=n;
165  }
166  else {
167  n->parent=last->parent->next;
168  last->parent->next->lchild=n;
169  }
170  last=n;
171  // bubble up if needed
172  while (!is_root(n) && cmp(n->data,n->parent->data)) {
173  Type* sw=n->data; n->data=n->parent->data; n->parent->data=sw;
174  n = n->parent;
175  }
176  }
