155 void
156 Accumulator::myProcess(realvec& in, realvec& out)
157 {
158  mrs_natural o,c,t;
159 
160  if (!marsystems_.size())
161  {
162  out = in;
163  return;
164  }
165 
166  if (ctrl_mode_->to<mrs_string>() == "explicitFlush")
167  {
168  mrs_natural timesCount = keptOnSamples_/childOnSamples_;
169 
170  while ((!ctrl_flush_->to<mrs_bool>() && timesCount < ctrl_maxTimes_->to<mrs_natural>())
171  || timesCount < ctrl_minTimes_->to<mrs_natural>()
172  || timesCount <= ctrl_timesToKeep_->to<mrs_natural>())
173  {
174  // child MarSystem should have a control linked to the Accumulator flush control
175  // so it can signal the end of this loop (e.g. when an onset is detected or some other similar event)
176  marsystems_[0]->process(in, childOut_);
177 
178  //accumulate output from child process()into temp buffer
179  for (o=0; o < onObservations_; o++)
180  {
181  for (t = 0; t < childOnSamples_; t++)
182  {
183  tout_(o, t + timesCount * childOnSamples_) = childOut_(o,t);
184  }
185  }
186  timesCount++;
187  }
188 
189 #ifdef MARSYAS_LOG_DIAGNOSTICS
190  if (!ctrl_flush_->to<mrs_bool>())
191  {
192  MRWDIAG("Accumulator::myProcess() - maxTimes reached without a flush event!");
193  }
194 #endif
195 
196  //adjust output number of samples dynamically (this calls update()!!)
197  //to the number of accumulated samples (minus the ones to keep for next time)
198  ctrl_nTimes_->setValue(timesCount - ctrl_timesToKeep_->to<mrs_natural>());
199 
200  keptOnSamples_ = ctrl_timesToKeep_->to<mrs_natural>() * childOnSamples_;
201 
202  //copy data in tmp buffer to the output
203  for (o=0; o < onObservations_; ++o)
204  {
205  for (t = 0; t < ctrl_onSamples_->to<mrs_natural>(); ++t)
206  {
207  out(o,t) = tout_(o,t);
208  }
209  }
210 
211 
212  //store samples to keep into the beginning of the temp buffer
213  //for next call to process()
214  for (t = 0; t < keptOnSamples_; ++t)
215  {
216  for (o=0; o < onObservations_; ++o)
217  {
218  tout_(o,t) = tout_(o, t + ctrl_onSamples_->to<mrs_natural>());
219  }
220  }
221 
222  //reset flush flag
223  ctrl_flush_->setValue(false);
224  }
225  else if (ctrl_mode_->to<mrs_string>() == "countTicks")
226  {
227  //nTimes_ = ctrl_nTimes_->to<mrs_natural>();
228  ctrl_flush_->setValue(false);
229  for (c = 0; c < nTimes_; ++c)
230  {
231 
232  marsystems_[0]->process(in, childOut_);
233 
234  for (o=0; o < onObservations_; o++)
235  {
236  for (t = 0; t < childOnSamples_; t++)
237  {
238  out(o, t + c * childOnSamples_) = childOut_(o,t);
239  }
240  }
241  }
242  ctrl_flush_->setValue(true);
243  }
244

246  //MATLAB_PUT(out, "Accum_out");
247  //MATLAB_EVAL("segmentData = [segmentData, Accum_out];");
248 }
