97 void
98 LPC::autocorrelationWarped(const realvec& in, realvec& r, mrs_real& pitch, mrs_real lambda)
99 {
100  //*Based on the code from: http://www.musicdsp.org/showone.php?id=137
101 
102  //find the order-P autocorrelation array, R, for the sequence x
103  //of length L and using a warping factor of lambda
104 
105  mrs_real* x = in.getData();
106  mrs_natural L = in.getSize();
107  mrs_real* R = r.getData();
108  mrs_natural P = in.getSize()/2;//order_;
109 
110  mrs_real* dl = new mrs_real[L];
111  mrs_real* Rt = new mrs_real[L];
112  mrs_real r1,r2,r1t;
113  R[0]=0;
114  Rt[0]=0;
115  r1=0;
116  r2=0;
117  r1t=0;
118  for(mrs_natural k=0; k<L; k++)
119  {
120  Rt[0]+=x[k]*x[k];
121 
122  dl[k]=r1-lambda*(x[k]-r2);
123  r1 = x[k];
124  r2 = dl[k];
125  }
126  for(mrs_natural i=1; i<=P; ++i)
127  {
128  Rt[i]=0;
129  r1=0;
130  r2=0;
131  for(mrs_natural k=0; k<L; k++)
132  {
133  Rt[i]+=dl[k]*x[k];
134 
135  r1t = dl[k];
136  dl[k]=r1-lambda*(r1t-r2);
137  r1 = r1t;
138  r2 = dl[k];
139  }
140  }
141 
142  for(long i=0; i<=P; ++i)
143  R[i]=Rt[i]/in.getSize(); // [ML] change /
144 
145  delete[] dl;
146  delete[] Rt;
147 
148  //----------------------------------------------------
149  // estimate pitch
150  //----------------------------------------------------
151  mrs_real temp = r(0);
152  //set peak searching start point to 2% of total window size [?]
153  mrs_real j = (mrs_real)in.getSize() * 0.02;
154  //detect first local minimum...
155  while (r((mrs_natural)j) < temp && j < in.getSize()/2)
156  {
157  temp = r((mrs_natural)j);
158  j++;
159  }
160  //... and then from there, detect higher peak => period estimate!
161  temp = 0.0;
162  for (mrs_natural i=(mrs_natural)j; i < in.getSize() * 0.5; ++i)
163  {
164  if (r(i) > temp)
165  {
166  j = i;
167  temp = r(i);
168  }
169  }
170 
171  // This code is from the original Marsyas0.1 AutoCorrelation class
172  //this seems like some sort of Narrowed Autocorrelation (NAC)... [?][!]
173  //however, it does not seem to fit the NAC definition...
174  //so, what is this for??
175  // mrs_real norm = 1.0 / (mrs_real)in.getSize();
176  // mrs_natural k = in.getSize()/2;
177  // for (mrs_natural i=0; i < in.getSize()/2; ++i)
178  // r(i) *= (k-i) * norm;
179 
180  //if autocorr peak not very prominent => not a good period estimate
181  //so discard it...
182  if ((r((mrs_natural)j) / r(0)) < 0.4) j=0;
183  //avoid detection of too high fundamental freqs (e.g. harmonics)?
184  if (j > in.getSize()/4) j = 0;
185 
186  //pitch gets in fact the period (i.e. autocorr lag)
187  //measured in time samples... maybe this should be converted
188  //to a more convenient representation (e.g. pitch in Hz).
189  //Such a conversion would have to depend on the warp factor lambda... [!]
190  pitch = (mrs_real)j;
191 
192  //MATLAB engine stuff - for testing and validation purposes only!
193 #ifdef _MATLAB_LPC_
194  MATLAB_PUT(pitch, "pitch");
195  MATLAB_EVAL("LPC_pitch = [LPC_pitch pitch];");
196 #endif
197 }

200 void
201 LPC::LevinsonDurbin(const realvec& r, realvec& a, realvec& kVec, mrs_real& e)
202 {
203  //*Based on the code from: http://www.musicdsp.org/showone.php?id=137
204 
205  mrs_natural P = order_;
206  mrs_real* R = r.getData();
207  mrs_real* A = a.getData();
208  mrs_real* K = kVec.getData();
209  e = 0.0; //prediction error;
210 
211  mrs_real Am1[62];
212 
213  if(R[0]==0.0)
214  {
215  for(mrs_natural i=1; i<=P; ++i)
216  {
217  K[i]=0.0;
218  A[i]=0.0;
219  }
220  }
221  else
222  {
223  mrs_real km,Em1,Em;
224  mrs_natural k,s,m;
225 
226  for (k=0; k<=P; k++) {
227  A[k]=0;
228  Am1[k]=0;
229  }
230 
231  A[0]=1;
232  Am1[0]=1;
233  km=0;
234  Em1=R[0];
235 
236  for (m=1; m<=P; m++) //m=2:N+1
237  {
238  mrs_real err=0.0f; //err = 0;
239 
240  for (k=1; k<=m-1; k++) //for k=2:m-1
241  err += Am1[k]*R[m-k]; // err = err + am1(k)*R(m-k+1);
242 
243  km = (R[m]-err)/Em1; //km=(R(m)-err)/Em1;
244  K[m-1] = -km;
245  A[m]=km; //am(m)=km;
246 
247  for (k=1; k<=m-1; k++) //for k=2:m-1
248  A[k]=Am1[k]-km*Am1[m-k]; // am(k)=am1(k)-km*am1(m-k+1);
249 
250  Em=(1-km*km)*Em1; //Em=(1-km*km)*Em1;
251 
252  for(s=0; s<=P; s++) //for s=1:N+1
253  Am1[s] = A[s]; // am1(s) = am(s)
254 
255  Em1 = Em; //Em1 = Em;
256  //e = Em1; //prediction error
257  e = Em1*Em1; //RMS prediction error
258  }
259  e = sqrt(e/(mrs_real)P); //RMS prediction error
260  }

263 mrs_real
264 LPC::predictionError(const realvec& data, const realvec& coeffs)
265 {
266  mrs_natural i, j;
267  mrs_real power = 0.0;
268  mrs_real error, tmp;
269 
270  //load delay line with input data...
271  for (i=0; i< order_; ++i)
272  {
273  Zs_(i) = data(order_-i-1);
274  }
275  //apply LPC filter and estimate RMS of the error (=~ LPC Gain)
276  mrs_natural count = 0;
277  for (i=order_; i<(mrs_natural)data.getSize() ; ++i)
278  {
279  tmp = 0.0;
280  for (j=0; j< order_; j++)
281  tmp += Zs_(j) * coeffs(j);
282  for (j=order_-1; j>0; j--)
283  Zs_(j) = Zs_(j-1);
284 
285  Zs_(0) = data(i);
286  error = data(i) - tmp;
287  power += error * error;
288  count ++;
289  }
290  return sqrt(power/(mrs_real)count);//=~ RMS LPC Gain
291 }

337 mrs_real
338 LPC::SPcorXpc (mrs_real * rxx, mrs_real * pc, mrs_natural Np)
339 {
340  mrs_natural i, j, k;
341  mrs_real rc, tp;
342  mrs_real perr, t, sum;
343 
344  perr = rxx[0];
345 
346 
347  for (k = 0; k < Np; ++k) {
348 
349  /* Calculate the next reflection coefficient */
350  sum = rxx[k+1];
351  for (i = 0; i < k; ++i)
352  sum = sum - rxx[k-i] * pc[i];
353 
354  /* Check for zero prediction error */
355  if (perr == 0.0 && sum == 0.0)
356  rc = 0.0;
357  else
358  rc = (-sum / perr);
359 
360  /* Calculate the error (equivalent to perr = perr * (1-rc^2))
361  A change in sign of perr means that rc (reflection coefficient for stage k)
362  has a magnitude greater than unity (corresponding to an unstable synthesis
363  filter)
364  */
365  t = perr + rc * sum;
366 
367  perr = t;
368 
369  /* Convert from reflection coefficients to predictor coefficients */
370  pc[k] = -rc;
371  for (i = 0, j = k - 1; i < j; ++i, --j) {
372  tp = pc[i] + rc * pc[j];
373  pc[j] = pc[j] + rc * pc[i];
374  pc[i] = tp;
375  }
376  if (i == j)
377  pc[i] = pc[i] + rc * pc[i];
378  }
379 
380  return perr;
381 }
