154  template<typename T>
155  T value( const std::string & long_name ) const
156  {
157  std::map<std::string, Option*>::const_iterator it = m_options.find(long_name);
158  if (it == m_options.end())
159  return T();
160  OptionT<T> *typed_option = option_cast<T>(it->second);
161  if (!typed_option)
162  return T();
163  return typed_option->is_set ? typed_option->value : typed_option->default_value;
164  }
