97 PeakRatio::myProcess(realvec& in, realvec& out)
98 {

100 
101 
102 
103  mrs_real max_ = -1.0 * DBL_MAX;
104  mrs_real min_ = DBL_MAX;
105  mrs_real avg_ = 0.0;
106 
107 
108  for (t=0; t < inSamples_; t++)
109  {
110  for (o=0; o < inObservations_; o++)
111  {
112  out(o,t)=in(o,t); // copy everything from input to output
113 
114  if (in(o,t) > max_)
115  {
116  max_ = in(o,t);
117  }
118  if (in(o,t) < min_)
119  {
120  min_ = in(o,t);
121  }
122  avg_=avg_+in(o,t);
123 
124  }
125  avg_=avg_/(inSamples_*inObservations_);
126  maxima_(t)=max_;
127  minima_(t)=min_;
128  }
129  mrs_real res1=0.0;
130  mrs_real res2=0.0;
131 
132  for (t=0; t < inSamples_; t++)
133  {
134  //compute ratios
135  if(minima_(t)!=0.0) res1=maxima_(t)/minima_(t);
136  if(minima_(t)!=0.0) res2=maxima_(t)/avg_;
137 
138  //add the ratios to the output
139  out(onObservations_-1,t)=res1;
140  out(onObservations_-2,t)=res2;
