95 mrs_real& peakView::operator()(const mrs_natural peakIndex, const pkParameter param, const mrs_natural frame, const mrs_natural group)
96 {
97  //if peakIndex is a global index (i.e. independent of the group)
98  //just return the corresponding parameter value
99  if(group == -1)
100  return vec_(peakIndex + param * frameMaxNumPeaks_, frame);
101  else
102  {
103  //if peakIndex is an index of a peak belonging to the passed group
104  //we have to convert it to the absolute (i.e. unrelated to any group) index first
105  mrs_natural gp = 0;
106  for(mrs_natural p = 0; p < this->getFrameNumPeaks(frame); ++p)
107  {
108  if(vec_(p + pkGroup * frameMaxNumPeaks_, frame) == group)
109  {
110  if(peakIndex == gp)
111  return vec_(p + param * frameMaxNumPeaks_, frame);
112  gp++;
113  }
114  }
115  //if the passed peakIndex does not exist in the passed group
116  //return an "invalid" value and issue an error...
117  MRSERR("peakView::operator() - peakIndex " << peakIndex << " not found in passed group " << group);
118  return vec_(peakIndex + param * frameMaxNumPeaks_, frame);
119  }
120 }
