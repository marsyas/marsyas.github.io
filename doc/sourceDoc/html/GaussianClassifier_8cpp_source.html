83  mrs_string mode = getctrl("mrs_string/mode")->to<mrs_string>();
84 
85  if (active_) {
86  // Don't bother resetting anything if this system is inactive.
87  // This allows us to load an entire .mpl file without incorrectly
88  // resetting the means and covars a bunch of times.
89 
90  if ((nClasses != mrows) || (inObservations_ != mcols))
91  {
92  MarControlAccessor acc_means(ctrl_means_);
93  MarControlAccessor acc_covars(ctrl_covars_);
94

96  realvec& covars = acc_covars.to<mrs_realvec>();
97 
98  means.create(nClasses, inObservations_);
99  covars.create(nClasses, inObservations_);
100  labelSizes_.create(nClasses);
101  }
102 
103  if ((prev_mode_ == "train") && (mode == "predict"))
104  {
105 
106  MarControlAccessor acc_means(ctrl_means_);
107  MarControlAccessor acc_covars(ctrl_covars_);
108 
109  realvec& means = acc_means.to<mrs_realvec>();
110  realvec& covars = acc_covars.to<mrs_realvec>();
111 
112 
113  for (int l=0; l < nClasses; l++)
114  {
115  for (o=0; o < inObservations_; o++)
116  {
117  means(l,o) = means(l,o) / labelSizes_(l);
118  covars(l,o) = covars(l,o) / labelSizes_(l);
119  covars(l, o) = covars(l,o) - (means(l,o) * means(l,o));

131 void
132 GaussianClassifier::myProcess(realvec& in, realvec& out)
133 {
134  mrs_natural o,t;
135  mrs_real v;
136  mrs_string mode = ctrl_mode_->to<mrs_string>();
137  mrs_natural nClasses = ctrl_nClasses_->to<mrs_natural>();
138 
139  mrs_natural l;
140  mrs_natural prediction = 0;
141  mrs_real label;
142 
143  mrs_real diff;
144  mrs_real sq_sum=0.0;
145 
146  MarControlAccessor acc_means(ctrl_means_);
147  MarControlAccessor acc_covars(ctrl_covars_);
148  realvec& means = acc_means.to<mrs_realvec>();
149  realvec& covars = acc_covars.to<mrs_realvec>();
150 
151 
152 
153  // reset
154  if ((prev_mode_ == "predict") && (mode == "train"))
155  {
156  means.setval(0.0);
157  covars.setval(0.0);
158  labelSizes_.setval(0.0);
159  }
160 
161  if (mode == "train")
162  {
163  for (t = 0; t < inSamples_; t++)
164  {
165  label = in(inObservations_-1, t);
166  if(label>=0) //unlabeled data (i.e. label = -1) should be ignored
167  {
168  for (o=0; o < inObservations_-1; o++)
169  {
170  v = in(o,t);
171  means((mrs_natural)label,o) = means((mrs_natural)label,o) + v;
172  covars((mrs_natural)label,o) = covars((mrs_natural)label,o) + v*v;
173  out(0,t) = (mrs_real)label;
174  out(1,t) = (mrs_real)label;
175  for (int j=0; j < nClasses; j++)
176  {
177  out(j,t) = (mrs_real)0;
178  if (j == label)
179  out(j,t) = (mrs_real)1;
180  }
181  }
182  labelSizes_((mrs_natural)label) = labelSizes_((mrs_natural)label) + 1;
183  }
184  }
185  }
186 
187  if (mode == "predict")
188  {
189 
190  mrs_real min = MAXREAL;
191 
192  for (t = 0; t < inSamples_; t++)
193  {
194  label = in(inObservations_-1, t);
195 
196  for (l=0; l < nClasses; l++)
197  {
198  sq_sum = 0.0;
199 
200  for (o=0; o < inObservations_-1; o++)
201  {
202  v = in(o,t);
203  diff = (v - means(l,o));
204  sq_sum += (diff * covars(l,o) * diff);
205  }
206 
207  if (sq_sum < min)
208  {
209  min = sq_sum;
210  prediction = l;
211  }
212 
213  // output the distances in the probability slots
214  out (2+l, t) = sq_sum;
215  }
216  out(0,t) = (mrs_real)prediction;
217  out(1,t) = (mrs_real)label;
218  }
219  }
220 
221  prev_mode_ = mode;
222 }
