74 ShiftInput::myUpdate(MarControlPtr sender)
75 {
76  (void) sender; //suppress warning of unused parameter(s)
77 
78  // Cache the window and hop size for use in myProcess().
79  winSize_ = ctrl_winSize_->to<mrs_natural>();
80  hopSize_ = ctrl_inSamples_->to<mrs_natural>();
81 
82  // If the hop size is smaller than the window size, we need buffering to
83  // handle the overlap between successive windows.
84  // When the hop size is larger than the window size, we don't have overlap
85  // and don't need buffering.
86  if (hopSize_ < winSize_)
87  {
88  outSavedData_.stretch(ctrl_inObservations_->to<mrs_natural>(), winSize_ - hopSize_);
89  if (hopSize_ == 0) {
90  addToStabilizingDelay_ = 0;
91  } else {
92  mrs_natural full = (mrs_natural) (((mrs_real) winSize_ / hopSize_));
93  addToStabilizingDelay_ = ((mrs_natural) ceil(((mrs_real) winSize_ / hopSize_)-full)) + full - 1;
94  //cout<<winSize_ <<"\t"<<hopSize_ <<"\t"<< (mrs_real) winSize_ / hopSize_<<"\t"<<full<<"\t"<<addToStabilizingDelay_<<endl;
95  }
96  } else {
97  addToStabilizingDelay_ = 0;
98  }
99 
100  // Update the output stream format.
101  ctrl_onSamples_->setValue(ctrl_winSize_, NOUPDATE);
102  ctrl_onObservations_->setValue(ctrl_inObservations_, NOUPDATE);
103  ctrl_osrate_->setValue(ctrl_israte_, NOUPDATE);
104 
105  // Update the onObsNames string.
106  ostringstream prefix_oss;
107  prefix_oss << "HopSize" << hopSize_ << "_WinSize" << winSize_ << "_";
108  mrs_string onObsNames = obsNamesAddPrefix(ctrl_inObsNames_->to<mrs_string>(), prefix_oss.str());
109  ctrl_onObsNames_->setValue(onObsNames, NOUPDATE);

112 void
113 ShiftInput::myProcess(realvec& in, realvec& out)
114 {
115  mrs_natural t,o;
116 
117 
118  for (o = 0; o<inObservations_; ++o)
119  {
120  if (hopSize_ < winSize_)
121  {
122  // When the hop size is smaller than the window size, we have to work
123  // with a buffer.
124 
125  // Check if we must clear the audio buffer (due to a reset call).
126  // TODO: shouldn't this be done in myUpdate?
127  if (ctrl_reset_->to<mrs_bool>())
128  {
129  outSavedData_.setval(0.0);
130  ctrl_reset_->setValue(false);
131  }
132 
133  // Check if requested a call for a partial clean of the buffer
134  // TODO: shouldn't this be done in myUpdate?
135  if(ctrl_clean_->to<mrs_bool>())
136  {
137  // round up with ceil()
138  mrs_natural lowCleanLimit = (mrs_natural) ceil(winSize_
139  * getctrl("mrs_real/lowCleanLimit")->to<mrs_real>());
140  mrs_natural highCleanLimit = (mrs_natural) (lowCleanLimit + (ceil( winSize_
141  * getctrl("mrs_real/highCleanLimit")->to<mrs_real>()
142  ) - lowCleanLimit));
143 
144  for (t = lowCleanLimit; t < highCleanLimit; t++)
145  outSavedData_(o, t) = 0.0;
146 
147  ctrl_clean_->setValue(false);
148  }
149 
150  // TODO: this can be done more efficiently with a circular buffer
151  // instead of shifting the buffer data all the time.
152 
153  // First part of output: copy the buffered data.
154  for (t = 0; t < winSize_ - hopSize_; t++)
155  {
156  out(o, t) = outSavedData_(o, t);
157  }
158  // Second part of output: copy the new data from the input.
159  for (t = winSize_ - hopSize_; t < winSize_; t++)
160  {
161  out(o, t) = in(o, t - (winSize_ - hopSize_));
162  }
163  // Store part of the output to the buffer for the next time.
164  for (t = 0; t < winSize_ - hopSize_; t++)
165  {
166  outSavedData_(o, t) = out(o, t + hopSize_);
167  }
168  }
169  else
170  {
171  // When the hop size is larger than the window size, we just need to
172  // copy the first onSamples_ samples and discard the rest.
173  for (t = 0; t < onSamples_; ++t)
174  {
175  out(o, t) = in(o, t);
176  }
177  }
178  }
