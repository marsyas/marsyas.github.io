375 mrs_natural
376 WavFileSource::getLinear8(realvec& slice)
377 {
378  mrs_natural t;
379  mrs_natural c = 0;
380  fseek(sfp_, pos_ * nChannels_ + sfp_begin_, SEEK_SET);
381 
382  samplesToRead_ = inSamples_ * nChannels_;
383  samplesRead_ = (mrs_natural)fread(cdata_, sizeof(unsigned char), samplesToRead_, sfp_);
384 
385 
386  if (samplesRead_ != samplesToRead_)
387  {
388  for (c=0; c < nChannels_; ++c)
389  for (t=0; t < inSamples_; t++)
390  {
391  slice(c,t) = 0.0;
392  }
393  samplesToWrite_ = samplesRead_ / nChannels_;
394  }
395  else
396  samplesToWrite_ = inSamples_;
397 
398  for (t=0; t < samplesToWrite_; t++)
399  {
400  for (c=0; c < nChannels_; ++c)
401  {
402  slice(c, t) = (mrs_real)-1.0 + (mrs_real) cdata_[nChannels_ * t + c] / 127;
403  }
404  }
405 
406  pos_ += samplesToWrite_;
407  return pos_;
408 }
