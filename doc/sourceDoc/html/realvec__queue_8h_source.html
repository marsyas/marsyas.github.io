164  mrs_natural write_capacity()
165  {
166  mrs_natural read_pos = m_read_position.load(memory_order_relaxed);
167  mrs_natural write_pos = m_write_position.load(memory_order_relaxed);
168  mrs_natural capacity = m_capacity.load(memory_order_relaxed);
169  mrs_natural size = samples();
170 
171  if (capacity > 0 )
172  --capacity; // Should never write the last sample.
173 
174  mrs_natural written;
175  if (write_pos >= read_pos)
176  written = write_pos - read_pos;
177  else
178  written = size - (read_pos - write_pos);
179 
180  if (written >= capacity)
181  return 0;
182  else
183  return capacity - written;
184  }
185 
189  mrs_natural read_capacity()
190  {
191  mrs_natural read_pos = m_read_position.load(memory_order_relaxed);
192  mrs_natural write_pos = m_write_position.load(memory_order_acquire);
193  mrs_natural available;
194  if (write_pos >= read_pos)
195  available = write_pos - read_pos;
196  else
197  available = samples() - (read_pos - write_pos);
198  return available;
199  }
