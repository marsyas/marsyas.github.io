114 void
115 Combinator::myProcess(realvec& in, realvec& out)
116 {
117  mrs_natural k,i,j;
118  mrs_real weight = 1;
119  mrs_bool useWeights = false;
120 
121  const Combinators_t combi = GetCombinatorIdx (ctrl_combinator_->to<mrs_string>());
122  const mrs_natural numInputs = ctrl_numInputs_->to<mrs_natural>();
123 
124  // initialize output buffer
125  if(combi == kAdd)
126  out.setval(0); //identity operator
127  if(combi == kMult)
128  out.setval(1); //identity operator
129  if(combi == kMax)
130  out.setval(-1.0*MAXREAL);
131  if(combi == kMin)
132  out.setval(MAXREAL);
133 
134  // sanity check
135  if ((inObservations_%numInputs) != 0)
136  {
137  MRSWARN("Combinator: dimension mismatch");
138  MRSASSERT(false);
139  out.setval(0);
140  return;
141  }
142 
143  if (ctrl_weights_->to<mrs_realvec>().getSize () == numInputs)
144  useWeights = true;
145 
147  for (k = 0; k < numInputs; k++)
148  {
149  weight = (useWeights) ? (ctrl_weights_->to<mrs_realvec>())(k) : weight;
150 
151  for (i = 0; i < onObservations_; i++)
152  {
153  // do the col-loop inside the switch to avoid calling the switch for every entry
154  // we might also do this with function pointers...
155  switch (combi)
156  {
157  case kAdd:
158  default:
159  {
160  for (j = 0; j < onSamples_; j++)
161  out(i,j) += weight * in(i+k*onObservations_,j);
162  break;
163  }
164  case kMult:
165  {
166  if (weight == 1.0) // avoid pow if possible
167  {
168  for (j = 0; j < onSamples_; j++)
169  out(i,j) *= in(i+k*onObservations_,j);
170  }
171  else
172  {
173  for (j = 0; j < onSamples_; j++)
174  out(i,j) *= pow (in(i+k*onObservations_,j), weight);
175  }
176  break;
177  }
178  case kMax:
179  {
180  for (j = 0; j < onSamples_; j++)
181  out(i,j) = max (out(i,j),in(i+k*onObservations_,j));
182  break;
183  }
184  case kMin:
185  {
186  for (j = 0; j < onSamples_; j++)
187  out(i,j) = min (out(i,j),in(i+k*onObservations_,j));
188  break;
189  }
190  }
191  }
192  }
193 }
