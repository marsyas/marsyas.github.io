125  {
126  const mrs_realvec & vector = m_control->to<mrs_realvec>();
127  m_rows = vector.getRows();
128  m_columns = vector.getCols();
129  m_value.create( m_rows, m_columns );
130  m_stage.clear( m_value );
131  }
132 
133  any value()
134  {
135  const_cast<stage<mrs_realvec>&>(m_stage).pop();
136  if (m_stage.has_front())
137  m_value = m_stage.front();
138  return any(m_value);
139  }
140 
141  void push()
142  {
143  const mrs_realvec & value = m_control->to<mrs_realvec>();
144  if (value.getRows() == m_rows && value.getCols() == m_columns)
145  {
146  mrs_realvec & stage_value = m_stage.back();
147  std::memcpy( stage_value.getData(),
148  value.getData(),
149  sizeof(mrs_real) * value.getSize() );
150  m_stage.push();
151  }
152  else
153  {
154  MRSERR("AtomicControlT<mrs_realvec>::push(): realvec format mismatch!");
155  }
156  }
